001package org.kohsuke.stapler;
002
003/**
004 * Argument to expressions that indicates this value is raw HTML
005 * and therefore should not be further escaped.
006 */
007public class RawHtmlArgument {
008    private final Object value;
009
010    public RawHtmlArgument(Object value) {
011        this.value = value;
012    }
013
014    public Object getValue() {
015        return value;
016    }
017
018    @Override
019    public String toString() {
020        return value==null?"null":value.toString();
021    }
022}
