028import org.apache.commons.jelly.JellyException;
029import org.apache.commons.jelly.JellyTagException;
030import org.apache.commons.jelly.Script;
031import org.apache.commons.jelly.TagLibrary;
032import org.apache.commons.jelly.XMLOutput;
033import org.apache.commons.jelly.impl.DefaultTagFactory;
034import org.apache.commons.jelly.impl.TagScript;
035import org.xml.sax.Attributes;
036
037import javax.servlet.http.HttpServletRequest;
038import java.util.HashSet;
039import java.util.Set;
040
041/**
042 * @author Kohsuke Kawaguchi
043 */
044public class StaplerTagLibrary extends TagLibrary {
045    public StaplerTagLibrary() {
046        registerTag("adjunct",AdjunctTag.class);
047        registerTag("bind",BindTag.class);
048        registerTag("compress",CompressTag.class);
049        registerTag("contentType",ContentTypeTag.class);
050        registerTag("copyStream",CopyStreamTag.class);
051        registerTag("doctype",DoctypeTag.class);
052        registerTag("findAncestor",FindAncestorTag.class);
053        registerTag("header",HeaderTag.class);   // deprecated. for compatibility
054        registerTag("addHeader",HeaderTag.class);
055        registerTag("setHeader",SetHeaderTag.class);
056        registerTag("include",IncludeTag.class);
057        registerTag("isUserInRole",IsUserInRoleTag.class);
058        registerTag("nbsp",NbspTag.class);
059        registerTag("out",OutTag.class);
060        registerTag("parentScope",ParentScopeTag.class);
061        registerTag("redirect",RedirectTag.class);
062        registerTag("statusCode",StatusCodeTag.class);
063        registerTag("structuredMessageArgument",StructuredMessageArgumentTag.class);
064        registerTag("structuredMessageFormat",StructuredMessageFormatTag.class);
065    }
066
067    @Override
068    public TagScript createTagScript(String name, Attributes attributes) throws JellyException {
069        // performance optimization
070        if (name.equals("documentation"))
071            return new TagScript() {
072                public void run(JellyContext context, XMLOutput output) throws JellyTagException {
073                    // noop
074                }
075
076                @Override
077                public void setTagBody(Script tagBody) {
078                    // noop, we don't evaluate the body, so don't even keep it in memory.
079                }
080            };
081
082        if (name.equals("getOutput"))
083            return new TagScript() {
084                /**
085                 * Adds {@link XMLOutput} to the context.
086                 */
087                public void run(JellyContext context, XMLOutput output) throws JellyTagException {
088                    context.setVariable(getAttribute("var").evaluateAsString(context),output);
089                }
090            };
091
092        if (name.equals("once"))
093            return new TagScript() {
094                /**
095                 * Adds {@link XMLOutput} to the context.
096                 */
097                public void run(JellyContext context, XMLOutput output) throws JellyTagException {
098                    HttpServletRequest request = (HttpServletRequest)context.getVariable("request");
099                    Set<String> executedScripts = (Set<String>) request.getAttribute(ONCE_TAG_KEY);
100                    if(executedScripts==null)
101                        request.setAttribute(ONCE_TAG_KEY,executedScripts=new HashSet<String>());
102
103                    String key = getFileName()+':'+getLineNumber()+':'+getColumnNumber();
104
105                    if(executedScripts.add(key)) // run it just for the first time
106                        getTagBody().run(context,output);
107                }
108            };
109
110        if (!DISABLE_INCLUDE_TAG_CLASS_ATTRIBUTE_REWRITING && name.equals("include")) {
111            // Retain backward compatibility with all views setting the obsolete 'class' attribute.
112            // See IncludeTag#setClazz for details.
113            final AttributeNameRewritingTagScript script = new AttributeNameRewritingTagScript("class", "clazz");
114            script.setTagFactory(new DefaultTagFactory(IncludeTag.class));
115            return script;
116        }
117
118        return super.createTagScript(name, attributes);
119    }
120
121    private static final String ONCE_TAG_KEY = "stapler.once";
122
123    // Disable the st:include compatibility workaround to allow testing
124    @SuppressFBWarnings(value = "MS_SHOULD_BE_FINAL", justification = "Exposed for tests")
125    public static /* non-final */ boolean DISABLE_INCLUDE_TAG_CLASS_ATTRIBUTE_REWRITING = false;
126}
