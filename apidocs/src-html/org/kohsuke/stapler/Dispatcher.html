024package org.kohsuke.stapler;
025
026import javax.servlet.ServletException;
027import java.io.IOException;
028import java.lang.reflect.InvocationTargetException;
029import java.util.ArrayList;
030import java.util.Arrays;
031import java.util.List;
032import java.util.logging.Level;
033import java.util.logging.Logger;
034
035/**
036 * Controls the dispatching of incoming HTTP requests.
037 *
038 * @author Kohsuke Kawaguchi
039 */
040public abstract class Dispatcher {
041    /**
042     * Tries to handle the given request and returns true
043     * if it succeeds. Otherwise false.
044     *
045     * <p>
046     * We have a few known strategies for handling requests
047     * (for example, one is to try to treat the request as JSP invocation,
048     * another might be try getXXX(), etc) So we use a list of
049     * {@link Dispatcher} and try them one by one until someone
050     * returns true.
051     */
052    public abstract boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node)
053        throws IOException, ServletException, IllegalAccessException, InvocationTargetException;
054
055    /**
056     * Diagnostic string that explains this dispatch rule.
057     */
058    public abstract String toString();
059
060
061    public static boolean traceable() {
062        return TRACE || TRACE_PER_REQUEST || LOGGER.isLoggable(Level.FINE);
063    }
064
065    public static void traceEval(StaplerRequest req, StaplerResponse rsp, Object node) {
066        trace(req,rsp,String.format("-> evaluate(%s%s,\"%s\")",
067                node==null?"null":'<'+node.toString()+'>',
068                node==null?"":" :"+node.getClass().getName(),
069                ((RequestImpl)req).tokens.assembleOriginalRestOfPath()));
070    }
071
072    public static void anonymizedTraceEval(StaplerRequest req, StaplerResponse rsp, Object node, String format, String... args) {
073        List<String> arg = new ArrayList<>();
074        arg.add(node == null ? "(null)" : node.getClass().getName());
075        arg.addAll(Arrays.asList(args));
076        EvaluationTrace.ApplicationTracer.trace(req, String.format(format, arg.toArray()));
077    }
078
079    public static void traceEval(StaplerRequest req, StaplerResponse rsp, Object node, String prefix, String suffix) {
080        trace(req,rsp,String.format("-> evaluate(%s<%s>%s,\"%s\")",
081                prefix,node,suffix,
082                ((RequestImpl)req).tokens.assembleOriginalRestOfPath()));
083    }
084
085    public static void traceEval(StaplerRequest req, StaplerResponse rsp, Object node, String expression) {
086        trace(req,rsp,String.format("-> evaluate(<%s>.%s,\"%s\")",
087                node,expression,
088                ((RequestImpl)req).tokens.assembleOriginalRestOfPath()));
089    }
090
091    public static void trace(StaplerRequest req, StaplerResponse rsp, String msg, Object... args) {
092        trace(req,rsp,String.format(msg,args));
093    }
094
095    public static void trace(StaplerRequest req, StaplerResponse rsp, String msg) {
096        if(isTraceEnabled(req))
097            EvaluationTrace.get(req).trace(rsp,msg);
098        if(LOGGER.isLoggable(Level.FINE))
099            LOGGER.fine(msg);
100    }
101
102    /**
103     * Checks if tracing is enabled for the given request. Tracing can be
104     * enabled globally with the "stapler.trace=true" system property. Tracing
105     * can be enabled per-request by setting "stapler.trace.per-request=true"
106     * and sending an "X-Stapler-Trace" header set to "true" with the request.
107     */
108    public static boolean isTraceEnabled(StaplerRequest req) {
109        if (TRACE)
110            return true;
111
112        if (TRACE_PER_REQUEST && "true".equals(req.getHeader("X-Stapler-Trace")))
113            return true;
114
115        return false;
116    }
117
118    /**
119     * This flag will activate the evaluation trace.
120     * It adds the evaluation process as HTTP headers,
121     * and when the evaluation failed, special diagnostic 404 page will be rendered.
122     * Useful for developer assistance.
123     */
124    public static boolean TRACE = Boolean.getBoolean("stapler.trace");
125
126    /**
127     * This flag will activate the per-request evaluation trace for requests
128     * that have X-Stapler-Trace set to "true".
129     * It adds the evaluation process as HTTP headers,
130     * and when the evaluation failed, special diagnostic 404 page will be rendered.
131     * Useful for developer assistance.
132     */
133    public static boolean TRACE_PER_REQUEST = Boolean.getBoolean("stapler.trace.per-request");
134
135    private static final Logger LOGGER = Logger.getLogger(Dispatcher.class.getName());
136}
