001package org.kohsuke.stapler.compression;
002
003import com.jcraft.jzlib.GZIPOutputStream;
004
005import javax.servlet.ServletOutputStream;
006import javax.servlet.http.HttpServletResponse;
007import javax.servlet.http.HttpServletResponseWrapper;
008import java.io.IOException;
009import java.io.OutputStreamWriter;
010import java.io.PrintWriter;
011
012/**
013 * {@link HttpServletResponse} that recognizes Content-Encoding: gzip in the response header
014 * and acts accordingly.
015 *
016 * @author Kohsuke Kawaguchi
017 */
018public class CompressionServletResponse extends HttpServletResponseWrapper {
019    /**
020     * If not null, we are compressing the stream.
021     */
022    private ServletOutputStream stream;
023    private PrintWriter writer;
024
025    public CompressionServletResponse(HttpServletResponse response) {
026        super(response);
027    }
028
029    /**
030     * If we've already inserted gzip compression filter, then we can't let the content length set from the app
031     * because that refers to the size of the uncompressed content, where we actually need the size of the compressed
032     * content.
033     *
034     * For the same reason, if the content length is set before we can decide whether to
035     */
036    @Override
037    public void setContentLength(int len) {
038        if (stream!=null)       return;
039        super.setContentLength(len);
040    }
041
042    @Override
043    public PrintWriter getWriter() throws IOException {
044        if (writer!=null)   return writer;
045        if (stream!=null) {
046            writer = new PrintWriter(new OutputStreamWriter(stream,getCharacterEncoding()));
047            return writer;
048        }
049        return super.getWriter();
050    }
051
052    @Override
053    public ServletOutputStream getOutputStream() throws IOException {
054        if (stream!=null)   return stream;
055        return super.getOutputStream();
056    }
057
058    public void activate() throws IOException {
059        if (stream==null) {
060            super.setHeader("Content-Encoding", "gzip");
061            stream = new FilterServletOutputStream(new GZIPOutputStream(super.getOutputStream()));
062        }
063    }
064
065    public void close() throws IOException {
066        if (stream!=null)
067            stream.close();
068    }
069}
