029import static org.junit.Assert.assertEquals;
030import static org.junit.Assert.fail;
031
032import java.io.ByteArrayOutputStream;
033import java.io.EOFException;
034import java.io.FileOutputStream;
035import java.io.IOException;
036import java.io.OutputStream;
037import java.nio.charset.StandardCharsets;
038import java.nio.file.Files;
039import java.nio.file.Path;
040
041import com.google.common.base.Strings;
042import org.apache.commons.io.output.NullOutputStream;
043import org.junit.Ignore;
044import org.junit.Test;
045import org.jvnet.hudson.test.Issue;
046
047public class LargeTextTest {
048
049    @Issue("JENKINS-37664")
050    @Test
051    public void writeLogTo() throws Exception {
052        assertEquals("", tail("", 0));
053        assertEquals("abcde", tail("abcde", 0));
054        assertEquals("de", tail("abcde", 3));
055        assertEquals("", tail("abcde", 5));
056        try {
057            fail(tail("abcde", 6));
058        } catch (EOFException x) {
059            // right
060        }
061        try {
062            fail(tail("abcde", 99));
063        } catch (EOFException x) {
064            // right
065        }
066    }
067
068    String tail(String text, long start) throws IOException {
069        LargeText t;
070        try (ByteBuffer bb = new ByteBuffer()) {
071            bb.write(text.getBytes(), 0, text.length());
072
073            t = new LargeText(bb, true);
074        }
075        ByteArrayOutputStream baos = new ByteArrayOutputStream();
076        assertEquals(text.length(), t.writeLogTo(start, baos));
077        return baos.toString();
078    }
079
080    @Issue("#141")
081    @Test
082    @Ignore
083    public void writeLogToWithLargeFile() throws Exception {
084        Path path = Files.createTempFile("stapler-test", ".log.gz");
085        long size = Integer.MAX_VALUE + 256L;
086        writeLargeFile(path, size);
087
088        LargeText t = new LargeText(path.toFile(), StandardCharsets.US_ASCII, true);
089
090        try (OutputStream os = new NullOutputStream()) {
091            long writenCount = t.writeLogTo(0, os);
092
093            assertEquals(size, writenCount);
094        }
095
096        Files.delete(path);
097    }
098
099    private void writeLargeFile(Path path, long size) {
100        // Write the same data over and over again, so the bytes written is high, but the file is
101        // actually very small
102        int chunkSize = 1024;
103        byte[] bytesChunk = Strings.repeat("0", chunkSize).getBytes(StandardCharsets.US_ASCII);
104        try (OutputStream stream = new FileOutputStream(path.toFile())) {
105            long remaining = size;
106            while (remaining > chunkSize) {
107                stream.write(bytesChunk);
108                remaining -= chunkSize;
109            }
110            stream.write(bytesChunk, 0, (int) remaining);
111            stream.flush();
112        } catch (IOException e) {
113            fail(e.getMessage());
114        }
115    }
