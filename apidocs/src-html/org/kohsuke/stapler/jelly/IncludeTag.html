024package org.kohsuke.stapler.jelly;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027import org.apache.commons.jelly.JellyContext;
028import org.apache.commons.jelly.JellyTagException;
029import org.apache.commons.jelly.Script;
030import org.apache.commons.jelly.TagSupport;
031import org.apache.commons.jelly.XMLOutput;
032import org.apache.commons.jelly.impl.TagScript;
033import org.kohsuke.stapler.MetaClass;
034import org.kohsuke.stapler.WebApp;
035import org.xml.sax.SAXException;
036import org.jvnet.maven.jellydoc.annotation.Required;
037
038import java.util.logging.Level;
039import java.util.logging.Logger;
040
041/**
042 * Tag that includes views of the object.
043 *
044 * @author Kohsuke Kawaguchi
045 */
046public class IncludeTag extends TagSupport {
047    public static final Logger LOGGER = Logger.getLogger(IncludeTag.class.getName());
048
049    @SuppressFBWarnings("MS_SHOULD_BE_FINAL")
050    public static /* non-final for script console */ boolean SKIP_LOGGING_CLASS_SETTER = Boolean.getBoolean(IncludeTag.class.getName() + ".skipLoggingClassSetter");
051
052    private Object it;
053
054    private String page;
055
056    private Object from;
057
058    private boolean optional;
059
060    private Class clazz;
061
062    /**
063     * Specifies the name of the JSP to be included.
064     */
065    @Required
066    public void setPage(String page) {
067        this.page = page;
068    }
069
070    /**
071     * Specifies the object for which JSP will be included.
072     * Defaults to the "it" object in the current context.
073     */
074    public void setIt(Object it) {
075        this.it = it;
076    }
077
078    /**
079     * When loading the script, use the classloader from this object
080     * to locate the script. Otherwise defaults to "it" object.
081     */
082    public void setFrom(Object from) {
083        this.from = from;
084    }
085
086    /**
087     * When loading script, load from this class.
088     *
089     * By default this is "from.getClass()". This takes
090     * precedence over the {@link #setFrom(Object)} method.
091     *
092     * This used to be called {@code setClass}, but that ended up causing
093     * problems with new commons-beanutils restrictions via
094     * {@code ConvertingWrapDynaBean} use in {@code JellyBuilder}.
095     * {@link StaplerTagLibrary} uses {@link AttributeNameRewritingTagScript}
096     * to ensure attempts to set {@code class} instead set {@code clazz}, and
097     * that attempts to set {@code clazz} directly that way fail.
098     */
099    public void setClazz(Class clazz) {
100        this.clazz = clazz;
101    }
102
103    @Deprecated // TODO Remove this method?
104    public void setClass(Class clazz) {
105        if (!SKIP_LOGGING_CLASS_SETTER) {
106            LOGGER.log(Level.WARNING, "Unexpected call to #setClass", new Exception());
107        }
108        this.clazz = clazz;
109    }
110
111    /**
112     * If true, not finding the page is not an error.
113     * (And in such a case, the body of the include tag is evaluated instead.)
114     */
115    public void setOptional(boolean optional) {
116        this.optional = optional;
117    }
118
119    public void doTag(XMLOutput output) throws JellyTagException {
120        if(page==null) {
121            // this makes it convenient when the caller wants to gracefully the expression for @page
122            // otherwise this results in http://pastie.org/601828
123            if (optional) {
124                invokeBody(output);
125                return;
126            }
127            throw new JellyTagException("The page attribute is not specified");
128        }
129        Object it = this.it;
130        if(it==null)
131            it = getContext().getVariable("it");
132
133        MetaClass c = WebApp.getCurrent().getMetaClass(getScriptClass(it));
134        Script script;
135        try {
136            script = c.loadTearOff(JellyClassTearOff.class).findScript(page);
137        } catch (RuntimeException e) {
138            throw e;
139        } catch (Exception e) {
140            throw new JellyTagException("Error loading '"+page+"' for "+c.klass,e);
141        }
142
143        if(script==null) {
144            if(optional) {
145                invokeBody(output);
146                return;
147            }
148            throw new JellyTagException("No page found '"+page+"' for "+c.klass);
149        }
150
151        context = new JellyContext(getContext());
152        context.setExportLibraries(false);
153        if(this.it!=null)
154            context.setVariable("it",this.it);
155        context.setVariable("from", from!=null?from:it);
156
157        ClassLoader old = Thread.currentThread().getContextClassLoader();
158        Thread.currentThread().setContextClassLoader(c.classLoader.loader);
159        try {
160            String source = null;
161            if(JellyFacet.TRACE) {
162                if (script instanceof TagScript) {
163                    TagScript ts = (TagScript) script;
164                    source = ts.getFileName();
165                } else
166                    source = page+" (exact source location unknown)";
167
168                String msg = "\nBegin " + source+'\n';
169                output.comment(msg.toCharArray(),0,msg.length());
170            }
171            script.run(context,output);
172            if(source!=null) {
173                String msg = "\nEnd " + source+'\n';
174                output.comment(msg.toCharArray(),0,msg.length());
175            }
176        } catch (SAXException e) {
177            throw new JellyTagException(e);
178        } finally {
179            Thread.currentThread().setContextClassLoader(old);
180        }
181    }
182
183    private Class getScriptClass(Object it) {
184        if (clazz != null) return clazz;
185        if (from != null) return from.getClass();
186        else return it.getClass();
187    }
188}
