001package org.kohsuke.stapler.compression;
002
003import javax.servlet.ServletOutputStream;
004import javax.servlet.WriteListener;
005
006import java.io.FilterOutputStream;
007import java.io.IOException;
008import java.io.OutputStream;
009
010/**
011 * {@link ServletOutputStream} that writes to the specified output stream.
012 *
013 * @author Kohsuke Kawaguchi
014 */
015public class FilterServletOutputStream extends ServletOutputStream {
016    private final OutputStream out;
017    private final ServletOutputStream realSream;
018    
019    /**
020     * Constructs a new {@link FilterOutputStream}.
021     * @param out the stream that sits above the realStream, performing some filtering.  This must be eventually delegating eventual writes to {@code realStream}.
022     * @param realStream the actual underlying ServletOutputStream from the container.  Used to check the {@link #isReady()} state and to add {@link WriteListener}s. 
023     */
024    public FilterServletOutputStream(OutputStream out, ServletOutputStream realStream) {
025        this.out = out;
026        this.realSream = realStream;
027    }
028
029    @Override
030    public void write(int b) throws IOException {
031        out.write(b);
032    }
033
034    @Override
035    public void write(byte[] b) throws IOException {
036        out.write(b);
037    }
038
039    @Override
040    public void write(byte[] b, int off, int len) throws IOException {
041        out.write(b, off, len);
042    }
043
044    @Override
045    public void close() throws IOException {
046        out.close();
047    }
048
049    @Override
050    public void flush() throws IOException {
051        out.flush();
052    }
053
054    @Override
055    public boolean isReady() {
056        return realSream.isReady();
057    }
058
059    @Override
060    public void setWriteListener(WriteListener writeListener) {
061        realSream.setWriteListener(writeListener);
062    }
063}
