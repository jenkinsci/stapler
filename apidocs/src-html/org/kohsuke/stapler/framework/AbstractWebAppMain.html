024package org.kohsuke.stapler.framework;
025
026import org.jvnet.localizer.LocaleProvider;
027import org.kohsuke.stapler.StaplerRequest;
028import org.kohsuke.stapler.Stapler;
029import org.kohsuke.stapler.framework.errors.NoHomeDirError;
030
031import javax.naming.Context;
032import javax.naming.InitialContext;
033import javax.naming.NamingException;
034import javax.servlet.ServletContext;
035import javax.servlet.ServletContextEvent;
036import javax.servlet.ServletContextListener;
037import java.io.File;
038import java.util.Locale;
039import java.util.logging.Level;
040import java.util.logging.Logger;
041
042/**
043 * Entry point for web applications.
044 *
045 * <p>
046 * Applications that use stapler can use this as the base class,
047 * and then register that as the servlet context listener.
048 *
049 * @param <T>
050 *      The type of the root object instance
051 * @author Kohsuke Kawaguchi
052 */
053public abstract class AbstractWebAppMain<T> implements ServletContextListener {
054
055    protected final Class<T> rootType;
056
057    private static final String APP = "app";
058
059    protected ServletContext context;
060
061    /**
062     * Once the home directory is determined, this value is set to that directory.
063     */
064    protected File home;
065
066    protected AbstractWebAppMain(Class<T> rootType) {
067        this.rootType = rootType;
068    }
069
070    /**
071     * Returns the application name, like "Hudson" or "Torricelli".
072     *
073     * The method should always return the same value. The name
074     * should only contain alpha-numeric character.
075     */
076    protected abstract String getApplicationName();
077
078    /**
079     * If the root application object is loaded asynchronously,
080     * override this method to return the place holder object
081     * to serve the request in the mean time.
082     *
083     * @return
084     *      null to synchronously load the application object.
085     */
086    protected Object createPlaceHolderForAsyncLoad() {
087        return null;
088    }
089
090    /**
091     * Creates the root application object.
092     */
093    protected abstract Object createApplication() throws Exception;
094
095    public void contextInitialized(ServletContextEvent event) {
096        try {
097            context = event.getServletContext();
098
099            installLocaleProvider();
100
101            if (!checkEnvironment())
102                return;
103
104            Object ph = createPlaceHolderForAsyncLoad();
105            if(ph!=null) {
106                // asynchronous load
107                context.setAttribute(APP,ph);
108
109                new Thread(getApplicationName()+" initialization thread") {
110                    public void run() {
111                        setApplicationObject();
112                    }
113                }.start();
114            } else {
115                setApplicationObject();
116            }
117        } catch (Error e) {
118            LOGGER.log(Level.SEVERE, "Failed to initialize "+getApplicationName(),e);
119            throw e;
120        } catch (RuntimeException e) {
121            LOGGER.log(Level.SEVERE, "Failed to initialize "+getApplicationName(),e);
122            throw e;
123        }
124    }
125
126    /**
127     * Sets the root application object.
128     */
129    protected void setApplicationObject() {
130        try {
131            context.setAttribute(APP,createApplication());
132        } catch (Error e) {
133            LOGGER.log(Level.SEVERE, "Failed to initialize "+getApplicationName(),e);
134            throw e;
135        } catch (RuntimeException e) {
136            LOGGER.log(Level.SEVERE, "Failed to initialize "+getApplicationName(),e);
137            throw e;
138        } catch (Exception e) {
139            LOGGER.log(Level.SEVERE, "Failed to initialize "+getApplicationName(),e);
140            throw new Error(e);
141        }
142    }
143
144    /**
145     * Performs pre start-up environment check.
146     *
147     * @return
148     *      false if a check fails. Webapp will fail to boot in this case.
149     */
150    protected boolean checkEnvironment() {
151        home = getHomeDir().getAbsoluteFile();
152        home.mkdirs();
153        LOGGER.info(getApplicationName()+" home directory: "+home);
154
155        // check that home exists (as mkdirs could have failed silently), otherwise throw a meaningful error
156        if (!home.exists()) {
157            context.setAttribute(APP,new NoHomeDirError(home));
158            return false;
159        }
160
161        return true;
162    }
163
164    /**
165     * Install {@link LocaleProvider} that uses the current request to determine the language.
166     */
167    private void installLocaleProvider() {
168        LocaleProvider.setProvider(new LocaleProvider() {
169            public Locale get() {
170                Locale locale=null;
171                StaplerRequest req = Stapler.getCurrentRequest();
172                if(req!=null)
173                    locale = req.getLocale();
174                if(locale==null)
175                    locale = Locale.getDefault();
176                return locale;
177            }
178        });
179    }
180
181    /**
182     * Determines the home directory for the application.
183     *
184     * People makes configuration mistakes, so we are trying to be nice
185     * with those by doing {@link String#trim()}.
186     */
187    protected File getHomeDir() {
188        // check JNDI for the home directory first
189        String varName = getApplicationName().toUpperCase() + "_HOME";
190        try {
191            InitialContext iniCtxt = new InitialContext();
192            Context env = (Context) iniCtxt.lookup("java:comp/env");
193            String value = (String) env.lookup(varName);
194            if(value!=null && value.trim().length()>0)
195                return new File(value.trim());
196            // look at one more place. See HUDSON-1314
197            value = (String) iniCtxt.lookup(varName);
198            if(value!=null && value.trim().length()>0)
199                return new File(value.trim());
200        } catch (NamingException e) {
201            // ignore
202        }
203
204        // finally check the system property
205        String sysProp = System.getProperty(varName);
206        if(sysProp!=null)
207            return new File(sysProp.trim());
208
209        // look at the env var next
210        String env = System.getenv(varName);
211        if(env!=null)
212            return new File(env.trim()).getAbsoluteFile();
213
214        return getDefaultHomeDir();
215    }
216
217    /**
218     * If no home directory is configured, this method is called
219     * to determine the default location, which is "~/.appname".
220     *
221     * Override this method to change that behavior.
222     */
223    protected File getDefaultHomeDir() {
224        return new File(new File(System.getProperty("user.home")),'.'+getApplicationName().toLowerCase());
225    }
226
227    public void contextDestroyed(ServletContextEvent event) {
228        Object o = event.getServletContext().getAttribute(APP);
229        if(rootType.isInstance(o)) {
230            cleanUp(rootType.cast(o));
231        }
232    }
233
234    /**
235     * Called during the destructino of the web app to perform
236     * any clean up act on the application object.
237     */
238    protected void cleanUp(T app) {}
239
240    private static final Logger LOGGER = Logger.getLogger(AbstractWebAppMain.class.getName());
241}
