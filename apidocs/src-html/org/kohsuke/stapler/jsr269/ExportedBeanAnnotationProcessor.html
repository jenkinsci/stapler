029@SuppressWarnings({"Since15"})
030@SupportedSourceVersion(SourceVersion.RELEASE_6)
031@SupportedAnnotationTypes("org.kohsuke.stapler.export.Exported")
032@MetaInfServices(Processor.class)
033public class ExportedBeanAnnotationProcessor extends AbstractProcessorImpl {
034
035    private Set<String> exposedBeanNames;
036
037    @Override
038    public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
039        try {
040            if (roundEnv.processingOver()) {
041                FileObject beans = createResource(STAPLER_BEAN_FILE);
042                PrintWriter w = new PrintWriter(new OutputStreamWriter(beans.openOutputStream(), "UTF-8"));
043                for (String beanName : exposedBeanNames) {
044                    w.println(beanName);
045                }
046                w.close();
047                return false;
048            }
049
050            // collect all exposed properties
051            PoormansMultimap<TypeElement, Element/*member decls*/> props = new PoormansMultimap<TypeElement,Element>();
052
053            for (Element exported : roundEnv.getElementsAnnotatedWith(Exported.class)) {
054                Element type = exported.getEnclosingElement();
055                if (type.getKind().isClass() || type.getKind().isInterface()) {
056                    props.put((TypeElement)type, exported);
057                }
058            }
059
060            if (exposedBeanNames == null) {
061                scanExisting();
062            }
063
064            for (Entry<TypeElement, Collection<Element>> e : props.asMap().entrySet()) {
065                exposedBeanNames.add(e.getKey().getQualifiedName().toString());
066
067                final Properties javadocs = new Properties();
068                for (Element md : e.getValue()) {
069                    switch (md.getKind()) {
070                    case FIELD:
071                    case METHOD:
072                       String javadoc = getJavadoc(md);
073                       if(javadoc!=null)
074                           javadocs.put(md.getSimpleName().toString(), javadoc);
075                        break;
076                    default:
077                        throw new AssertionError("Unexpected element type: "+md);
078                    }
079                        // TODO: possibly a proper method signature generation, but it's too tedious
080                        // way too tedious.
081                        //private String getSignature(MethodDeclaration m) {
082                        //    final StringBuilder buf = new StringBuilder(m.getSimpleName());
083                        //    buf.append('(');
084                        //    boolean first=true;
085                        //    for (ParameterDeclaration p : m.getParameters()) {
086                        //        if(first)   first = false;
087                        //        else        buf.append(',');
088                        //        p.getType().accept(new SimpleTypeVisitor() {
089                        //            public void visitPrimitiveType(PrimitiveType pt) {
090                        //                buf.append(pt.getKind().toString().toLowerCase());
091                        //            }
092                        //            public void visitDeclaredType(DeclaredType dt) {
093                        //                buf.append(dt.getDeclaration().getQualifiedName());
094                        //            }
095                        //
096                        //            public void visitArrayType(ArrayType at) {
097                        //                at.getComponentType().accept(this);
098                        //                buf.append("[]");
099                        //            }
100                        //
101                        //            public void visitTypeVariable(TypeVariable tv) {
102                        //
103                        //                // TODO
104                        //                super.visitTypeVariable(typeVariable);
105                        //            }
106                        //
107                        //            public void visitVoidType(VoidType voidType) {
108                        //                // TODO
109                        //                super.visitVoidType(voidType);
110                        //            }
111                        //        });
112                        //    }
113                        //    buf.append(')');
114                        //    // TODO
115                        //    return null;
116                        //}
117                }
118
119                String javadocFile = e.getKey().getQualifiedName().toString().replace('.', '/') + ".javadoc";
120                notice("Generating "+ javadocFile, e.getKey());
121                writePropertyFile(javadocs, javadocFile);
122            }
123
124        } catch (IOException x) {
125            error(x);
126        } catch (RuntimeException e) {
127            // javac sucks at reporting errors in annotation processors
128            e.printStackTrace();
129            throw e;
130        } catch (Error e) {
131            e.printStackTrace();
132            throw e;
133        }
134        return false;
135    }
136
137    private void scanExisting() throws IOException {
138        exposedBeanNames = new TreeSet<String>();
139
140        try {
141            FileObject beans = getResource(STAPLER_BEAN_FILE);
142            BufferedReader in = new BufferedReader(new InputStreamReader(beans.openInputStream(),"UTF-8"));
143            String line;
144            while((line=in.readLine())!=null)
145                exposedBeanNames.add(line.trim());
146            in.close();
147        } catch (FileNotFoundException e) {
148            // no existing file, which is fine
149        }
150    }
151
152    static final String STAPLER_BEAN_FILE = "META-INF/exposed.stapler-beans";
153}
