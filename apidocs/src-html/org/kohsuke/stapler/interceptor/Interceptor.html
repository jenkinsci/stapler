001package org.kohsuke.stapler.interceptor;
002
003import org.kohsuke.stapler.CancelRequestHandlingException;
004import org.kohsuke.stapler.Function;
005import org.kohsuke.stapler.StaplerRequest;
006import org.kohsuke.stapler.StaplerResponse;
007
008import java.lang.reflect.InvocationTargetException;
009import org.kohsuke.stapler.HttpResponses;
010
011import javax.servlet.ServletException;
012
013/**
014 * Intercepts the domain method call from Stapler.
015 *
016 * @author Kohsuke Kawaguchi
017 * @see InterceptorAnnotation
018 */
019public abstract class Interceptor {
020    protected Function target;
021
022    /**
023     * Called by Stapler to set up the target of the interceptor.
024     * This function object represents a method on which your annotation is placed.
025     * 
026     * This happens once before this instance takes any calls.
027     */
028    public void setTarget(Function target) {
029        this.target = target;
030    }
031
032    /**
033     * Intercepts the call.
034     *
035     * <p>
036     * The minimal no-op interceptor would do {@code target.invoke(request,response,instance,arguments)},
037     * but the implementation is free to do additional pre/post processing.
038     *
039     * @param request
040     *      The current request we are processing.
041     * @param response
042     *      The current response object.
043     * @param instance
044     *      The domain object instance whose method we are about to invoke.
045     * @param arguments
046     *      Arguments of the method call.
047     *
048     * @return
049     *      Return value from the method.
050     * @throws InvocationTargetException if you want to send e.g. something from {@link HttpResponses}
051     * @throws CancelRequestHandlingException
052     *      to cancel this request handling and moves on to the next available dispatch mechanism.
053     */
054    public abstract Object invoke(StaplerRequest request, StaplerResponse response,
055                                  Object instance, Object[] arguments)
056            throws IllegalAccessException, InvocationTargetException, ServletException;
057}
