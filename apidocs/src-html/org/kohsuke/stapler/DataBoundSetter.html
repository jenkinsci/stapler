001package org.kohsuke.stapler;
002
003import net.sf.json.JSONObject;
004
005import java.beans.Introspector;
006import java.lang.annotation.Documented;
007import java.lang.annotation.Retention;
008import java.lang.annotation.Target;
009
010import static java.lang.annotation.ElementType.*;
011import static java.lang.annotation.RetentionPolicy.*;
012
013/**
014 * Designates a setter method used to databind JSON vlues into objects in methods like
015 * {@link StaplerRequest#bindJSON(Class, JSONObject)} and
016 * {@link StaplerRequest#bindParameters(Class, String)}.
017 *
018 * <p>
019 * Stapler will first invoke {@link DataBoundConstructor}-annotated constructor, and if there's any
020 * remaining properties in JSON, it'll try to find a matching {@link DataBoundSetter}-annotated setter.
021 *
022 * <p>
023 * The setter method is discovered through {@link Introspector}, so setter method name must match
024 * the property name (such as <tt>setFoo</tt> for the <tt>foo</tt> property.)
025 *
026 * @author Kohsuke Kawaguchi
027 */
028@Retention(RUNTIME)
029@Target(METHOD)
030@Documented
031public @interface DataBoundSetter {
032}
