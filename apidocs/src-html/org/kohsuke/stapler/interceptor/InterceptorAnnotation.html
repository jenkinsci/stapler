001package org.kohsuke.stapler.interceptor;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.Retention;
005import java.lang.annotation.Target;
006
007import static java.lang.annotation.ElementType.*;
008import static java.lang.annotation.RetentionPolicy.*;
009
010/**
011 * Marks the annotation as an interceptor annotation,
012 * which executes before/after the method invocation of domain objects happen
013 * as a part of the request processing.
014 *
015 * <p>
016 * This mechanism is useful for performing declarative processing/check on domain objects,
017 * such as checking HTTP headers, performing the access control, etc.
018 *
019 * @author Kohsuke Kawaguchi
020 * @see Interceptor
021 * @see RequirePOST
022 */
023@Target(ANNOTATION_TYPE)
024@Retention(RUNTIME)
025@Documented
026public @interface InterceptorAnnotation  {
027    /**
028     * Actual interceptor logic. Must have a default constructor.
029     */
030    Class<? extends Interceptor> value();
031
032    /**
033     * The point of invocation of this interceptor.
034     */
035    Stage stage() default Stage.PREINVOKE;
036}
