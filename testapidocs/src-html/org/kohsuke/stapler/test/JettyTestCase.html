001package org.kohsuke.stapler.test;
002
003import com.gargoylesoftware.htmlunit.WebClient;
004import junit.framework.TestCase;
005import org.kohsuke.stapler.Stapler;
006import org.kohsuke.stapler.WebApp;
007import org.mortbay.jetty.Server;
008import org.mortbay.jetty.bio.SocketConnector;
009import org.mortbay.jetty.servlet.Context;
010import org.mortbay.jetty.servlet.ServletHolder;
011import org.mortbay.jetty.webapp.WebAppContext;
012
013import javax.servlet.ServletContext;
014import java.net.URL;
015
016/**
017 * Base test case for embedded Jetty.
018 *
019 * Your test method runs with embedded Jetty with the instance of your test class bound to the root "application"
020 * object in Stapler. You can use {@link WebClient} to make HTTP calls, receive that on your test class,
021 * and inspect side-effects, etc.
022 *
023 * @author Kohsuke Kawaguchi
024 */
025public abstract class JettyTestCase extends TestCase {
026    protected Server server;
027    /**
028     * The top URL of this test web application.
029     */
030    protected URL url;
031    protected Stapler stapler;
032    protected ServletContext servletContext;
033    protected WebApp webApp;
034
035    @Override
036    protected void setUp() throws Exception {
037        super.setUp();
038        server = new Server();
039
040        server.setHandler(new WebAppContext("/noroot", ""));
041
042        final Context context = new Context(server, getContextPath(), Context.SESSIONS);
043        configure(context);
044        server.setHandler(context);
045
046        SocketConnector connector = new SocketConnector();
047        server.addConnector(connector);
048        server.start();
049
050        url = new URL("http://localhost:"+connector.getLocalPort()+getContextPath()+"/");
051
052        servletContext = context.getServletContext();
053        webApp = WebApp.get(servletContext);
054
055        // export the test object as the root as a reasonable default.
056        webApp.setApp(this);
057    }
058
059    /**
060     * Can be used to set different context path for the root object.
061     *
062     * For example, "/foo/bar"
063     */
064    protected String getContextPath() {
065        return "";
066    }
067
068    /**
069     * Sets up how the servlet/filters are bound.
070     */
071    protected void configure(Context context) {
072        context.addServlet(new ServletHolder(new Stapler()), "/*");
073    }
074
075    @Override
076    protected void tearDown() throws Exception {
077        super.tearDown();
078        server.stop();
079    }
080}
