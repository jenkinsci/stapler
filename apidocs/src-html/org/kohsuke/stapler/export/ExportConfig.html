001package org.kohsuke.stapler.export;
002
003/**
004 * Controls the output behaviour.
005 *
006 * @author Kohsuke Kawaguchi
007 */
008public class ExportConfig {
009    /**
010     * @deprecated
011     *      Use getter and setter
012     */
013    public boolean prettyPrint;
014
015    private ClassAttributeBehaviour classAttribute = ClassAttributeBehaviour.IF_NEEDED;
016
017    private ExportInterceptor exportInterceptor = ExportInterceptor.DEFAULT;
018
019    private boolean skipIfFail = false;
020
021    private Flavor flavor = Flavor.JSON;
022
023    /**
024     * If true, output will be indented to make it easier for humans to understand.
025     */
026    public boolean isPrettyPrint() {
027        return prettyPrint;
028    }
029
030    /**
031     * If true, output will be indented to make it easier for humans to understand.
032     */
033    public ExportConfig withPrettyPrint(boolean prettyPrint) {
034        this.prettyPrint = prettyPrint;
035        return this;
036    }
037
038    public ClassAttributeBehaviour getClassAttribute() {
039        return classAttribute;
040    }
041
042    /**
043     * Controls the behaviour of the class attribute to be produced.
044     */
045    public ExportConfig withClassAttribute(ClassAttributeBehaviour cab) {
046        if (cab==null)  throw new NullPointerException();
047        this.classAttribute = cab;
048        return this;
049    }
050
051    /**
052     * Controls serialization of {@link Exported} properties.
053     */
054    public ExportInterceptor getExportInterceptor() {
055        return exportInterceptor;
056    }
057
058    public ExportConfig withExportInterceptor(ExportInterceptor interceptor){
059        this.exportInterceptor = interceptor;
060        return this;
061    }
062
063    public ExportConfig withSkipIfFail(boolean skipIfFail){
064        this.skipIfFail = skipIfFail;
065        return this;
066    }
067
068    /**
069     * Turn on or off pretty printing of serialized data.
070     */
071    public boolean isSkipIfFail() {
072        return skipIfFail;
073    }
074
075    /**
076     * Gives {@link Flavor}
077     */
078    public Flavor getFlavor(){
079        return flavor;
080    }
081
082    public ExportConfig withFlavor(Flavor flavor){
083        this.flavor = flavor;
084        return this;
085    }
086}
