001package org.kohsuke.stapler.test;
002
003import junit.framework.TestCase;
004import org.kohsuke.stapler.Stapler;
005import org.kohsuke.stapler.WebApp;
006import org.mortbay.jetty.Server;
007import org.mortbay.jetty.bio.SocketConnector;
008import org.mortbay.jetty.servlet.Context;
009import org.mortbay.jetty.servlet.ServletHolder;
010import org.mortbay.jetty.webapp.WebAppContext;
011
012import javax.servlet.ServletContext;
013import java.net.URL;
014
015/**
016 * Base test case for embedded Jetty.
017 * 
018 * @author Kohsuke Kawaguchi
019 */
020public abstract class JettyTestCase extends TestCase {
021    protected Server server;
022    /**
023     * The top URL of this test web application.
024     */
025    protected URL url;
026    protected Stapler stapler;
027    protected ServletContext servletContext;
028    protected WebApp webApp;
029
030    @Override
031    protected void setUp() throws Exception {
032        super.setUp();
033        server = new Server();
034
035        server.setHandler(new WebAppContext("/noroot", ""));
036
037        final Context context = new Context(server, "", Context.SESSIONS);
038        context.addServlet(new ServletHolder(new Stapler()), "/*");
039        server.setHandler(context);
040
041        SocketConnector connector = new SocketConnector();
042        server.addConnector(connector);
043        server.start();
044
045        url = new URL("http://localhost:"+connector.getLocalPort()+"/");
046
047        servletContext = context.getServletContext();
048        webApp = WebApp.get(servletContext);
049
050        // export the test object as the root as a reasonable default.
051        webApp.setApp(this);
052    }
053
054    @Override
055    protected void tearDown() throws Exception {
056        super.tearDown();
057        server.stop();
058    }
059}
