001package org.kohsuke.stapler.lang;
002
003import java.lang.annotation.Annotation;
004import java.lang.reflect.InvocationTargetException;
005import java.lang.reflect.Method;
006import java.lang.reflect.Modifier;
007
008/**
009 * @author Kohsuke Kawaguchi
010 * @since 1.220
011 */
012public abstract class MethodRef extends AnnotatedRef {
013    /**
014     * Returns true if this method is a 'public' method that should be used for routing requests.
015     */
016    public boolean isRoutable() {
017        return true;
018    }
019
020    public abstract Object invoke(Object _this, Object... args) throws InvocationTargetException, IllegalAccessException;
021
022    public static MethodRef wrap(final Method m) {
023        m.setAccessible(true);
024
025        return new MethodRef() {
026            @Override
027            public <T extends Annotation> T getAnnotation(Class<T> type) {
028                return m.getAnnotation(type);
029            }
030
031            @Override
032            public boolean isRoutable() {
033                if (m.isBridge())    return false;
034                return (m.getModifiers() & Modifier.PUBLIC)!=0;
035            }
036
037            @Override
038            public Object invoke(Object _this, Object... args) throws InvocationTargetException, IllegalAccessException {
039                return m.invoke(_this,args);
040            }
041        };
042    }
043}
