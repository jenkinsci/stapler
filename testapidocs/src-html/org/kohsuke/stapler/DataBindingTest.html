247    public static class Point {
248        int x,y;
249
250        @DataBoundConstructor
251        public Point(int x, int y) {
252            this.x = x;
253            this.y = y;
254        }
255
256        @Override
257        public boolean equals(Object o) {
258            if (this == o) return true;
259            if (o == null || getClass() != o.getClass()) return false;
260
261            Point rhs = (Point) o;
262            return x == rhs.x && y == rhs.y;
263        }
264
265        @Override
266        public int hashCode() {
267            int result = x;
268            result = 31 * result + y;
269            return result;
270        }
271    }

273    public void testSetterInvocation() {
274        SetterBinding r = bind("{x:1,y:2,z:3,w:1, children:[{x:5,y:5,z:5},{x:6,y:6,z:6}], anotherObject:{$class:'org.kohsuke.stapler.DataBindingTest$Point', x:1,y:1} }",SetterBinding.class);
275        assertEquals(1,r.x);
276        assertEquals(2,r.y);
277        assertEquals(3,r.z);
278        assertEquals(0,r.w);
279
280        assertEquals(2,r.children.size());
281        SetterBinding c1 = r.children.get(0);
282        assertEquals(5, c1.x);
283        assertEquals(5, c1.y);
284        assertEquals(5, c1.z);
285
286        SetterBinding c2 = r.children.get(1);
287        assertEquals(6, c2.x);
288        assertEquals(6, c2.y);
289        assertEquals(6, c2.z);
290
291        Point o = (Point)r.o;
292        assertEquals(1,o.x);
293        assertEquals(1,o.y);
294    }
