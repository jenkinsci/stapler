024package org.kohsuke.stapler.jelly;
025
026import org.kohsuke.stapler.Stapler;
027import org.kohsuke.stapler.StaplerRequest;
028import org.kohsuke.stapler.StaplerResponse;
029import org.apache.commons.jelly.Script;
030import org.apache.commons.jelly.JellyTagException;
031import org.apache.commons.jelly.XMLOutput;
032import org.apache.commons.jelly.XMLOutputFactory;
033import org.apache.commons.jelly.JellyContext;
034import org.apache.commons.jelly.impl.TagScript;
035
036import javax.servlet.ServletContext;
037import java.io.IOException;
038import java.io.OutputStream;
039import java.io.BufferedOutputStream;
040import java.io.FilterOutputStream;
041import java.io.Writer;
042import java.util.Enumeration;
043
044/**
045 * Standard implementation of {@link ScriptInvoker}.
046 * 
047 * @author Kohsuke Kawaguchi
048 */
049public class DefaultScriptInvoker implements ScriptInvoker, XMLOutputFactory {
050    public void invokeScript(StaplerRequest req, StaplerResponse rsp, Script script, Object it) throws IOException, JellyTagException {
051        XMLOutput xmlOutput = createXMLOutput(req, rsp, script, it);
052
053        invokeScript(req,rsp,script,it,xmlOutput);
054        
055        xmlOutput.flush();
056        xmlOutput.close();
057    }
058
059    public void invokeScript(StaplerRequest req, StaplerResponse rsp, Script script, Object it, XMLOutput out) throws IOException, JellyTagException {
060        JellyContext context = createContext(req,rsp,script,it);
061        exportVariables(req, rsp, script, it, context);
062
063        script.run(context,out);
064    }
065
066    protected XMLOutput createXMLOutput(StaplerRequest req, StaplerResponse rsp, Script script, Object it) throws IOException {
067        // TODO: make XMLOutput auto-close OutputStream to avoid leak
068        HTMLWriterOutput hwo = HTMLWriterOutput.create(createOutputStream(req, rsp, script, it));
069        String ct = rsp.getContentType();
070        if (ct != null && !ct.startsWith("text/html"))
071            hwo.useHTML(false);
072        return hwo;
073    }
074
075    private boolean doCompression(Script script) {
076        if (COMPRESS_BY_DEFAULT)    return true;
077        if (script instanceof TagScript) {
078            TagScript ts = (TagScript) script;
079            if(ts.getLocalName().equals("compress"))
080                return true;
081        }
082        return false;
083    }
084    protected OutputStream createOutputStream(StaplerRequest req, StaplerResponse rsp, Script script, Object it) throws IOException {
085        // do we want to do compression?
086        OutputStream output=null;
087        if (doCompression(script))
088            output = rsp.getCompressedOutputStream(req);
089        if(output==null)    // nope
090            output = new BufferedOutputStream(rsp.getOutputStream());
091
092        output = new FilterOutputStream(output) {
093            public void flush() {
094                // flushing ServletOutputStream causes Tomcat to
095                // send out headers, making it impossible to set contentType from the script.
096                // so don't let Jelly flush.
097            }
098            public void write(byte b[], int off, int len) throws IOException {
099                out.write(b, off, len);
100            }
101        };
102        return output;
103    }
104
105    protected void exportVariables(StaplerRequest req, StaplerResponse rsp, Script script, Object it, JellyContext context) {
106        Enumeration en = req.getAttributeNames();
107        // expose request attributes, just like JSP
108        while (en.hasMoreElements()) {
109            String name = (String) en.nextElement();
110            context.setVariable(name,req.getAttribute(name));
111        }
112
113        context.setVariable("request",req);
114        context.setVariable("response",rsp);
115        context.setVariable("it",it);
116        ServletContext servletContext = req.getServletContext();
117        context.setVariable("servletContext",servletContext);
118        context.setVariable("app",servletContext.getAttribute("app"));
119        // property bag to store request scope variables
120        context.setVariable("requestScope",context.getVariables());
121        // this variable is needed to make "jelly:fmt" taglib work correctly
122        context.setVariable("org.apache.commons.jelly.tags.fmt.locale",req.getLocale());
123    }
124
125    protected JellyContext createContext(final StaplerRequest req, StaplerResponse rsp, Script script, Object it) {
126        CustomJellyContext context = new CustomJellyContext();
127        // let Jelly see the whole classes
128        context.setClassLoader(req.getWebApp().getClassLoader());
129        // so TagScript.getBodyText() will use HTMLWriterOutput
130        context.setVariable(XMLOutputFactory.class.getName(), this);
131        return context;
132    }
133
134    public XMLOutput createXMLOutput(Writer writer, boolean escapeText) {
135        StaplerResponse rsp = Stapler.getCurrentResponse();
136        String ct = rsp!=null ? rsp.getContentType() : "?";
137        if (ct != null && !ct.startsWith("text/html"))
138            return XMLOutput.createXMLOutput(writer, escapeText);
139        return HTMLWriterOutput.create(writer, escapeText);
140    }
141
142    /**
143     * Whether gzip compression of the dynamic content is enabled by default or not.
144     *
145     * <p>
146     * For non-trivial web applications, where the performance matters, it is normally a good trade-off to spend
147     * a bit of CPU cycles to compress data. This is because:
148     *
149     * <ul>
150     * <li>CPU is already 1 or 2 order of magnitude faster than RAM and network.
151     * <li>CPU is getting faster than any other components, such as RAM and network.
152     * <li>Because of the TCP window slow start, on a large latency network, compression makes difference in
153     *     the order of 100ms to 1sec to the completion of a request by saving multiple roundtrips.
154     * </ul>
155     *
156     * Stuff rendered by Jelly is predominantly text, so the compression would work well.
157     *
158     * @see http://www.slideshare.net/guest22d4179/latency-trumps-all
159     */
160    public static boolean COMPRESS_BY_DEFAULT = Boolean.parseBoolean(System.getProperty(DefaultScriptInvoker.class.getName()+".compress","true"));
161}
