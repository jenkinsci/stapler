024package org.kohsuke.stapler;
025
026import javax.servlet.ServletException;
027import javax.servlet.RequestDispatcher;
028import java.io.IOException;
029import java.util.HashMap;
030import java.util.Map;
031import java.util.Map.Entry;
032
033/**
034 * {@link HttpResponse} that forwards to a {@link RequestDispatcher}, such as a view.
035 * Extends from {@link RuntimeException} so that you can throw it.
036 *
037 * @author Kohsuke Kawaguchi
038 */
039public class ForwardToView extends RuntimeException implements HttpResponse {
040    private final DispatcherFactory factory;
041    private boolean optional;
042    private final Map<String,Object> attributes = new HashMap<String, Object>();
043
044    private interface DispatcherFactory {
045        RequestDispatcher get(StaplerRequest req) throws IOException;
046    }
047
048    public ForwardToView(final RequestDispatcher dispatcher) {
049        this.factory = new DispatcherFactory() {
050            public RequestDispatcher get(StaplerRequest req) {
051                return dispatcher;
052            }
053        };
054    }
055
056    public ForwardToView(final Object it, final String view) {
057        this.factory = new DispatcherFactory() {
058            public RequestDispatcher get(StaplerRequest req) throws IOException {
059                return req.getView(it,view);
060            }
061        };
062    }
063
064    public ForwardToView(final Class c, final String view) {
065        this.factory = new DispatcherFactory() {
066            public RequestDispatcher get(StaplerRequest req) throws IOException {
067                return req.getView(c,view);
068            }
069        };
070    }
071
072    /**
073     * Forwards to the view with specified attributes exposed as a variable binding.
074     */
075    public ForwardToView with(String varName, Object value) {
076        attributes.put(varName,value);
077        return this;
078    }
079
080    public ForwardToView with(Map<String,?> attributes) {
081        this.attributes.putAll(attributes);
082        return this;
083    }
084
085    /**
086     * Make this forwarding optional. Render nothing if a view doesn't exist.
087     */
088    public ForwardToView optional() {
089        optional = true;
090        return this;
091    }
092
093    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
094        for (Entry<String, Object> e : attributes.entrySet())
095            req.setAttribute(e.getKey(),e.getValue());
096        RequestDispatcher rd = factory.get(req);
097        if (rd==null && optional)
098            return;
099        rd.forward(req, rsp);
100    }
101}
