001package org.kohsuke.stapler.lang;
002
003import org.kohsuke.stapler.MetaClassLoader;
004
005import java.net.URL;
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * Strategy pattern to provide navigation across class-like objects in other languages of JVM.
011 *
012 * <p>
013 * Implementations should be stateless and typically a singleton.
014 *
015 * @author Kohsuke Kawaguchi
016 */
017public abstract class KlassNavigator<C> {
018    /**
019     * Loads the resources associated with this class.
020     *
021     * <p>
022     * In stapler, the convention is that the "associated" resources live in the directory named after
023     * the fully qualified class name (as opposed to the behavior of {@link Class#getResource(String)},
024     * that looks for resources in the same package as the class.)
025     *
026     * <p>
027     * But other languages can choose their own conventions if it makes more sense to do so.
028     * For example, stapler-jruby uses camelized class name.
029     *
030     * <p>
031     * Implementation must consult {@link MetaClassLoader#debugLoader} if it's available. Implementation
032     * must not look for resources in the base type. That operation is performed by the caller when
033     * needed.
034     *
035     * @return
036     *      non-null if the resource is found. Otherwise null.
037     */
038    public abstract URL getResource(C clazz, String resourceName);
039
040    /**
041     * Lists up all the ancestor classes, from specific to general, without any duplicate.
042     *
043     * This is used to look up a resource.
044     */
045    public abstract Iterable<Klass<?>> getAncestors(C clazz);
046
047    /**
048     * Gets the super class.
049     *
050     * @return
051     *      Can be null.
052     */
053    public abstract Klass<?> getSuperClass(C clazz);
054
055    /**
056     * For backward compatibility, map the given class to the closest Java equivalent.
057     * In the worst case, this is Object.class
058     */
059    public abstract Class toJavaClass(C clazz);
060
061    public static final KlassNavigator<Class> JAVA = new KlassNavigator<Class>() {
062        @Override
063        public URL getResource(Class clazz, String resourceName) {
064            ClassLoader cl = clazz.getClassLoader();
065            if (cl==null)   return null;
066
067            String fullName;
068            if (resourceName.startsWith("/"))
069                fullName = resourceName.substring(1);
070            else
071                fullName = clazz.getName().replace('.','/').replace('$','/')+'/'+resourceName;
072
073            if (MetaClassLoader.debugLoader!=null) {
074                URL res = MetaClassLoader.debugLoader.loader.getResource(fullName);
075                if (res!=null)  return res;
076            }
077            return cl.getResource(fullName);
078        }
079
080        @Override
081        public Klass<?> getSuperClass(Class clazz) {
082            return Klass.java(clazz.getSuperclass());
083        }
084
085        @Override
086        public Iterable<Klass<?>> getAncestors(Class clazz) {
087            // TODO: shall we support interfaces?
088            List<Klass<?>> r = new ArrayList<Klass<?>>();
089            for (; clazz!=null; clazz=clazz.getSuperclass()) {
090                r.add(Klass.java(clazz));
091            }
092            return r;
093        }
094
095        @Override
096        public Class toJavaClass(Class clazz) {
097            return clazz;
098        }
099    };
100}
