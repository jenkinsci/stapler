024package org.kohsuke.stapler.export;
025
026import org.kohsuke.stapler.export.TreePruner.ByDepth;
027
028import java.io.IOException;
029import java.lang.reflect.InvocationTargetException;
030import java.lang.reflect.Type;
031import java.lang.reflect.Array;
032import java.net.URL;
033import java.util.Arrays;
034import java.util.Calendar;
035import java.util.Collection;
036import java.util.Collections;
037import java.util.Date;
038import java.util.HashSet;
039import java.util.Map;
040import java.util.Set;
041import java.util.logging.Level;
042import java.util.logging.Logger;
043
044/**
045 * Exposes one {@link Exported exposed property} of {@link ExportedBean} to
046 * {@link DataWriter}.
047 *  
048 * @author Kohsuke Kawaguchi
049 */
050public abstract class Property implements Comparable<Property> {
051    /**
052     * Name of the property.
053     */
054    public final String name;
055    final ModelBuilder owner;
056    /**
057     * Visibility depth level of this property.
058     *
059     * @see Exported#visibility()
060     */
061    public final int visibility;
062
063    /**
064     * Model to which this property belongs to.
065     * Never null.
066     */
067    public final Model parent;
068
069    /**
070     * @see Exported#inline()
071     */
072    public final boolean inline;
073
074    private String[] verboseMap;
075
076    Property(Model parent, String name, Exported exported) {
077        this.parent = parent;
078        this.owner = parent.parent;
079        this.name = exported.name().length()>1 ? exported.name() : name;
080        int v = exported.visibility();
081        if(v==0)
082            v = parent.defaultVisibility;
083        this.visibility = v;
084        this.inline = exported.inline();
085        String[] s = exported.verboseMap().split("/");
086        if (s.length<2)
087            this.verboseMap = null;
088        else
089            this.verboseMap = s;
090    }
091
092    public int compareTo(Property that) {
093        return this.name.compareTo(that.name);
094    }
095
096    public abstract Type getGenericType();
097    public abstract Class getType();
098
099    /**
100     * Gets the associated javadoc, if any, or null.
101     */
102    public abstract String getJavadoc();
103
104    /**
105     * Writes one property of the given object to {@link DataWriter}.
106     *
107     * @param pruner
108     *      Determines how to prune the object graph tree.
109     */
110    public void writeTo(Object object, TreePruner pruner, DataWriter writer) throws IOException {
111        TreePruner child = pruner.accept(object, this);
112        if (child==null)        return;
113
114        try {
115            writer.name(name);
116            writeValue(getValue(object),child,writer);
117        } catch (IllegalAccessException e) {
118            IOException x = new IOException("Failed to write " + name);
119            x.initCause(e);
120            throw x;
121        } catch (InvocationTargetException e) {
122            IOException x = new IOException("Failed to write " + name);
123            x.initCause(e);
124            throw x;
125        }
126    }
127
128    /**
129     * @deprecated as of 1.139
130     */
131    public void writeTo(Object object, int depth, DataWriter writer) throws IOException {
132        writeTo(object,new ByDepth(depth),writer);
133    }
134
135    /**
136     * Writes one value of the property to {@link DataWriter}.
137     */
138    private void writeValue(Object value, TreePruner pruner, DataWriter writer) throws IOException {
139        writeValue(value,pruner,writer,false);
140    }
141
142    /**
143     * Writes one value of the property to {@link DataWriter}.
144     */
145    private void writeValue(Object value, TreePruner pruner, DataWriter writer, boolean skipIfFail) throws IOException {
146        if(value==null) {
147            writer.valueNull();
148            return;
149        }
150
151        if(value instanceof CustomExportedBean) {
152            writeValue(((CustomExportedBean)value).toExportedObject(),pruner,writer);
153            return;
154        }
155
156        Class c = value.getClass();
157
158        if(STRING_TYPES.contains(c)) {
159            writer.value(value.toString());
160            return;
161        }
162        if(PRIMITIVE_TYPES.contains(c)) {
163            writer.valuePrimitive(value);
164            return;
165        }
166        if(c.getComponentType()!=null) { // array
167            Range r = pruner.getRange();
168            writer.startArray();
169            if (value instanceof Object[]) {
170                // typical case
171                for (Object item : r.apply((Object[]) value)) {
172                    writeValue(item,pruner,writer,true);
173                }
174            } else {
175                // more generic case
176                int len = Math.min(r.max,Array.getLength(value));
177                for (int i=r.min; i<len; i++) {
178                    writeValue(Array.get(value,i),pruner,writer,true);
179                }
180            }
181            writer.endArray();
182            return;
183        }
184        if(value instanceof Collection) {
185            writer.startArray();
186            for (Object item : pruner.getRange().apply((Collection) value)) {
187                writeValue(item,pruner,writer,true);
188            }
189            writer.endArray();
190            return;
191        }
192        if(value instanceof Map) {
193            if (verboseMap!=null) {// verbose form
194                writer.startArray();
195                for (Map.Entry e : ((Map<?,?>) value).entrySet()) {
196                    writer.startObject();
197                    writer.name(verboseMap[0]);
198                    writeValue(e.getKey(),pruner,writer);
199                    writer.name(verboseMap[1]);
200                    writeValue(e.getValue(),pruner,writer);
201                    writer.endObject();
202                }
203                writer.endArray();
204            } else {// compact form
205                writer.startObject();
206                for (Map.Entry e : ((Map<?,?>) value).entrySet()) {
207                    writer.name(e.getKey().toString());
208                    writeValue(e.getValue(),pruner,writer);
209                }
210                writer.endObject();
211            }
212            return;
213        }
214        if(value instanceof Date) {
215            writer.valuePrimitive(((Date) value).getTime());
216            return;
217        }
218        if(value instanceof Calendar) {
219            writer.valuePrimitive(((Calendar) value).getTimeInMillis());
220            return;
221        }
222        if(value instanceof Enum) {
223            writer.value(value.toString());
224            return;
225        }
226
227        // otherwise handle it as a bean
228        writer.startObject();
229        Model model = null;
230        try {
231            model = owner.get(c, parent.type, name);
232        } catch (NotExportableException e) {
233            if (skipIfFail) {
234                Logger.getLogger(Property.class.getName()).log(Level.FINE, e.getMessage());
235            } else {
236                throw e;
237            }
238            // otherwise ignore this error by writing empty object
239        }
240        if(model!=null)
241            model.writeNestedObjectTo(value, pruner, writer, Collections.<String>emptySet());
242        writer.endObject();
243    }
244
245    /**
246     * Gets the value of this property from the bean.
247     */
248    protected abstract Object getValue(Object bean) throws IllegalAccessException, InvocationTargetException;
249
250    /*package*/ static final Set<Class> STRING_TYPES = new HashSet<Class>(Arrays.asList(
251        String.class,
252        URL.class
253    ));
254
255    /*package*/ static final Set<Class> PRIMITIVE_TYPES = new HashSet<Class>(Arrays.asList(
256        Integer.class,
257        Long.class,
258        Boolean.class,
259        Short.class,
260        Character.class,
261        Float.class,
262        Double.class
263    ));
