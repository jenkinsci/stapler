024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyException;
027import org.apache.commons.jelly.Script;
028import org.kohsuke.stapler.AbstractTearOff;
029import static org.kohsuke.stapler.Dispatcher.trace;
030import static org.kohsuke.stapler.Dispatcher.traceable;
031
032import org.kohsuke.stapler.Facet;
033import org.kohsuke.stapler.MetaClass;
034import org.kohsuke.stapler.StaplerRequest;
035import org.kohsuke.stapler.StaplerResponse;
036import org.kohsuke.stapler.WebApp;
037
038import javax.servlet.RequestDispatcher;
039import javax.servlet.ServletException;
040import java.io.IOException;
041import java.net.URL;
042import java.util.logging.Logger;
043
044/**
045 * @author Kohsuke Kawaguchi
046 */
047public class JellyClassTearOff extends AbstractTearOff<JellyClassLoaderTearOff,Script,JellyException> {
048    private JellyFacet facet;
049
050    public JellyClassTearOff(MetaClass owner) {
051        super(owner,JellyClassLoaderTearOff.class);
052        facet = owner.webApp.getFacet(JellyFacet.class);
053    }
054
055    protected Script parseScript(URL res) throws JellyException {
056        return new JellyViewScript(owner.klass, res, classLoader.createContext().compileScript(res));
057    }
058
059    @Override
060    protected String getDefaultScriptExtension() {
061        return ".jelly";
062    }
063
064    @Override
065    protected boolean hasAllowedExtension(String name) {
066        return name.endsWith(".jelly") || name.endsWith(".jellytag");
067    }
068
069    /**
070     * Aside from looking into our own, consult other facets that can handle Jelly-compatible scripts.
071     */
072    @Override
073    public Script resolveScript(String name) throws JellyException {
074        // cut off the extension so that we can search all the extensions
075        String shortName;
076        int dot = name.lastIndexOf('.');
077        if (dot>name.lastIndexOf('/'))      shortName = name.substring(0, dot);
078        else                                shortName = name;
079
080        for (Facet f : owner.webApp.facets) {
081            if (f instanceof JellyCompatibleFacet && !(f instanceof JellyFacet)) {
082                JellyCompatibleFacet jcf = (JellyCompatibleFacet) f;
083                for (Class<? extends AbstractTearOff<?,? extends Script,?>> ct : jcf.getClassTearOffTypes()) {
084                    try {
085                        Script s = owner.loadTearOff(ct).resolveScript(shortName);
086                        if (s!=null)    return s;
087                    } catch (Exception e) {
088                        throw new JellyException("Failed to load "+shortName+" from "+jcf,e);
089                    }
090                }
091            }
092        }
093
094        return super.resolveScript(name);
095    }
096
097    /**
098     * Serves <tt>index.jelly</tt> if it's available, and returns true.
099     */
100    public boolean serveIndexJelly(StaplerRequest req, StaplerResponse rsp, Object node) throws ServletException, IOException {
101        try {
102            Script script = findScript("index.jelly");
103            if(script!=null) {
104                if(traceable()) {
105                    String src = "index.jelly";
106                    if (script instanceof JellyViewScript) {
107                        JellyViewScript jvs = (JellyViewScript) script;
108                        src = jvs.getName();
109                    }
110                    trace(req,rsp,"-> %s on <%s>",src,node);
111                }
112                facet.scriptInvoker.invokeScript(req, rsp, script, node);
113                return true;
114            }
115            return false;
116        } catch (JellyException e) {
117            throw new ServletException(e);
118        }
119    }
120
121    /**
122     * Creates a {@link RequestDispatcher} that forwards to the jelly view, if available.
123     */
124    public RequestDispatcher createDispatcher(Object it, String viewName) throws IOException {
125        try {
126            // backward compatible behavior that expects full file name including ".jelly"
127            Script script = findScript(viewName);
128            if(script!=null)
129                return new JellyRequestDispatcher(it,script);
130            
131            // this is what the look up was really supposed to be.
132            script = findScript(viewName+".jelly");
133            if(script!=null)
134                return new JellyRequestDispatcher(it,script);
135            return null;
136        } catch (JellyException e) {
137            IOException io = new IOException(e.getMessage());
138            io.initCause(e);
139            throw io;
140        }
141    }
142
143    private static final Logger LOGGER = Logger.getLogger(JellyClassTearOff.class.getName());
144}
