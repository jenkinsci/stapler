001package org.kohsuke.stapler.compression;
002
003import javax.servlet.Filter;
004import javax.servlet.FilterChain;
005import javax.servlet.FilterConfig;
006import javax.servlet.ServletContext;
007import javax.servlet.ServletException;
008import javax.servlet.ServletRequest;
009import javax.servlet.ServletResponse;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import java.io.IOException;
013
014/**
015 * Pimps up {@link HttpServletResponse} so that it understands "Content-Encoding: gzip" and compress the response.
016 * 
017 * <p>
018 * When exceptions are processed within web applications, different unrelated parts of the webapp can end up calling
019 * {@link HttpServletResponse#getOutputStream()}. This fundamentally doesn't work with the notion that the application
020 * needs to process "Content-Encoding:gzip" on its own. Such app has to maintain a GZIP output stream on its own,
021 * since {@link HttpServletResponse} doesn't know that its output is written through a compressed stream.
022 *
023 * <p>
024 * Another place this break-down can be seen is when a servlet throws an exception that the container handles.
025 * It tries to render an error page, but of course it wouldn't know that "Content-Encoding:gzip" is set, so it
026 * will fail to write in the correct format.
027 * 
028 * <p>
029 * The only way to properly fix this is to make {@link HttpServletResponse} smart enough that it returns
030 * the compression-transparent stream from {@link HttpServletResponse#getOutputStream()} (and it would also
031 * have to process the exception thrown from the app.) This filter does exactly that.
032 *
033 * @author Kohsuke Kawaguchi
034 */
035public class CompressionFilter implements Filter {
036    private ServletContext context;
037
038    public void init(FilterConfig filterConfig) throws ServletException {
039        context = filterConfig.getServletContext();
040    }
041
042    public void doFilter(ServletRequest _req, ServletResponse _rsp, FilterChain filterChain) throws IOException, ServletException {
043        Object old1 = swapAttribute(_req, CompressionFilter.class, true);
044
045        CompressionServletResponse rsp = new CompressionServletResponse(((HttpServletResponse) _rsp));
046        Object old2 = swapAttribute(_req,CompressionServletResponse.class,rsp);
047
048        try {
049            filterChain.doFilter(_req, rsp);
050        } catch (IOException e) {
051            if (DISABLED)   throw e;
052            reportException(e,(HttpServletRequest)_req,rsp);
053        } catch (ServletException e) {
054            if (DISABLED)   throw e;
055            reportException(e,(HttpServletRequest)_req,rsp);
056        } catch (RuntimeException e) {
057            if (DISABLED)   throw e;
058            reportException(e,(HttpServletRequest)_req,rsp);
059        } catch (Error e) {
060            if (DISABLED)   throw e;
061            reportException(e,(HttpServletRequest)_req,rsp);
062        } finally {
063            rsp.close();
064
065            _req.setAttribute(CompressionFilter.class.getName(),old1);
066            _req.setAttribute(CompressionServletResponse.class.getName(),old2);
067        }
068    }
069
070    private Object swapAttribute(ServletRequest req, Class<?> key, Object value) {
071        Object old = req.getAttribute(key.getName());
072        req.setAttribute(key.getName(), value);
073        return old;
074    }
075
076    private void reportException(Throwable e, HttpServletRequest req, HttpServletResponse rsp) throws IOException, ServletException {
077        getUncaughtExceptionHandler(context).reportException(e, context, req, rsp);
078    }
079
080    public void destroy() {
081    }
082
083    public static void setUncaughtExceptionHandler(ServletContext context, UncaughtExceptionHandler handler) {
084        context.setAttribute(UncaughtExceptionHandler.class.getName(),handler);
085    }
086
087    public static UncaughtExceptionHandler getUncaughtExceptionHandler(ServletContext context) {
088        UncaughtExceptionHandler h = (UncaughtExceptionHandler) context.getAttribute(UncaughtExceptionHandler.class.getName());
089        if (h==null)    h=UncaughtExceptionHandler.DEFAULT;
090        return h;
091    }
092
093    /**
094     * Is this request already wrapped into {@link CompressionFilter}?
095     */
096    public static boolean has(ServletRequest req) {
097        return req.getAttribute(CompressionServletResponse.class.getName())!=null;
098    }
099
100    /**
101     * Is this request already wrapped into {@link CompressionFilter},
102     * activate that, so that {@link ServletResponse#getOutputStream()} will return
103     * a stream that automatically handles compression.
104     */
105    public static boolean activate(ServletRequest req) throws IOException {
106        CompressionServletResponse rsp = (CompressionServletResponse) req.getAttribute(CompressionServletResponse.class.getName());
107        if (rsp!=null) {
108            rsp.activate();
109            return true;
110        } else {
111            return false;
112        }
113    }
114
115    public static boolean DISABLED = Boolean.getBoolean(CompressionFilter.class.getName()+".disabled");
116}
