024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyTagException;
027import org.apache.commons.jelly.XMLOutput;
028
029import java.io.StringWriter;
030
031/**
032 * Body is evaluated and is used as an argument for the surrounding &lt;structuredMessageFormat> element.
033 *
034 * @author Kohsuke Kawaguchi
035 */
036public class StructuredMessageArgumentTag extends AbstractStaplerTag {
037    public void doTag(XMLOutput output) throws JellyTagException {
038        StructuredMessageFormatTag tag = (StructuredMessageFormatTag)findAncestorWithClass(StructuredMessageFormatTag.class);
039        if(tag == null)
040            throw new JellyTagException("This tag must be enclosed inside a <structuredMessageFormat> tag" );
041
042        StringWriter sw = new StringWriter();
043        XMLOutput o = XMLOutput.createXMLOutput(sw);
044        invokeBody(o);
045
046        tag.addArgument(sw);
047    }
048}
