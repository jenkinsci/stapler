024package org.kohsuke.stapler.export;
025
026import java.util.Map;
027import java.util.concurrent.ConcurrentHashMap;
028
029/**
030 * Creates and maintains {@link Model}s, that are used to write out
031 * the value representation of {@link ExportedBean exposed beans}.
032 * @author Kohsuke Kawaguchi
033 */
034public class ModelBuilder {
035    /**
036     * Instantiated {@link Model}s.
037     * Registration happens in {@link Model#Model(ModelBuilder,Class)} so that cyclic references
038     * are handled correctly.
039     */
040    /*package*/ final Map<Class, Model> models = new ConcurrentHashMap<Class, Model>();
041
042    public <T> Model<T> get(Class<T> type) {
043        Model m = models.get(type);
044        if(m==null) {
045            m = new Model<T>(this,type);
046        }
047        return m;
048    }
049}
