001package org.kohsuke.stapler;
002
003import java.util.AbstractList;
004import java.util.Iterator;
005
006/**
007 * Single linked list which allows sharing of the suffix.
008 *
009 * @author Kohsuke Kawaguchi
010 */
011public class SingleLinkedList<T> extends AbstractList<T> {
012    public final T head;
013    public final SingleLinkedList<T> tail;
014
015    public SingleLinkedList(T head, SingleLinkedList<T> tail) {
016        this.head = head;
017        this.tail = tail;
018    }
019
020    /**
021     * Creates a new list by adding a new element as the head.
022     */
023    public SingleLinkedList<T> grow(T item) {
024        return new SingleLinkedList<T>(item,this);
025    }
026
027    @Override
028    public Iterator<T> iterator() {
029        return new Iterator<T>() {
030            SingleLinkedList<T> next = SingleLinkedList.this;
031
032            public boolean hasNext() {
033                return next!=EMPTY_LIST;
034            }
035
036            public T next() {
037                T r = next.head;
038                next = next.tail;
039                return r;
040            }
041
042            public void remove() {
043                throw new UnsupportedOperationException();
044            }
045        };
046    }
047
048    @Override
049    public T get(int index) {
050        return null;
051    }
052
053    @Override
054    public int size() {
055        int sz = 0;
056        for (SingleLinkedList<T> head=this; head!=EMPTY_LIST; head=head.tail)
057            sz++;
058        return sz;
059    }
060
061    @Override
062    public boolean isEmpty() {
063        return this==EMPTY_LIST;
064    }
065
066    public static <T> SingleLinkedList<T> empty() {
067        return EMPTY_LIST;
068    }
069
070    private static final SingleLinkedList EMPTY_LIST = new SingleLinkedList(null,null);
071}
