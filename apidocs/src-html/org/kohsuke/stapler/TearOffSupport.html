024package org.kohsuke.stapler;
025
026import java.lang.reflect.InvocationTargetException;
027import java.util.HashMap;
028import java.util.Map;
029
030/**
031 * Allows "tear-off" objects to be linked to the parent object.
032 *
033 * <p>
034 * This mechanism is used to avoid static linking optional packages,
035 * so that stapler can work even when the optional dependencies are missing.
036 *
037 * @author Kohsuke Kawaguchi
038 */
039public abstract class TearOffSupport {
040    private volatile Map<Class,Object> tearOffs;
041
042    public final <T> T getTearOff(Class<T> t) {
043        Map<Class,Object> m = tearOffs;
044        if(m==null)     return null;
045        return (T)m.get(t);
046    }
047
048    public final <T> T loadTearOff(Class<T> t) {
049        T o = getTearOff(t);
050        if(o==null) {
051            try {
052                o = t.getConstructor(getClass()).newInstance(this);
053                setTearOff(t,o);
054            } catch (InstantiationException e) {
055                throw new InstantiationError(e.getMessage());
056            } catch (IllegalAccessException e) {
057                throw new IllegalAccessError(e.getMessage());
058            } catch (InvocationTargetException e) {
059                Throwable ex = e.getTargetException();
060                if(ex instanceof RuntimeException)
061                    throw (RuntimeException)ex;
062                if(ex instanceof Error)
063                    throw (Error)ex;
064                throw new Error(e);
065            } catch (NoSuchMethodException e) {
066                throw new NoSuchMethodError(e.getMessage());
067            }
068        }
069        return o;
070    }
071
072    public synchronized <T> void setTearOff(Class<T> type, T instance) {
073        Map<Class,Object> m = tearOffs;
074        Map<Class,Object> r = m!=null ? new HashMap<Class, Object>(tearOffs) : new HashMap<Class,Object>();
075        r.put(type,instance);
076        tearOffs = r;
077    }
078}
