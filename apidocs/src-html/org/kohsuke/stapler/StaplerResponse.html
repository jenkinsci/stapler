024package org.kohsuke.stapler;
025
026import edu.umd.cs.findbugs.annotations.NonNull;
027import net.sf.json.JsonConfig;
028import org.kohsuke.stapler.export.Flavor;
029
030import javax.servlet.ServletException;
031import javax.servlet.http.HttpServletResponse;
032import javax.servlet.http.HttpServletRequest;
033import java.io.IOException;
034import java.io.InputStream;
035import java.io.OutputStream;
036import java.io.Writer;
037import java.net.URL;
038import java.net.URLConnection;
039import org.kohsuke.stapler.export.Model;
040import org.kohsuke.stapler.export.NamedPathPruner;
041
042/**
043 * Defines additional operations made available by Stapler.
044 *
045 * @see Stapler#getCurrentResponse()
046 * @author Kohsuke Kawaguchi
047 */
048public interface StaplerResponse extends HttpServletResponse {
049    /**
050     * Evaluates the url against the given object and
051     * forwards the request to the result.
052     *
053     * <p>
054     * This can be used for example inside an action method
055     * to forward a request to a JSP.
056     *
057     * @param it
058     *      the URL is evaluated against this object. Must not be null.
059     * @param url
060     *      the relative URL (e.g., "foo" or "foo/bar") to resolve
061     *      against the "it" object.
062     * @param request
063     *      Request to be forwarded.
064     */
065    void forward(Object it, String url, StaplerRequest request) throws ServletException, IOException;
066    /**
067     * Redirects the browser to where it came from (the referer.)
068     */
069    void forwardToPreviousPage(StaplerRequest request) throws ServletException, IOException;
070
071    /**
072     * Works like {@link #sendRedirect(String)} except that this method
073     * escapes the URL.
074     */
075    void sendRedirect2(@NonNull String url) throws IOException;
076
077    /**
078     * Works like {@link #sendRedirect2(String)} but allows the caller to specify the HTTP status code.
079     */
080    void sendRedirect(int statusCore, @NonNull String url) throws IOException;
081
082    /**
083     * Serves a static resource.
084     *
085     * <p>
086     * This method sets content type, HTTP status code, sends the complete data
087     * and closes the response. This method also handles cache-control HTTP headers
088     * like "If-Modified-Since" and others.
089     */
090    void serveFile(StaplerRequest request, URL res) throws ServletException, IOException;
091
092    void serveFile(StaplerRequest request, URL res, long expiration) throws ServletException, IOException;
093
094    /**
095     * Works like {@link #serveFile(StaplerRequest, URL)} but chooses the locale specific
096     * version of the resource if it's available. The convention of "locale specific version"
097     * is the same as that of property files.
098     * So Japanese resource for <tt>foo.html</tt> would be named <tt>foo_ja.html</tt>.
099     */
100    void serveLocalizedFile(StaplerRequest request, URL res) throws ServletException, IOException;
101
102    /**
103     * Works like {@link #serveFile(StaplerRequest, URL, long)} but chooses the locale
104     * specific version of the resource if it's available.
105     *
106     * See {@link #serveLocalizedFile(StaplerRequest, URL)} for more details.
107     */
108    void serveLocalizedFile(StaplerRequest request, URL res, long expiration) throws ServletException, IOException;
109
110    /**
111     * Serves a static resource.
112     *
113     * <p>
114     * This method works like {@link #serveFile(StaplerRequest, URL)} but this version
115     * allows the caller to fetch data from anywhere.
116     *
117     * @param data
118     *      {@link InputStream} that contains the data of the static resource.
119     * @param lastModified
120     *      The timestamp when the resource was last modified. See {@link URLConnection#getLastModified()}
121     *      for the meaning of the value. 0 if unknown, in which case "If-Modified-Since" handling
122     *      will not be performed.
123     * @param expiration
124     *      The number of milliseconds until the resource will "expire".
125     *      Until it expires the browser will be allowed to cache it
126     *      and serve it without checking back with the server.
127     *      After it expires, the client will send conditional GET to
128     *      check if the resource is actually modified or not.
129     *      If 0, it will immediately expire.
130     * @param contentLength
131     *      if the length of the input stream is known in advance, specify that value
132     *      so that HTTP keep-alive works. Otherwise specify -1 to indicate that the length is unknown.
133     * @param fileName
134     *      file name of this resource. Used to determine the MIME type.
135     *      Since the only important portion is the file extension, this could be just a file name,
136     *      or a full path name, or even a pseudo file name that doesn't actually exist.
137     *      It supports both '/' and '\\' as the path separator.
138     *
139     *      If this string starts with "mime-type:", like "mime-type:foo/bar", then "foo/bar" will
140     *      be used as a MIME type without consulting the servlet container.
141     */
142    void serveFile(StaplerRequest req, InputStream data, long lastModified, long expiration, long contentLength, String fileName) throws ServletException, IOException;
143
144    /**
145     * @Deprecated use form with long contentLength
146     */
147    void serveFile(StaplerRequest req, InputStream data, long lastModified, long expiration, int contentLength, String fileName) throws ServletException, IOException;
148
149    /**
150     * Serves a static resource.
151     *
152     * Expiration date is set to the value that forces browser to do conditional GET
153     * for all resources.
154     *
155     * @see #serveFile(StaplerRequest, InputStream, long, long, int, String)
156     */
157    void serveFile(StaplerRequest req, InputStream data, long lastModified, long contentLength, String fileName) throws ServletException, IOException;
158
159    /**
160     * @Deprecated use form with long contentLength
161     */
162    void serveFile(StaplerRequest req, InputStream data, long lastModified, int contentLength, String fileName) throws ServletException, IOException;
163
164    /**
165     * Serves the exposed bean in the specified flavor.
166     *
167     * <p>
168     * This method performs the complete output from the header to the response body.
169     * If the flavor is JSON, this method also supports JSONP via the {@code jsonp} query parameter.
170     * 
171     * <p>The {@code depth} parameter may be used to specify a recursion depth
172     * as in {@link Model#writeTo(Object,int,DataWriter)}.
173     * 
174     * <p>As of 1.146, the {@code tree} parameter may be used to control the output
175     * in detail; see {@link NamedPathPruner#NamedPathPruner(String)} for details.
176     */
177    void serveExposedBean(StaplerRequest req, Object exposedBean, Flavor flavor) throws ServletException,IOException;
178
179    /**
180     * Works like {@link #getOutputStream()} but tries to send the response
181     * with gzip compression if the client supports it.
182     *
183     * <p>
184     * This method is useful for sending out a large text content.
185     *
186     * @param req
187     *      Used to determine whether the client supports compression
188     */
189    OutputStream getCompressedOutputStream(HttpServletRequest req) throws IOException;
190
191    /**
192     * Works like {@link #getCompressedOutputStream(HttpServletRequest)} but this
193     * method is for {@link #getWriter()}.
194     */
195    Writer getCompressedWriter(HttpServletRequest req) throws IOException;
196
197    /**
198     * Performs the reverse proxy to the given URL.
199     *
200     * @return
201     *      The status code of the response.
202     */
203    int reverseProxyTo(URL url, StaplerRequest req) throws IOException;
204
205    /**
206     * The JsonConfig to be used when serializing java beans from js bound methods to JSON.
207     * Setting this to null will make the default config to be used.
208     *
209     * @param config the config
210     */
211    void setJsonConfig(JsonConfig config);
212
213    /**
214     * The JsonConfig to be used when serializing java beans to JSON previously set by {@link #setJsonConfig(JsonConfig)}.
215     * Will return the default config if nothing has previously been set.
216     *
217     * @return the config
218     */
219    JsonConfig getJsonConfig();
220}
