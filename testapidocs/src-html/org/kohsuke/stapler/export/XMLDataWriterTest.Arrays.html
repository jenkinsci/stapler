025    private <T> String serialize(T bean, Class<T> clazz) throws IOException {
026        StringWriter w = new StringWriter();
027        Model<T> model = new ModelBuilder().get(clazz);
028        model.writeTo(bean, Flavor.XML.createDataWriter(bean, w, config));
029        return w.toString();
030    }

102    @Test
103    public void testNestedBeans() throws Exception {
104        System.out.println(serialize(new Job(), Job.class));
105        assertEquals("<job _class='Job'><action _class='ParameterAction'><parameter><names>foo</names><values>bar</values></parameter></action><action _class='CauseAction'><cause>xyz</cause></action><name>job1</name></job>",
106                serialize(new Job(), Job.class));
107    }

120    @ExportedBean(defaultVisibility=2) public static abstract class Super {
121        @Exported public String basic = "super";
122        @Exported public abstract String generic();
123    }
124    public static class Sub extends Super {
125        public String generic() {return "sub";}
126        @Exported public String specific() {return "sub";}
127    }
128    @ExportedBean public static class Container {
129        @Exported public Super polymorph = new Sub();
130    }
131    public void testInheritance() throws Exception {
132        assertEquals("<container _class='Container'><polymorph _class='Sub'><basic>super</basic><generic>sub</generic>" +
133                "<specific>sub</specific></polymorph></container>",
134                serialize(new Container(), Container.class));
135    }

145    private void assertValidXML(String s) throws Exception {
146        SAXParser p = SAXParserFactory.newInstance().newSAXParser();
147        p.parse(new InputSource(new StringReader(s)),new DefaultHandler());
148    }
