024package org.kohsuke.stapler;
025
026import java.lang.annotation.Annotation;
027import java.util.*;
028
029/**
030 * Immutable list of {@link Function}s.
031 *
032 * @author Kohsuke Kawaguchi
033 */
034public final class FunctionList extends AbstractList<Function> {
035    private final Function[] functions;
036
037    public FunctionList(Function... functions) {
038        this.functions = functions;
039    }
040
041    public FunctionList(Collection<Function> functions) {
042        this.functions = functions.toArray(new Function[0]);
043    }
044
045    private FunctionList filter(Filter f) {
046        List<Function> r = new ArrayList<Function>();
047        for (Function m : functions)
048            if (f.keep(m))
049                r.add(m);
050        return new FunctionList(r.toArray(new Function[0]));
051    }
052
053    @Override
054    public Function get(int index) {
055        return functions[index];
056    }
057
058    public int size() {
059        return functions.length;
060    }
061
062    /**
063     * Compute set unions of two lists.
064     */
065    public FunctionList union(FunctionList that) {
066        Set<Function> combined = new LinkedHashSet<Function>();
067        combined.addAll(Arrays.asList(this.functions));
068        combined.addAll(Arrays.asList(that.functions));
069        return new FunctionList(combined);
070    }
071
072    //public int length() {

075    //
076    //public Method get(int i) {

079
080    public interface Filter {
081        boolean keep(Function m);
082    }
083
084    /**
085     * Returns {@link Function}s that start with the given prefix.
086     */
087    public FunctionList prefix(final String prefix) {
088        return filter(new Filter() {
089            public boolean keep(Function m) {
090                return m.getName().startsWith(prefix);
091            }
092        });
093    }
094
095    /**
096     * Returns {@link Function}s that are annotated with the given annotation.
097     */
098    public FunctionList annotated(final Class<? extends Annotation> ann) {
099        return filter(new Filter() {
100            public boolean keep(Function m) {
101                return m.getAnnotation(ann)!=null;
102            }
103        });
104    }
105
106    /**
107     * Returns {@link Function}s that have the given name.
108     */
109    public FunctionList name(final String name) {
110        return filter(new Filter() {
111            public boolean keep(Function m) {
112                return m.getName().equals(name);
113            }
114        });
115    }
116
117    /**
118     * Returns {@link Function}s that has the given type parameters
119     */
120    public FunctionList signature(final Class... args) {
121        return filter(new Filter() {
122            public boolean keep(Function m) {
123                return Arrays.equals(m.getParameterTypes(), args);
124            }
125        });
126    }
127
128    /**
129     * Returns {@link Function}s that are either explicitly {@link WebMethod} or
130     * implicitly so (by having its name start with 'do')
131     */
132    public FunctionList webMethods() {
133        return filter(new Filter() {
134            public boolean keep(Function m) {
135                return m.getName().startsWith("do") || m.getAnnotation(WebMethod.class)!=null;
136            }
137        });
138    }
139
140    /**
141     * Returns {@link Function}s that has the parameters
142     * that start with given types (but can have additional parameters.)
143     */
144    public FunctionList signatureStartsWith(final Class... args) {
145        return filter(new Filter() {
146            public boolean keep(Function m) {
147                Class[] params = m.getParameterTypes();
148                if(params.length<args.length)  return false;
149                for( int i=0; i<args.length; i++ ) {
150                    if(params[i]!=args[i])
151                        return false;
152                }
153                return true;
154            }
155        });
156    }
157}
