001package org.kohsuke.stapler;
002
003import javax.servlet.ServletException;
004import java.lang.annotation.Annotation;
005import java.lang.reflect.InvocationTargetException;
006import java.lang.reflect.Type;
007
008/**
009 * {@link Function} that forwards calls to another. Usually used
010 * as a basis of decorator.
011 *
012 * @author Kohsuke Kawaguchi
013 */
014public class ForwardingFunction extends Function {
015    protected final Function next;
016
017    public ForwardingFunction(Function next) {
018        this.next = next;
019    }
020
021    public String getName() {
022        return next.getName();
023    }
024
025    public String getDisplayName() {
026        return next.getDisplayName();
027    }
028
029    @Override
030    public String getQualifiedName() {
031        return next.getQualifiedName();
032    }
033
034    public Class[] getParameterTypes() {
035        return next.getParameterTypes();
036    }
037
038    @Override
039    public Class getReturnType() {
040        return next.getReturnType();
041    }
042
043    @Override
044    public Type[] getGenericParameterTypes() {
045        return next.getGenericParameterTypes();
046    }
047
048    public Annotation[][] getParameterAnnotations() {
049        return next.getParameterAnnotations();
050    }
051
052    public String[] getParameterNames() {
053        return next.getParameterNames();
054    }
055
056    public Object invoke(StaplerRequest req, StaplerResponse rsp, Object o, Object... args) throws IllegalAccessException, InvocationTargetException, ServletException {
057        return next.invoke(req, rsp, o, args);
058    }
059
060    public <A extends Annotation> A getAnnotation(Class<A> annotation) {
061        return next.getAnnotation(annotation);
062    }
063
064    @Override
065    public Annotation[] getAnnotations() {
066        return next.getAnnotations();
067    }
068}
