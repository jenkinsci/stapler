024package org.kohsuke.stapler.tags;
025
026import javax.servlet.RequestDispatcher;
027import javax.servlet.ServletConfig;
028import javax.servlet.ServletContext;
029import javax.servlet.ServletException;
030import javax.servlet.http.HttpServletRequest;
031import javax.servlet.http.HttpServletResponse;
032import javax.servlet.http.HttpServletResponseWrapper;
033import javax.servlet.jsp.JspException;
034import javax.servlet.jsp.PageContext;
035import javax.servlet.jsp.tagext.SimpleTagSupport;
036import java.io.IOException;
037import java.io.PrintWriter;
038
039/**
040 * Includes a side JSP file for the "it" object.
041 *
042 * <p>
043 * This tag looks for a side JSP file of the given name
044 * from the inheritance hierarchy of the it object,
045 * and includes the contents of it, just like {@code <jsp:include>}.
046 *
047 * <p>
048 * For example, if the "it" object is an instance of the <tt>Foo</tt> class,
049 * which looks like the following:
050 *
051 * <pre>{@code
052 * class Foo extends Bar { ... }
053 * class Bar extends Zot { ... }
054 * }</pre>
055 *
056 * <p>
057 * And if you write:
058 * <pre>{@code
059 * <st:include page="abc.jsp"/>
060 * }</pre>
061 * then, it looks for the following files in this order,
062 * and includes the first one found.
063 * <ol>
064 *  <li>a side-file of the Foo class named abc.jsp (/WEB-INF/side-files/Foo/abc.jsp)
065 *  <li>a side-file of the Bar class named abc.jsp (/WEB-INF/side-files/Bar/abc.jsp)
066 *  <li>a side-file of the Zot class named abc.jsp (/WEB-INF/side-files/Zot/abc.jsp)
067 * </ol>
068 *
069 * @author Kohsuke Kawaguchi
070 */
071public class Include extends SimpleTagSupport {
072
073    private Object it;
074
075    private String page;
076
077    /**
078     * Specifies the name of the JSP to be included.
079     */
080    public void setPage(String page) {
081        this.page = page;
082    }
083
084    /**
085     * Specifies the object for which JSP will be included.
086     */
087    public void setIt(Object it) {
088        this.it = it;
089    }
090
091    private Object getPageObject( String name ) {
092        return getJspContext().getAttribute(name,PageContext.PAGE_SCOPE);
093    }
094
095    public void doTag() throws JspException, IOException {
096        Object it = getJspContext().getAttribute("it",PageContext.REQUEST_SCOPE);
097        final Object oldIt = it;
098        if(this.it!=null)
099            it = this.it;
100
101        ServletConfig cfg = (ServletConfig) getPageObject(PageContext.CONFIG);
102        ServletContext sc = cfg.getServletContext();
103
104        for( Class c = it.getClass(); c!=Object.class; c=c.getSuperclass() ) {
105            String name = "/WEB-INF/side-files/"+c.getName().replace('.','/')+'/'+page;
106            if(sc.getResource(name)!=null) {
107                // Tomcat returns a RequestDispatcher even if the JSP file doesn't exist.
108                // so check if the resource exists first.
109                RequestDispatcher disp = sc.getRequestDispatcher(name);
110                if(disp!=null) {
111                    getJspContext().setAttribute("it",it,PageContext.REQUEST_SCOPE);
112                    try {
113                        HttpServletRequest request = (HttpServletRequest) getPageObject(PageContext.REQUEST);
114                        disp.include(
115                            request,
116                            new Wrapper(
117                                (HttpServletResponse)getPageObject(PageContext.RESPONSE),
118                                new PrintWriter(getJspContext().getOut()) )
119                        );
120                    } catch (ServletException e) {
121                        throw new JspException(e);
122                    } finally {
123                        getJspContext().setAttribute("it",oldIt,PageContext.REQUEST_SCOPE);
124                    }
125                    return;
126                }
127            }
128        }
129
130        throw new JspException("Unable to find '"+page+"' for "+it.getClass());
131    }
132}
133
134class Wrapper extends HttpServletResponseWrapper {
135    private final PrintWriter pw;
136
137    public Wrapper(HttpServletResponse httpServletResponse, PrintWriter w) {
138        super(httpServletResponse);
139        this.pw = w;
140    }
141
142    public PrintWriter getWriter() throws IOException {
143        return pw;
144    }
145}
