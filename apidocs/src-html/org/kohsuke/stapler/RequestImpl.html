201    @Override
202    public String[] getParameterValues(String name) {
203        if(!isMultipart()) {
204            return super.getParameterValues(name);
205        }
206        Map<String, String> data = getFormDataFormFields();
207        if (data.isEmpty()) {
208            return super.getParameterValues(name);
209        }
210        
211        String formFieldVal = data.get(name);
212        if (formFieldVal == null) {
213            return super.getParameterValues(name);
214        }
215
216        String[] values = super.getParameterValues(name);
217        if (values == null) {
218            values = new String[0];
219        }
220        String[] extValues = new String[values.length + 1];
221        System.arraycopy(values, 0, extValues, 0, values.length);
222        extValues[extValues.length - 1] = formFieldVal;
223
224        return extValues;
225    }

321    public boolean checkIfModified(long lastModified, StaplerResponse rsp, long expiration) {
322        if(lastModified<=0)
323            return false;
324
325        // send out Last-Modified, or check If-Modified-Since
326        String since = getHeader("If-Modified-Since");
327        SimpleDateFormat format = Stapler.HTTP_DATE_FORMAT.get();
328        if(since!=null) {
329            try {
330                long ims = format.parse(since).getTime();
331                if(lastModified<ims+1000) {
332                    // +1000 because date header is second-precision and Java has milli-second precision
333                    rsp.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
334                    return true;
335                }
336            } catch (NumberFormatException e) {
337                // just ignore and serve the content
338            } catch (ParseException e) {
339                // just ignore and serve the content
340            }
341        }
342        String tm = format.format(new Date(lastModified));
343        rsp.setHeader("Last-Modified", tm);
344        if(expiration==0) {
345            // don't let browsers
346            rsp.setHeader("Expires", tm);
347        } else {
348            // expire in "NOW+expiration" 
349            rsp.setHeader("Expires",format.format(new Date(new Date().getTime()+expiration)));
350        }
351        return false;
352    }

393    public <T>
394    List<T> bindParametersToList(Class<T> type, String prefix) {
395        List<T> r = new ArrayList<T>();
396
397        int len = Integer.MAX_VALUE;
398
399        Enumeration e = getParameterNames();
400        while(e.hasMoreElements()) {
401            String name = (String)e.nextElement();
402            if(name.startsWith(prefix))
403                len = Math.min(len,getParameterValues(name).length);
404        }
405
406        if(len==Integer.MAX_VALUE)
407            return r;   // nothing
408
409        try {
410            new ClassDescriptor(type).loadConstructorParamNames();
411            // use the designated constructor for databinding
412            for( int i=0; i<len; i++ )
413                r.add(bindParameters(type,prefix,i));
414        } catch (NoStaplerConstructorException _) {
415            // no designated data binding constructor. use reflection
416            try {
417                for( int i=0; i<len; i++ ) {
418                    T t = type.newInstance();
419                    r.add(t);
420
421                    e = getParameterNames();
422                    while(e.hasMoreElements()) {
423                        String name = (String)e.nextElement();
424                        if(name.startsWith(prefix))
425                            fill(t, name.substring(prefix.length()), getParameterValues(name)[i] );
426                    }
427                }
428            } catch (InstantiationException x) {
429                throw new InstantiationError(x.getMessage());
430            } catch (IllegalAccessException x) {
431                throw new IllegalAccessError(x.getMessage());
432            }
433        }
434
435        return r;
436    }

442    public <T> T bindParameters(Class<T> type, String prefix, int index) {
443        String[] names = new ClassDescriptor(type).loadConstructorParamNames();
444
445        // the actual arguments to invoke the constructor with.
446        Object[] args = new Object[names.length];
447
448        // constructor
449        Constructor<T> c = findConstructor(type, names.length);
450        Class[] types = c.getParameterTypes();
451
452        // convert parameters
453        for( int i=0; i<names.length; i++ ) {
454            String[] values = getParameterValues(prefix + names[i]);
455            String param;
456            if(values!=null)
457                param = values[index];
458            else
459                param = null;
460
461            Converter converter = Stapler.lookupConverter(types[i]);
462            if (converter==null)
463                throw new IllegalArgumentException("Unable to convert to "+types[i]);
464
465            args[i] = converter.convert(types[i],param);
466        }
467
468        return invokeConstructor(c, args);
469    }

479    public void bindJSON(Object bean, JSONObject src) {
480        try {
481            for( String key : (Set<String>)src.keySet() ) {
482                TypePair type = getPropertyType(bean, key);
483                if(type==null)
484                    continue;
485
486                fill(bean,key, type.convertJSON(src.get(key)));
487            }
488        } catch (IllegalAccessException e) {
489            IllegalAccessError x = new IllegalAccessError(e.getMessage());
490            x.initCause(e);
491            throw x;
492        } catch (InvocationTargetException x) {
493            Throwable e = x.getTargetException();
494            if(e instanceof RuntimeException)
495                throw (RuntimeException)e;
496            if(e instanceof Error)
497                throw (Error)e;
498            throw new RuntimeException(x);
499        }
500    }

502    public <T> List<T> bindJSONToList(Class<T> type, Object src) {
503        ArrayList<T> r = new ArrayList<T>();
504        if (src instanceof JSONObject) {
505            JSONObject j = (JSONObject) src;
506            r.add(bindJSON(type,j));
507        }
508        if (src instanceof JSONArray) {
509            JSONArray a = (JSONArray) src;
510            for (Object o : a) {
511                if (o instanceof JSONObject) {
512                    JSONObject j = (JSONObject) o;
513                    r.add(bindJSON(type,j));
514                }
515            }
516        }
517        return r;
518    }

521    private <T> T invokeConstructor(Constructor<T> c, Object[] args) {
522        try {
523            return c.newInstance(args);
524        } catch (InstantiationException e) {
525            InstantiationError x = new InstantiationError(e.getMessage());
526            x.initCause(e);
527            throw x;
528        } catch (IllegalAccessException e) {
529            IllegalAccessError x = new IllegalAccessError(e.getMessage());
530            x.initCause(e);
531            throw x;
532        } catch (InvocationTargetException e) {
533            Throwable x = e.getTargetException();
534            if(x instanceof Error)
535                throw (Error)x;
536            if(x instanceof RuntimeException)
537                throw (RuntimeException)x;
538            throw new IllegalArgumentException(x);
539        } catch (IllegalArgumentException e) {
540            throw new IllegalArgumentException("Failed to invoke "+c+" with "+ Arrays.asList(args),e);
541        }
542    }

544    private <T> Constructor<T> findConstructor(Class<T> type, int length) {
545        Constructor<?>[] ctrs = type.getConstructors();
546        // one with DataBoundConstructor is the most reliable
547        for (Constructor c : ctrs) {
548            if(c.getAnnotation(DataBoundConstructor.class)!=null) {
549                if(c.getParameterTypes().length!=length)
550                    throw new IllegalArgumentException(c+" has @DataBoundConstructor but it doesn't match with your .stapler file. Try clean rebuild");
551                return c;
552            }
553        }
554        // if not, maybe this was from @stapler-constructor,
555        // so look for the constructor with the expected argument length.
556        // this is not very reliable.
557        for (Constructor c : ctrs) {
558            if(c.getParameterTypes().length==length)
559                return c;
560        }
561        throw new IllegalArgumentException(type+" does not have a constructor with "+length+" arguments");
562    }

564    private static void fill(Object bean, String key, Object value) {
565        StringTokenizer tokens = new StringTokenizer(key);
566        while(tokens.hasMoreTokens()) {
567            String token = tokens.nextToken();
568            boolean last = !tokens.hasMoreTokens();  // is this the last token?
569
570            try {
571                if(last) {
572                    copyProperty(bean,token,value);
573                } else {
574                    bean = BeanUtils.getProperty(bean,token);
575                }
576            } catch (IllegalAccessException x) {
577                throw new IllegalAccessError(x.getMessage());
578            } catch (InvocationTargetException x) {
579                Throwable e = x.getTargetException();
580                if(e instanceof RuntimeException)
581                    throw (RuntimeException)e;
582                if(e instanceof Error)
583                    throw (Error)e;
584                throw new RuntimeException(x);
585            } catch (NoSuchMethodException e) {
586                // ignore if there's no such property
587            }
588        }
589    }

593     */
594    private final class TypePair {
595        final Type genericType;
596        /**
597         * Erasure of {@link #genericType}
598         */
599        final Class type;
600
601        TypePair(Type genericType, Class type) {
602            this.genericType = genericType;
603            this.type = type;
604        }
605
606        TypePair(Field f) {
607            this(f.getGenericType(),f.getType());
608        }
609
610        /**
611         * Converts the given JSON object (either {@link JSONObject}, {@link JSONArray}, or other primitive types
612         * in JSON, to the type represented by the 'this' object.
613         */
614        public Object convertJSON(Object o) {
615            Object r = bindInterceptor.onConvert(genericType, type, o);
616            if (r!= BindInterceptor.DEFAULT)    return r; // taken over by the interceptor
617
618            for (BindInterceptor i : getWebApp().bindInterceptors) {
619                r = i.onConvert(genericType, type, o);
620                if (r!= BindInterceptor.DEFAULT)    return r; // taken over by the interceptor
621            }
622
623            if(o==null || o instanceof JSONNull) {
624                // this method returns null if the type is not primitive, which works.
625                return ReflectionUtils.getVmDefaultValueFor(type);
626            }
627
628            if (type==JSONArray.class) {
629                if (o instanceof JSONArray) return o;
630
631                JSONArray a = new JSONArray();
632                a.add(o);
633                return a;
634            }
635
636            Lister l = Lister.create(type,genericType);
637
638            if (o instanceof JSONObject) {
639                JSONObject j = (JSONObject) o;
640
641                if (j.isNullObject())   // another flavor of null. json-lib sucks.
642                    return ReflectionUtils.getVmDefaultValueFor(type);
643
644                if(l==null) {// single value conversion
645                    try {
646                        Class actualType = type;
647                        String className = null;
648                        if(j.has("stapler-class")) {
649                            // deprecated as of 2.4-jenkins-4 but left here for a while until we are sure nobody uses this
650                            className = j.getString("stapler-class");
651                            LOGGER.log(FINE, "stapler-class is deprecated in favor of $class: {0}", className);
652                        }
653                        if(j.has("$class")) {
654                            className = j.getString("$class");
655                        }
656
657                        if (className != null) {
658                            // sub-type is specified in JSON.
659                            // note that this can come from malicious clients, so we need to make sure we don't have security issues.
660
661                            ClassLoader cl = stapler.getWebApp().getClassLoader();
662                            try {
663                                Class<?> subType = cl.loadClass(className);
664                                if(!actualType.isAssignableFrom(subType))
665                                    throw new IllegalArgumentException("Specified type "+subType+" is not assignable to the expected "+actualType);
666                                actualType = (Class)subType; // I'm being lazy here
667                            } catch (ClassNotFoundException e) {
668                                throw new IllegalArgumentException("Class "+className+" is specified in JSON, but no such class found in "+cl,e);
669                            }
670                        }
671
672                        return instantiate(actualType, j);
673                    } catch (IllegalArgumentException e) {
674                        throw new IllegalArgumentException("Failed to instantiate "+type+" from "+j,e);
675                    }
676                } else {// collection conversion
677                    if(j.has("stapler-class-bag")) {
678                        // this object is a hash from class names to their parameters
679                        // build them into a collection via Lister
680
681                        ClassLoader cl = stapler.getWebApp().getClassLoader();
682                        for (Map.Entry<String,Object> e : (Set<Map.Entry<String,Object>>)j.entrySet()) {
683                            Object v = e.getValue();
684
685                            String className = e.getKey().replace('-','.'); // decode JSON-safe class name escaping
686                            try {
687                                Class<?> itemType = cl.loadClass(className);
688                                if (v instanceof JSONObject) {
689                                    l.add(bindJSON(itemType, (JSONObject) v));
690                                }
691                                if (v instanceof JSONArray) {
692                                    for(Object i : bindJSONToList(itemType, (JSONArray) v))
693                                        l.add(i);
694                                }
695                            } catch (ClassNotFoundException e1) {
696                                // ignore unrecognized element
697                            }
698                        }
699                    } else if (Enum.class.isAssignableFrom(l.itemType)) {
700                        // this is a hash of element names as enum constant names
701                        for (Map.Entry<String,Object> e : (Set<Map.Entry<String,Object>>)j.entrySet()) {
702                            Object v = e.getValue();
703                            if (v==null || (v instanceof Boolean && !(Boolean)v))
704                                continue;       // skip if the value is null or false
705
706                            l.add(Enum.valueOf(l.itemType,e.getKey()));
707                        }
708                    } else {
709                        // only one value given to the collection
710                        l.add(new TypePair(l.itemGenericType,l.itemType).convertJSON(j));
711                    }
712                    return l.toCollection();
713                }
714            }
715            if (o instanceof JSONArray) {
716                JSONArray a = (JSONArray) o;
717                TypePair itemType = new TypePair(l.itemGenericType,l.itemType);
718                for (Object item : a)
719                    l.add(itemType.convertJSON(item));
720                return l.toCollection();
721            }
722
723            if(Enum.class.isAssignableFrom(type))
724                return Enum.valueOf(type,o.toString());
725
726            if (l==null) {// single value conversion
727                Converter converter = Stapler.lookupConverter(type);
728                if (converter==null)
729                    throw new IllegalArgumentException("Unable to convert to "+type);
730
731                return converter.convert(type,o);
732            } else {// single value in a collection
733                Converter converter = Stapler.lookupConverter(l.itemType);
734                if (converter==null)
735                    throw new IllegalArgumentException("Unable to convert to "+l.itemType);
736
737                l.add(converter.convert(type,o));
738                return l.toCollection();
739            }
740        }
741    }

743    /**
744     * Called after the actual type of the binding is figured out.
745     */
746    private Object instantiate(Class actualType, JSONObject j) {
747        Object r = bindInterceptor.instantiate(actualType,j);
748        if (r!=BindInterceptor.DEFAULT) return r;
749        for (BindInterceptor bi : getWebApp().bindInterceptors) {
750            r = bi.instantiate(actualType,j);
751            if (r!=BindInterceptor.DEFAULT) return r;
752        }
753
754        if (actualType==JSONObject.class || actualType==JSON.class) return actualType.cast(j);
755
756        String[] names = new ClassDescriptor(actualType).loadConstructorParamNames();
757
758        // the actual arguments to invoke the constructor with.
759        Object[] args = new Object[names.length];
760
761        // constructor
762        Constructor c = findConstructor(actualType, names.length);
763        Class[] types = c.getParameterTypes();
764        Type[] genTypes = c.getGenericParameterTypes();
765
766        // convert parameters
767        for( int i=0; i<names.length; i++ ) {
768            try {
769                args[i] = bindJSON(genTypes[i],types[i],j.get(names[i]));
770            } catch (IllegalArgumentException e) {
771                throw new IllegalArgumentException("Failed to convert the "+names[i]+" parameter of the constructor "+c,e);
772            }
773        }
774
775        Object o = injectSetters(invokeConstructor(c, args), j, Arrays.asList(names));
776        o = bindResolve(o,j);
777
778        return o;
779    }

782     * Calls {@link DataBoundResolvable#bindResolve(StaplerRequest, JSONObject)} if the object has it.

784    private Object bindResolve(Object o, JSONObject src) {

786            DataBoundResolvable dbr = (DataBoundResolvable) o;

792    /**
793     * Performs {@link DataBoundSetter} injections.
794     *
795     * @param exclusions
796     *      Properties that are already injected through the constructor, thus not subject of the setter injection.
797     */
798    private <T> T injectSetters(T r, JSONObject j, Collection<String> exclusions) {
799        // try to assign rest of the properties
800        OUTER:
801        for (String key : (Set<String>)j.keySet()) {
802            if (!exclusions.contains(key)) {
803                try {
804                    // try field injection first
805                    for (Class c=r.getClass(); c!=null; c=c.getSuperclass()) {
806                        try {
807                            Field f = c.getDeclaredField(key);
808                            if (f.getAnnotation(DataBoundSetter.class)!=null) {
809                                f.setAccessible(true);
810                                f.set(r, bindJSON(f.getGenericType(), f.getType(), j.get(key)));
811                                continue OUTER;
812                            }
813                        } catch (NoSuchFieldException e) {
814                            // recurse into parents
815                        }
816                    }
817
818                    PropertyDescriptor pd = PropertyUtils.getPropertyDescriptor(r, key);
819                    if (pd==null)   continue;
820
821                    Method wm = pd.getWriteMethod();
822                    if (wm==null)   continue;
823                    if (wm.getAnnotation(DataBoundSetter.class)==null) {
824                        LOGGER.warning("Tried to set value to "+key+" but method "+wm+" is missing @DataBoundSetter");
825                        continue;
826                    }
827
828                    Class<?>[] pt = wm.getParameterTypes();
829
830                    if (pt.length!=1) {
831                        LOGGER.log(WARNING, "Setter method "+wm+" is expected to have 1 parameter");
832                        continue;
833                    }
834
835                    // only invoking public methods for security reasons
836                    wm.invoke(r, bindJSON(wm.getGenericParameterTypes()[0], pt[0], j.get(key)));
837                } catch (IllegalAccessException e) {
838                    LOGGER.log(WARNING, "Cannot access property " + key + " of " + r.getClass(), e);
839                } catch (InvocationTargetException e) {
840                    LOGGER.log(WARNING, "Cannot access property " + key + " of " + r.getClass(), e);
841                } catch (NoSuchMethodException e) {
842                    LOGGER.log(WARNING, "Cannot access property " + key + " of " + r.getClass(), e);
843                }
844            }
845        }
846
847        invokePostConstruct(getWebApp().getMetaClass(r).getPostConstructMethods(), r);
848
849        return r;
850    }

852    /**
853     * Invoke PostConstruct method from the base class to subtypes.
854     */
855    private void invokePostConstruct(SingleLinkedList<MethodRef> methods, Object r) {
856        if (methods.isEmpty())  return;
857
858        invokePostConstruct(methods.tail,r);
859        try {
860            methods.head.invoke(r);
861        } catch (InvocationTargetException e) {
862            throw new IllegalArgumentException("Unable to post-construct "+r,e);
863        } catch (IllegalAccessException e) {
864            throw (Error)new IllegalAccessError().initCause(e);
865        }
866
867    }

869    /**
870     * Gets the type of the field/property designate by the given name.
871     */
872    private TypePair getPropertyType(Object bean, String name) throws IllegalAccessException, InvocationTargetException {
873        try {
874            PropertyDescriptor propDescriptor = PropertyUtils.getPropertyDescriptor(bean, name);
875            if(propDescriptor!=null) {
876                Method m = propDescriptor.getWriteMethod();
877                if(m!=null)
878                    return new TypePair(m.getGenericParameterTypes()[0], m.getParameterTypes()[0]);
879            }
880        } catch (NoSuchMethodException e) {
881            // no such property
882        }
883
884        // try a field
885        try {
886            return new TypePair(bean.getClass().getField(name));
887        } catch (NoSuchFieldException e) {
888            // no such field
889        }
890
891        return null;
892    }

894    /**
895     * Sets the property/field value of the given name, by performing a value type conversion if necessary.
896     */
897    private static void copyProperty(Object bean, String name, Object value) throws IllegalAccessException, InvocationTargetException {
898        PropertyDescriptor propDescriptor;
899        try {
900            propDescriptor =
901                PropertyUtils.getPropertyDescriptor(bean, name);
902        } catch (NoSuchMethodException e) {
903            propDescriptor = null;
904        }
905        if ((propDescriptor != null) &&
906            (propDescriptor.getWriteMethod() == null)) {
907            propDescriptor = null;
908        }
909        if (propDescriptor != null) {
910            Converter converter = Stapler.lookupConverter(propDescriptor.getPropertyType());
911            if (converter != null)
912                value = converter.convert(propDescriptor.getPropertyType(), value);
913            try {
914                PropertyUtils.setSimpleProperty(bean, name, value);
915            } catch (NoSuchMethodException e) {
916                throw new NoSuchMethodError(e.getMessage());
917            }
918            return;
919        }
920
921        // try a field
922        try {
923            Field field = bean.getClass().getField(name);
924            Converter converter = ConvertUtils.lookup(field.getType());
925            if (converter != null)
926                value = converter.convert(field.getType(), value);
927            field.set(bean,value);
928        } catch (NoSuchFieldException e) {
929            // no such field
930        }
931    }

933    private void parseMultipartFormData() throws ServletException {
934        if(parsedFormData!=null)    return;
935
936        parsedFormData = new HashMap<String,FileItem>();
937        parsedFormDataFormFields = new HashMap<String, String>();
938        ServletFileUpload upload = new ServletFileUpload(new DiskFileItemFactory());
939        try {
940            for( FileItem fi : (List<FileItem>)upload.parseRequest(this) ) {
941                parsedFormData.put(fi.getFieldName(),fi);
942                if (fi.isFormField()) {
943                    parsedFormDataFormFields.put(fi.getFieldName(),fi.getString());
944                }
945            }
946        } catch (FileUploadException e) {
947            throw new ServletException(e);
948        }
949    }

952     * A version of parseMultipartFormData() that doesn't throw exceptions.

954    private Map<String, String> getFormDataFormFields() {
955        try {

957        } catch (Exception e) {
958            LOGGER.log(SEVERE, "Error parsing multipart/form-data.", e);
959        }
960        return parsedFormDataFormFields;

963    public JSONObject getSubmittedForm() throws ServletException {
964        if(structuredForm==null) {
965            String p = null;
966            boolean isSubmission; // for error diagnosis, if something is submitted, set to true
967
968            if(isMultipart()) {
969                isSubmission=true;
970                parseMultipartFormData();
971                FileItem item = parsedFormData.get("json");
972                if(item!=null) {
973                    if (item.getContentType() == null && getCharacterEncoding() != null) {
974                        // JENKINS-11543: If client doesn't set charset per part, use request encoding
975                        try {
976                            p = item.getString(getCharacterEncoding());
977                        } catch (java.io.UnsupportedEncodingException uee) {
978                            LOGGER.log(WARNING, "Request has unsupported charset, using default for 'json' parameter", uee);
979                            p = item.getString();
980                        }
981                    } else {
982                        p = item.getString();
983                    }
984                }
985            } else {
986                p = getParameter("json");
987                isSubmission = !getParameterMap().isEmpty();
988            }
989            
990            if(p==null || p.length() == 0) {
991                // no data submitted
992                try {
993                    StaplerResponse rsp = Stapler.getCurrentResponse();
994                    if(isSubmission)
995                        rsp.sendError(SC_BAD_REQUEST,"This page expects a form submission");
996                    else
997                        rsp.sendError(SC_BAD_REQUEST,"Nothing is submitted");
998                    throw new ServletException("This page expects a form submission but had only " + getParameterMap());
999                } catch (IOException e) {
1000                    throw new ServletException(e);
1001                }
1002            }
1003            try {
1004                structuredForm = JSONObject.fromObject(p);
1005            } catch (JSONException e) {
1006                throw new ServletException("Failed to parse JSON:" + p, e);
1007            }
1008        }
1009        return structuredForm;
1010    }

1013        String ct = getContentType();
