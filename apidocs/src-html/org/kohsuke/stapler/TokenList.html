024package org.kohsuke.stapler;
025
026import java.util.StringTokenizer;
027import java.io.ByteArrayOutputStream;
028import java.io.UnsupportedEncodingException;
029
030/**
031 * Tokenized path portion of the URL.
032 *
033 * For example "foo/bar/zot" is treated as ["foo","bar","zot"]
034 *
035 * @author Kohsuke Kawaguchi
036 */
037public final class TokenList {
038    /**
039     * URL-decoded tokens.
040     */
041    public final String[] tokens;
042    /**
043     * Like {@link #tokens} but before decoding.
044     */
045    public final String[] rawTokens;
046    /**
047     * Index of the next token.
048     */
049    public int idx;
050
051    /**
052     * If the request URL ends with the path separator.
053     */
054    public final boolean endsWithSlash;
055
056    TokenList(String url) {
057        // to avoid a directory traversal vulnerability in Windows, treat '\\' as a path separator just like '/'
058        StringTokenizer tknzr = new StringTokenizer(url,"/\\");
059        final int tokenCount = tknzr.countTokens();
060        tokens = new String[tokenCount];
061        rawTokens = new String[tokenCount];
062        for(int i=0; tknzr.hasMoreTokens(); i++) {
063            rawTokens[i] = tknzr.nextToken();
064            tokens[i] = decode(rawTokens[i]);
065            if (tokens[i].equals(".."))
066                throw new IllegalArgumentException(url);
067        }
068        endsWithSlash = url.endsWith("/") || url.endsWith("\\");
069    }
070
071    public boolean hasMore() {
072        return tokens.length!=idx;
073    }
074
075    public String peek() {
076        if(hasMore())
077            return tokens[idx];
078        else
079            return null;
080    }
081
082    public String next() {
083        return tokens[idx++];
084    }
085    public String prev() {
086        return tokens[--idx];
087    }
088    public int nextAsInt() throws NumberFormatException {
089        String p = peek();
090        if(p==null)
091            throw new NumberFormatException();  // no more token
092        int i = Integer.valueOf(p);
093        idx++;
094        return i;
095    }
096
097    public int length() {
098        return tokens.length;
099    }
100
101    public String get(int i) {
102        return tokens[i];
103    }
104
105    public int countRemainingTokens() {
106        return length()-idx;
107    }
108
109
110    public String toString() {
111        StringBuilder buf = new StringBuilder();
112        for( int i=0; i<tokens.length; i++) {
113            if(buf.length()>0)  buf.append('/');
114            if(i==idx)  buf.append('!');
115            buf.append(tokens[i]);
116        }
117        return buf.toString();
118    }
119
120    private String assembleRestOfPath(String[] tokens) {
121        StringBuilder buf = new StringBuilder();
122        for( int i=idx; i<length(); i++ ) {
123            buf.append('/');
124            buf.append(tokens[i]);
125        }
126        return buf.toString();
127    }
128
129    public String assembleRestOfPath() {
130        return assembleRestOfPath(tokens);
131    }
132
133    public String assembleOriginalRestOfPath() {
134        return assembleRestOfPath(rawTokens);
135    }
136
137    public static String decode(String s) {
138        int i = s.indexOf('%');
139        if (i < 0) return s;
140
141        try {
142            // to properly handle non-ASCII characters, decoded bytes need to be stored and translated in bulk.
143            // this complex set up is necessary for us to work gracefully if 's' already contains decoded non-ASCII chars.
144            ByteArrayOutputStream baos = new ByteArrayOutputStream();
145
146            StringBuilder buf = new StringBuilder(s.substring(0, i));
147            char c, upper, lower;
148            for (int m = s.length(); i < m; i++) {
149                c = s.charAt(i);
150                if (c == '%') {
151                    try {
152                        upper = s.charAt(++i);
153                        lower = s.charAt(++i);
154                        baos.write(fromHex(upper) * 16 + fromHex(lower));
155                    } catch (IndexOutOfBoundsException ignore) {
156                        // malformed %HH.
157                    }
158                } else {
159                    if (baos.size()>0) {
160                        buf.append(new String(baos.toByteArray(),"UTF-8"));
161                        baos.reset();
162                    }
163                    buf.append(c);
164                }
165            }
166            if (baos.size()>0)
167                buf.append(new String(baos.toByteArray(),"UTF-8"));
168            return buf.toString();
169        } catch (UnsupportedEncodingException e) {
170            throw new AssertionError(e); // UTF-8 is mandatory encoding
171        }
172    }
173
174    private static int fromHex(char upper) {
175        return ((upper & 0xF) + ((upper & 0x40) != 0 ? 9 : 0));
176    }
177}
