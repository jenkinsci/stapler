024package org.kohsuke.stapler.export;
025
026import java.io.IOException;
027import java.io.InputStream;
028import java.lang.reflect.Field;
029import java.lang.reflect.Method;
030import java.util.ArrayList;
031import java.util.Arrays;
032import java.util.Collections;
033import java.util.HashSet;
034import java.util.List;
035import java.util.Properties;
036import java.util.Set;
037import java.util.logging.Level;
038import java.util.logging.Logger;
039
040import javax.annotation.CheckForNull;
041import javax.annotation.Nullable;
042
043import com.google.common.base.Predicate;
044import org.kohsuke.stapler.export.TreePruner.ByDepth;
045
046/**
047 * Writes all the property of one {@link ExportedBean} to {@link DataWriter}.
048 *
049 * @author Kohsuke Kawaguchi
050 */
051public class Model<T> {
052    /**
053     * The class being modeled.
054     */
055    public final Class<T> type;
056
057    /**
058     * {@link Model} for the super class.
059     */
060    public final Model<? super T> superModel;
061
062    private final Property[] properties;
063
064    /*package*/ final ModelBuilder parent;
065    /*package*/ final int defaultVisibility;
066
067    /**
068     * Lazily loaded "*.javadoc" file for this model.
069     */
070    private volatile Properties javadoc;
071
072    private final Set<String> propertyNames = new HashSet<String>();
073
074    /*package*/ Model(ModelBuilder parent, Class<T> type, @CheckForNull Class<?> propertyOwner, @Nullable String property) throws NotExportableException {
075        this.parent = parent;
076        this.type = type;
077        ExportedBean eb = type.getAnnotation(ExportedBean.class);
078        if (eb == null) {
079            throw propertyOwner != null ? new NotExportableException(type, propertyOwner, property) : new NotExportableException(type);
080        }
081        this.defaultVisibility = eb.defaultVisibility();
082
083        Class<? super T> sc = type.getSuperclass();
084        if(sc!=null && sc.getAnnotation(ExportedBean.class)!=null)
085            superModel = parent.get(sc);
086        else
087            superModel = null;
088
089        List<Property> properties = new ArrayList<Property>();
090
091        // Use reflection to find out what properties are exposed.
092        for( Field f : type.getFields() ) {
093            if(f.getDeclaringClass()!=type) continue;
094            Exported exported = f.getAnnotation(Exported.class);
095            if(exported !=null)
096                properties.add(new FieldProperty(this,f, exported));
097        }
098
099        for( Method m : type.getMethods() ) {
100            if(m.getDeclaringClass()!=type) continue;
101            if(m.isSynthetic() && m.isBridge()) continue;
102            Exported exported = m.getAnnotation(Exported.class);
103            if(exported !=null) {
104                if (m.getParameterTypes().length > 0) {
105                    LOGGER.log(Level.WARNING, "Method " + m.getName() + " of " + type.getName() + " is annotated @Exported but requires arguments");
106                } else {
107                    properties.add(new MethodProperty(this, m, exported));
108                }
109            }
110        }
111
112        this.properties = properties.toArray(new Property[properties.size()]);
113        Arrays.sort(this.properties);
114        for (Property p : properties)
115            this.propertyNames.add(p.name);
116
117        parent.models.put(type,this);
118    }
119
120    /**
121     * Gets all the exported properties.
122     */
123    public List<Property> getProperties() {
124        return Collections.unmodifiableList(Arrays.asList(properties));
125    }
126
127    /**
128     * Does a property exist strictly in this class?
129     */
130    /*package*/ final Predicate<String> HAS_PROPERTY_NAME = new Predicate<String>() {
131        @Override
132        public boolean apply(@Nullable String name) {
133            return propertyNames.contains(name);
134        }
135    };
136    /**
137     * Does a property exist strictly in this class or its ancestors?
138     */
139    /*package*/ final Predicate<String> HAS_PROPERTY_NAME_IN_ANCESTRY = new Predicate<String>() {
140        @Override
141        public boolean apply(@Nullable String name) {
142            for (Model m=Model.this; m!=null; m=m.superModel)
143                if (m.propertyNames.contains(name))
144                    return true;
145            return false;
146        }
147    };
148
149    /**
150     * Loads the javadoc list and returns it as {@link Properties}.
151     *
152     * @return always non-null.
153     */
154    /*package*/ Properties getJavadoc() {
155        if(javadoc!=null)    return javadoc;
156        synchronized(this) {
157            if(javadoc!=null)    return javadoc;
158
159            // load
160            Properties p = new Properties();
161            InputStream is = type.getClassLoader().getResourceAsStream(type.getName().replace('$', '/').replace('.', '/') + ".javadoc");
162            if(is!=null) {
163                try {
164                    try {
165                        p.load(is);
166                    } finally {
167                        is.close();
168                    }
169                } catch (IOException e) {
170                    throw new RuntimeException("Unable to load javadoc for "+type,e);
171                }
172            }
173            javadoc = p;
174            return javadoc;
175        }
176    }
177
178    /**
179     * Writes the property values of the given object to the writer.
180     */
181    public void writeTo(T object, DataWriter writer) throws IOException {
182        writeTo(object, 0, writer);
183    }
184
185    /**
186     * Writes the property values of the given object to the writer.
187     *
188     * @param pruner
189     *      Controls which portion of the object graph will be sent to the writer.
190     */
191    public void writeTo(T object, TreePruner pruner, DataWriter writer) throws IOException {
192        try {
193            writer.type(null,object.getClass());
194        } catch (AbstractMethodError _) {
195            // legacy client that doesn't understand this
196        }
197        writer.startObject();
198        writeNestedObjectTo(object, pruner, writer);
199        writer.endObject();
200    }
201
202    /**
203     * Writes the property values of the given object to the writer.
204     *
205     * @param baseVisibility
206     *      This parameters controls how much data we'd be writing,
207     *      by adding bias to the sub tree cutting.
208     *      A property with {@link Exported#visibility() visibility} X will be written
209     *      if the current depth Y and baseVisibility Z satisfies {@code X + Z > Y}.
210     *
211     *      0 is the normal value. Positive value means writing bigger tree,
212     *      and negative value means writing smaller trees.
213     *
214     * @deprecated as of 1.139
215     */
216    @Deprecated
217    public void writeTo(T object, int baseVisibility, DataWriter writer) throws IOException {
218        writeTo(object,new ByDepth(1-baseVisibility),writer);
219    }
220
221    void writeNestedObjectTo(T object, TreePruner pruner, DataWriter writer) throws IOException {
222        if (superModel != null) {
223            superModel.writeNestedObjectTo(object, new FilteringTreePruner(HAS_PROPERTY_NAME,pruner), writer);
224        }
225
226        for (Property p : properties) {
227            p.writeTo(object, pruner, writer);
228        }
229    }
230
231    private static final Logger LOGGER = Logger.getLogger(Model.class.getName());
232}
