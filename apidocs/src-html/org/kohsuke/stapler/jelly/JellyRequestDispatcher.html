024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyTagException;
027import org.apache.commons.jelly.Script;
028import org.kohsuke.stapler.MetaClass;
029import org.kohsuke.stapler.StaplerRequest;
030import org.kohsuke.stapler.StaplerResponse;
031import org.kohsuke.stapler.WebApp;
032
033import javax.servlet.RequestDispatcher;
034import javax.servlet.ServletException;
035import javax.servlet.ServletRequest;
036import javax.servlet.ServletResponse;
037import java.io.IOException;
038
039/**
040 * @author Kohsuke Kawaguchi
041 */
042public final class JellyRequestDispatcher implements RequestDispatcher {
043    private final Object it;
044    private final Script script;
045    private final JellyFacet facet;
046
047    public JellyRequestDispatcher(Object it, Script script) {
048        this.it = it;
049        this.script = script;
050        facet = WebApp.getCurrent().getFacet(JellyFacet.class);
051    }
052
053    public void forward(ServletRequest servletRequest, ServletResponse servletResponse) throws ServletException, IOException {
054        try {
055            facet.scriptInvoker.invokeScript(
056                (StaplerRequest)servletRequest,
057                (StaplerResponse)servletResponse,
058                script, it);
059        } catch (JellyTagException e) {
060            throw new ServletException(e);
061        }
062    }
063
064    public void include(ServletRequest servletRequest, ServletResponse servletResponse) throws ServletException, IOException {
065        forward(servletRequest,servletResponse);
066    }
067}
