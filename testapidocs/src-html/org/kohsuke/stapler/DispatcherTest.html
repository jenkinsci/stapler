001package org.kohsuke.stapler;
002
003import com.gargoylesoftware.htmlunit.FailingHttpStatusCodeException;
004import com.gargoylesoftware.htmlunit.HttpMethod;
005import com.gargoylesoftware.htmlunit.TextPage;
006import com.gargoylesoftware.htmlunit.WebClient;
007import com.gargoylesoftware.htmlunit.WebRequestSettings;
008import org.kohsuke.stapler.test.JettyTestCase;
009import org.kohsuke.stapler.verb.GET;
010import org.kohsuke.stapler.verb.POST;
011
012import java.net.URL;
013
014/**
015 * @author Kohsuke Kawaguchi
016 */
017public class DispatcherTest extends JettyTestCase {
018
019    public final IndexDispatchByName indexDispatchByName = new IndexDispatchByName();
020
021    public class IndexDispatchByName {
022        @WebMethod(name="")
023        public HttpResponse doHelloWorld() {
024            return HttpResponses.plainText("Hello world");
025        }
026    }
027
028    /**
029     * Makes sure @WebMethod(name="") has the intended effect of occupying the root of the object in the URL space.
030     */
031    public void testIndexDispatchByName() throws Exception {
032        WebClient wc = new WebClient();
033        TextPage p = wc.getPage(new URL(url, "indexDispatchByName"));
034        assertEquals("Hello world\n", p.getContent());
035    }
036
037
038    //===================================================================
039
040
041    public final VerbMatch verbMatch = new VerbMatch();
042
043    public class VerbMatch {
044        @WebMethod(name="") @GET
045        public HttpResponse doGet() {
046            return HttpResponses.plainText("Got GET");
047        }
048
049        @WebMethod(name="") @POST
050        public HttpResponse doPost() {
051            return HttpResponses.plainText("Got POST");
052        }
053    }
054
055    /**
056     * Tests the dispatching of WebMethod based on verb
057     */
058    public void testVerbMatch() throws Exception {
059        WebClient wc = new WebClient();
060
061        check(wc, HttpMethod.GET);
062        check(wc, HttpMethod.POST);
063        try {
064            check(wc, HttpMethod.DELETE);
065            fail("There's no route for DELETE");
066        } catch (FailingHttpStatusCodeException e) {
067            assertEquals(404, e.getStatusCode());
068        }
069    }
070
071    private void check(WebClient wc, HttpMethod m) throws java.io.IOException {
072        TextPage p = wc.getPage(new WebRequestSettings(new URL(url, "verbMatch/"), m));
073        assertEquals("Got "+m.name()+"\n", p.getContent());
074    }
075
076
077    //===================================================================
078
079
080    public final ArbitraryWebMethodName arbitraryWebMethodName = new ArbitraryWebMethodName();
081
082    public class ArbitraryWebMethodName {
083        @WebMethod(name="")
084        public HttpResponse notDoPrefix() {
085            return HttpResponses.plainText("I'm index");
086        }
087
088        // this method is implicitly web method by its name
089        public HttpResponse doTheNeedful() {
090            return HttpResponses.plainText("DTN");
091        }
092    }
093
094
095    public void testArbitraryWebMethodName() throws Exception {
096        WebClient wc = new WebClient();
097        TextPage p = wc.getPage(new URL(url, "arbitraryWebMethodName"));
098        assertEquals("I'm index\n", p.getContent());
099
100        p = wc.getPage(new URL(url, "arbitraryWebMethodName/theNeedful"));
101        assertEquals("DTN\n", p.getContent());
102
103    }
104}
