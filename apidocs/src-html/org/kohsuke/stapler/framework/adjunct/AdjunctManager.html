024package org.kohsuke.stapler.framework.adjunct;
025
026import org.kohsuke.stapler.*;
027
028import javax.servlet.ServletException;
029import javax.servlet.ServletContext;
030import static javax.servlet.http.HttpServletResponse.SC_FORBIDDEN;
031import static javax.servlet.http.HttpServletResponse.SC_NOT_FOUND;
032import java.io.IOException;
033import java.net.URL;
034import java.util.concurrent.ConcurrentHashMap;
035
036/**
037 * This application-scoped object that exposes djuncts to URL.
038 *
039 * <p>
040 * Adjuncts are packaging of JavaScript, CSS, and other static assets in jar files with dependency
041 * information between them. This allows JavaScript libraries and other static assets to be reused
042 * across different projects through Maven/Ivy.
043 *
044 * <p>
045 * To use {@link AdjunctManager} in your application, create one instance, and bind it to URL
046 * (like you do any other objects.) The most typical way of doing this is to define it as a
047 * field in your top-level object.
048 *
049 * <pre>
050 * public class MyApplication {
051 *     public final AdjunctManager adjuncts = new AdjunctManager(context,getClass().getClassLoader(),"/adjuncts");
052 * }
053 * </pre>
054 *
055 * <p>
056 * How you include an adjunct will depend on your template language, but for example in Jelly you do:
057 * <pre>
058 * &lt;st:adjunct includes="org.kohsuke.stapler.bootstrap"/>
059 * </pre>
060 *
061 * Or from Groovy you do:
062 * <pre>
063 * adjunct "org.kohsuke.stapler.bootstrap"
064 * </pre>
065 *
066 * <p>
067 * ... and this produces a series of <tt>style</tt> and <tt>script</tt> tags that include all the
068 * necessary JavaScript, CSS, and their dependencies.
069 *
070 * <p>
071 * Internally, this class provides caching for {@link Adjunct}s.
072 *
073 * @author Kohsuke Kawaguchi
074 * @see Adjunct
075 */
076public class AdjunctManager {
077    private final ConcurrentHashMap<String, Adjunct> adjuncts = new ConcurrentHashMap<String,Adjunct>();
078
079    /**
080     * Map used as a set to remember which resources can be served.
081     */
082    private final ConcurrentHashMap<String,String> allowedResources = new ConcurrentHashMap<String,String>();
083
084    private final ClassLoader classLoader;
085
086    /**
087     * Absolute URL of the {@link AdjunctManager} in the calling application where it is bound to.
088     *
089     * <p>
090     * The path is treated relative from the context path of the application, and it
091     * needs to end without '/'. So it needs to be something like "foo/adjuncts" or
092     * just "adjuncts". Can be e.g. {@code adjuncts/uNiQuEhAsH} to improve caching behavior.
093     */
094    public final String rootURL;
095
096    /**
097     * Hint instructing adjuncts to load a debuggable non-minified version of the script,
098     * as opposed to the production version.
099     *
100     * This is only a hint, and so the semantics of it isn't very well defined. The intention
101     * is to assist JavaScript debugging.
102     */
103    public boolean debug = Boolean.getBoolean(AdjunctManager.class.getName()+".debug");
104
105    public final WebApp webApp;
106    private final long expiration;
107
108    @Deprecated
109    public AdjunctManager(ServletContext context,ClassLoader classLoader, String rootURL) {
110        this(context, classLoader, rootURL, /* one day */24L * 60 * 60 * 1000);
111    }
112
113    /**
114     * @param classLoader
115     *      ClassLoader to load adjuncts from.
116     * @param rootURL
117     *      See {@link #rootURL} for the meaning of this parameter.
118     * @param expiration milliseconds from service time until expiration, for {@link #doDynamic}
119     *                    (as in {@link StaplerResponse#serveFile(StaplerRequest, URL, long)});
120     *                    if {@link #rootURL} is unique per session then this can be very long;
121     *                    otherwise a day might be reasonable
122     */
123    public AdjunctManager(ServletContext context, ClassLoader classLoader, String rootURL, long expiration) {
124        this.classLoader = classLoader;
125        this.rootURL = rootURL;
126        this.webApp = WebApp.get(context);
127        this.expiration = expiration;
128        // register this globally
129        context.setAttribute(KEY,this);
130    }
131
132    public static AdjunctManager get(ServletContext context) {
133        return (AdjunctManager) context.getAttribute(KEY);
134    }
135
136    /**
137     * Obtains the adjunct.
138     *
139     * @return
140     *      always non-null.
141     * @throws IOException
142     *      if failed to locate {@link Adjunct}.
143     */
144    public Adjunct get(String name) throws IOException {
145        Adjunct a = adjuncts.get(name);
146        if(a!=null) return a;   // found it
147
148        synchronized (this) {
149            a = adjuncts.get(name);
150            if(a!=null) return a;   // one more check before we start loading
151            a = new Adjunct(this,name,classLoader);
152            adjuncts.put(name,a);
153            return a;
154        }
155    }
156
157    /**
158     * Serves resources in the class loader.
159     */
160    public void doDynamic(StaplerRequest req, StaplerResponse rsp) throws IOException, ServletException {
161        String path = req.getRestOfPath();
162        if (path.charAt(0)=='/') path = path.substring(1);
163
164        if(!allowedResources.containsKey(path)) {
165            if(!allowResourceToBeServed(path)) {
166                rsp.sendError(SC_FORBIDDEN);
167                return;
168            }
169            // remember URLs that we can serve. but don't remember error ones, as it might be unbounded
170            allowedResources.put(path,path);
171        }
172
173        URL res = classLoader.getResource(path);
174        if(res==null) {
175            throw HttpResponses.error(SC_NOT_FOUND,new IllegalArgumentException("No such adjunct found: "+path));
176        } else {
177            long expires = MetaClass.NO_CACHE ? 0 : expiration;
178            rsp.serveFile(req,res,expires);
179        }
180    }
181
182    /**
183     * Controls whether the given resource can be served to browsers.
184     *
185     * <p>
186     * This method can be overridden by the sub classes to change the access control behavior.
187     *
188     * <p>
189     * {@link AdjunctManager} is capable of serving all the resources visible
190     * in the classloader by default. If the resource files need to be kept private,
191     * return false, which causes the request to fail with 401. 
192     *
193     * Otherwise return true, in which case the resource will be served.
194     */
195    protected boolean allowResourceToBeServed(String absolutePath) {
196        // does it have an adjunct directory marker?
197        int idx = absolutePath.lastIndexOf('/');
198        if (idx>0 && classLoader.getResource(absolutePath.substring(0,idx)+"/.adjunct")!=null)
199            return true;
200
201        // backward compatible behaviour
202        return absolutePath.endsWith(".gif")
203            || absolutePath.endsWith(".png")
204            || absolutePath.endsWith(".css")
205            || absolutePath.endsWith(".js");
206    }
207
208    /**
209     * Key in {@link ServletContext} to look up {@link AdjunctManager}.
210     */
211    private static final String KEY = AdjunctManager.class.getName();
212}
