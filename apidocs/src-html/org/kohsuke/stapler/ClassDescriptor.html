059/**
060 * Reflection information of a {@link Class}.
061 *
062 * @author Kohsuke Kawaguchi
063 */
064public final class ClassDescriptor {
065    public final Class clazz;
066    public final FunctionList methods;
067    public final Field[] fields;
068
069    /**
070     * @param clazz
071     *      The class to build a descriptor around.
072     * @param wrappers
073     *      Optional wrapper duck-typing classes.
074     *      Static methods on this class that has the first parameter
075     *      as 'clazz' will be handled as if it's instance methods on
076     *      'clazz'. Useful for adding view/controller methods on
077     *      model classes.
078     */
079    public ClassDescriptor(Class clazz, Class... wrappers) {
080        this.clazz = clazz;
081        this.fields = clazz.getFields();
082
083        // instance methods
084        List<MethodMirror> methods = new ArrayList<MethodMirror>();
085        findMethods(clazz,clazz,methods,new HashSet<Class>());
086
087        // organize them into groups
088        Map<Signature,List<Method>> groups = new LinkedHashMap<>();
089        for (MethodMirror m : methods) {
090            List<Method> v = groups.get(m.sig);
091            if (v==null)    groups.put(m.sig, v=new ArrayList<Method>());
092            v.add(m.method);
093        }
094
095        // build functions from groups
096        List<Function> functions = new ArrayList<Function>();
097        for (List<Method> m : groups.values()) {
098            if (m.size()==1) {
099                Method one = m.get(0);
100                functions.add(new Function.InstanceFunction(one)
101                        .wrapByInterceptors(one));
102            } else {
103                Collections.reverse(m);
104                functions.add(new Function.OverridingInstanceFunction(m)
105                        .wrapByInterceptors(new UnionAnnotatedElement(m)));
106            }
107        }
108
109        if(wrappers!=null) {
110            for (Class w : wrappers) {
111                for (Method m : w.getMethods()) {
112                    if(!Modifier.isStatic(m.getModifiers()))
113                        continue;
114                    Class<?>[] p = m.getParameterTypes();
115                    if(p.length==0)
116                        continue;
117                    if(p[0].isAssignableFrom(clazz))
118                        continue;
119                    functions.add(new Function.StaticFunction(m).wrapByInterceptors(m));
120                }
121            }
122        }
123        this.methods = new FunctionList(functions);
124    }

126    /**
127     * Finds all the public methods of 'c' across class/interface hierarchy and accumulates into a list,
128     * from ancestor first.
129     *
130     * @param logical
131     *      Actual type parameterization of 'c' in this type hierarchy. For example, methods
132     *      we discover on {@code Collection<String>} is different from methods we discover on
133     *      {@code Collection<Integer>}. This type arguments decorates 'c' to carry this semantics.
134     *      Erasure of this is always {@code c}.
135     *
136     */
137    private List<MethodMirror> findMethods(Class c, java.lang.reflect.Type logical, List<MethodMirror> result, Set<Class> visited) {
138        if (!visited.add(c))
139            return result; // avoid visiting the same type twice
140
141        // visit interfaces first so that class methods are considered as overriding interface methods
142        for (Class i : c.getInterfaces()) {
143            findMethods(i, Types.getBaseClass(logical,i), result, visited);
144        }
145        Class sc = c.getSuperclass();
146        if (sc!=null)
147            findMethods(sc,Types.getBaseClass(logical,sc),result,visited);
148
149        Method[] declaredMethods = c.getDeclaredMethods();
150        Arrays.sort(declaredMethods, new Comparator<Method>() {
151            @Override
152            public int compare(Method m1, Method m2) {
153                boolean m1d = m1.getAnnotation(Deprecated.class) != null;
154                boolean m2d = m2.getAnnotation(Deprecated.class) != null;
155                if (m1d && !m2d) {
156                    // Prefer nondeprecated to deprecated.
157                    return 1;
158                } else if (!m1d && m2d) {
159                    return -1;
160                } else {
161                    // Sort by string representation, so for example doFoo() is preferred to doFoo(StaplerRequest, StaplerResponse).
162                    return m1.toString().compareTo(m2.toString());
163                }
164            }
165        });
166        for (Method m : declaredMethods) {
167            if (m.isBridge())    continue;
168            if ((m.getModifiers() & Modifier.PUBLIC)!=0) {
169                java.lang.reflect.Type[] paramTypes = m.getGenericParameterTypes();
170                Class[] erasedParamTypes = new Class[paramTypes.length];
171                for (int i = 0; i < paramTypes.length; i++) {
172                    if (logical instanceof ParameterizedType)
173                        erasedParamTypes[i] = Types.erasure(Types.bind(paramTypes[i], c, (ParameterizedType)logical));
174                    else
175                        erasedParamTypes[i] = Types.erasure(paramTypes[i]);
176                }
177
178                result.add(new MethodMirror(new Signature(m.getName(), erasedParamTypes), m));
179            }
180        }
181        return result;
182    }

184    /**
185     * Loads the list of parameter names of the given method, by using a stapler-specific way of getting it.
186     *
187     * <p>
188     * This is not the best place to expose this, but for now this would do.
189     */
190    public static String[] loadParameterNames(Method m) {
191        CapturedParameterNames cpn = m.getAnnotation(CapturedParameterNames.class);
192        if(cpn!=null)   return cpn.value();
193
194        // debug information, if present, is more trustworthy
195        try {
196            String[] n = ASM.loadParametersFromAsm(m);
197            if (n!=null)    return n;
198        } catch (LinkageError e) {
199            LOGGER.log(FINE, "Incompatible ASM", e);
200        } catch (IOException e) {
201            LOGGER.log(WARNING, "Failed to load a class file", e);
202        }
203
204        // otherwise check the .stapler file
205        Class<?> c = m.getDeclaringClass();
206        URL url = c.getClassLoader().getResource(
207                    c.getName().replace('.', '/').replace('$','/') + '/' + m.getName() + ".stapler");
208        if(url!=null) {
209            try {
210                return IOUtils.toString(url.openStream()).split(",");
211            } catch (IOException e) {
212                LOGGER.log(WARNING, "Failed to load "+url,e);
213                return EMPTY_ARRAY;
214            }
215        }
216
217        // couldn't find it
218        return EMPTY_ARRAY;
219    }

245    /**
246     * Determines the constructor parameter names.
247     *
248     * <p>
249     * First, try to load names from the debug information. Otherwise
250     * if there's the .stapler file, load it as a property file and determines the constructor parameter names.
251     * Otherwise, look for {@link CapturedParameterNames} annotation.
252     */
253    public String[] loadConstructorParamNames() {
254        Constructor<?>[] ctrs = clazz.getConstructors();
255        // which constructor was data bound?
256        Constructor<?> dbc = null;
257        for (Constructor<?> c : ctrs) {
258            if (c.getAnnotation(DataBoundConstructor.class) != null) {
259                dbc = c;
260                break;
261            }
262        }
263
264        if (dbc==null)
265            throw new NoStaplerConstructorException("There's no @DataBoundConstructor on any constructor of " + clazz);
266
267        String[] names = ClassDescriptor.loadParameterNames(dbc);
268        if (names.length==dbc.getParameterTypes().length)
269            return names;
270
271        String resourceName = clazz.getName().replace('.', '/').replace('$','/') + ".stapler";
272        ClassLoader cl = clazz.getClassLoader();
273        if(cl==null)
274            throw new NoStaplerConstructorException(clazz+" is a built-in type");
275        InputStream s = cl.getResourceAsStream(resourceName);
276        if (s != null) {// load the property file and figure out parameter names
277            try {
278                Properties p = new Properties();
279                p.load(s);
280                s.close();
281
282                String v = p.getProperty("constructor");
283                if (v.length() == 0) return new String[0];
284                return v.split(",");
285            } catch (IOException e) {
286                throw new IllegalArgumentException("Unable to load " + resourceName, e);
287            }
288        }
289
290        // no debug info and no stapler file
291        throw new NoStaplerConstructorException(
292                "Unable to find " + resourceName + ". " +
293                        "Run 'mvn clean compile' once to run the annotation processor.");
294    }

297    /**
298     * Isolate the ASM dependency to its own class, as otherwise this seems to cause linkage error on the whole {@link ClassDescriptor}.
299     */
300    /*package*/ static class ASM {
301        /**
302         * Try to load parameter names from the debug info by using ASM.
303         */
304        private static String[] loadParametersFromAsm(final Method m) throws IOException {
305            final String[] paramNames = new String[m.getParameterTypes().length];
306            if (paramNames.length==0) return paramNames;
307            Class<?> c = m.getDeclaringClass();
308            URL clazz = c.getClassLoader().getResource(c.getName().replace('.', '/') + ".class");
309            if (clazz==null)    return null;
310
311            final TreeMap<Integer,String> localVars = new TreeMap<Integer,String>();
312            ClassReader r = new ClassReader(clazz.openStream());
313            r.accept(new ClassVisitor(ASM5) {
314                final String md = Type.getMethodDescriptor(m);
315                // First localVariable is "this" for non-static method
316                final int limit = (m.getModifiers() & Modifier.STATIC) != 0 ? 0 : 1;
317                @Override public MethodVisitor visitMethod(int access, String methodName, String desc, String signature, String[] exceptions) {
318                    if (methodName.equals(m.getName()) && desc.equals(md))
319                        return new MethodVisitor(ASM5) {
320                            @Override public void visitLocalVariable(String name, String desc, String signature, Label start, Label end, int index) {
321                                if (index >= limit)
322                                    localVars.put(index, name);
323                            }
324                        };
325                    else
326                        return null; // ignore this method
327                }
328            }, 0);
329
330            // Indexes may not be sequential, but first set of local variables are method params
331            int i = 0;
332            for (String s : localVars.values()) {
333                paramNames[i] = s;
334                if (++i == paramNames.length) return paramNames;
335            }
336            return null; // Not enough data found to fill array
337        }
338
339        /**
340         * Try to load parameter names from the debug info by using ASM.
341         */
342        private static String[] loadParametersFromAsm(final Constructor m) throws IOException {
343            final String[] paramNames = new String[m.getParameterTypes().length];
344            if (paramNames.length==0) return paramNames;
345            Class<?> c = m.getDeclaringClass();
346            URL clazz = c.getClassLoader().getResource(c.getName().replace('.', '/') + ".class");
347            if (clazz==null)    return null;
348
349            final TreeMap<Integer,String> localVars = new TreeMap<Integer,String>();
350            InputStream is = clazz.openStream();
351            try {
352                ClassReader r = new ClassReader(is);
353                r.accept(new ClassVisitor(ASM5) {
354                    final String md = getConstructorDescriptor(m);
355                    public MethodVisitor visitMethod(int access, String methodName, String desc, String signature, String[] exceptions) {
356                        if (methodName.equals("<init>") && desc.equals(md))
357                            return new MethodVisitor(ASM5) {
358                                @Override public void visitLocalVariable(String name, String desc, String signature, Label start, Label end, int index) {
359                                    if (index>0)   // 0 is 'this'
360                                        localVars.put(index, name);
361                                }
362                            };
363                        else
364                            return null; // ignore this method
365                    }
366                }, 0);
367            } finally {
368                is.close();
369            }
370
371            // Indexes may not be sequential, but first set of local variables are method params
372            int i = 0;
373            for (String s : localVars.values()) {
374                paramNames[i] = s;
375                if (++i == paramNames.length) return paramNames;
376            }
377            return null; // Not enough data found to fill array
378        }
379
380        private static String getConstructorDescriptor(Constructor c) {
381            StringBuilder buf = new StringBuilder("(");
382            for (Class p : c.getParameterTypes())
383                buf.append(Type.getDescriptor(p));
384            return buf.append(")V").toString();
385        }
386    }

388    final class MethodMirror {
389        final Signature sig;
390        final Method method;
391
392        public MethodMirror(Signature sig, Method method) {
393            this.sig = sig;
394            this.method = method;
395        }
396    }

398    /**
399     * A method signature used to determine what methods override each other
400     */
401    final class Signature {
402        final String methodName;
403        final Class[] parameters;
404
405        Signature(String methodName, Class[] parameters) {
406            this.methodName = methodName;
407            this.parameters = parameters;
408        }
409
410        @Override
411        public boolean equals(Object o) {
412            if (this == o) return true;
413            if (o == null || getClass() != o.getClass()) return false;
414
415            Signature that = (Signature) o;
416
417            return this.methodName.equals(that.methodName)
418                && Arrays.equals(this.parameters, that.parameters);
419
420        }
421
422        @Override
423        public int hashCode() {
424            return 31 * methodName.hashCode() + Arrays.hashCode(parameters);
425        }
426    }
