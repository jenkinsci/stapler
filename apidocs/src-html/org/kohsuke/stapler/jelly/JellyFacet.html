049/**
050 * {@link Facet} that adds Jelly as the view.
051 * 
052 * @author Kohsuke Kawaguchi
053 */
054@MetaInfServices(Facet.class)
055public class JellyFacet extends Facet implements JellyCompatibleFacet {
056    /**
057     * Used to invoke Jelly script. Can be replaced to the custom object.
058     */
059    public volatile ScriptInvoker scriptInvoker = new DefaultScriptInvoker();
060
061    /**
062     * Used to load {@link ResourceBundle}s.
063     */
064    public volatile ResourceBundleFactory resourceBundleFactory = ResourceBundleFactory.INSTANCE;
065
066    public void buildViewDispatchers(final MetaClass owner, List<Dispatcher> dispatchers) {
067        dispatchers.add(new Dispatcher() {
068            final JellyClassTearOff tearOff = owner.loadTearOff(JellyClassTearOff.class);
069
070            public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
071                // check Jelly view
072                String next = req.tokens.peek();
073                if(next==null)  return false;
074
075                // only match the end of the URL
076                if (req.tokens.countRemainingTokens()>1)    return false;
077                // and avoid serving both "foo" and "foo/" as relative URL semantics are drastically different
078                if (req.tokens.endsWithSlash)      return false;
079
080                try {
081                    Script script = tearOff.findScript(next+".jelly");
082
083                    if(script==null)        return false;   // no Jelly script found
084
085                    req.tokens.next();
086
087                    if (traceable()) {
088                        // Null not expected here
089                        String src = next+".jelly";
090                        if (script instanceof JellyViewScript) {
091                            JellyViewScript jvs = (JellyViewScript) script;
092                            src = jvs.getName();
093                        }
094                        trace(req,rsp,"-> %s on <%s>", src, node);
095                    }
096
097                    scriptInvoker.invokeScript(req, rsp, script, node);
098
099                    return true;
100                } catch (RuntimeException e) {
101                    throw e;
102                } catch (IOException e) {
103                    throw e;
104                } catch (Exception e) {
105                    throw new ServletException(e);
106                }
107            }
108
109            public String toString() {
110                return "VIEW.jelly for url=/VIEW";
111            }
112        });
113    }
114
115    @Override
116    public void buildIndexDispatchers(MetaClass owner, List<Dispatcher> dispatchers) {
117        try {
118            if (owner.loadTearOff(JellyClassTearOff.class).findScript("index.jelly")!=null) {
119                super.buildIndexDispatchers(owner, dispatchers);
120            }
121        } catch (JellyException e) {
122            LOGGER.log(Level.WARNING, "Failed to parse index.jelly for "+owner, e);
123        }
124    }
125
126    public Collection<Class<JellyClassTearOff>> getClassTearOffTypes() {
127        return TEAROFF_TYPES;
128    }
129
130    public Collection<String> getScriptExtensions() {
131        return EXTENSION;
132    }
133
134    public RequestDispatcher createRequestDispatcher(RequestImpl request, Klass<?> type, Object it, String viewName) throws IOException {
135        TearOffSupport mc = request.stapler.getWebApp().getMetaClass(type);
136        return mc.loadTearOff(JellyClassTearOff.class).createDispatcher(it,viewName);
137    }
138
139    public boolean handleIndexRequest(RequestImpl req, ResponseImpl rsp, Object node, MetaClass nodeMetaClass) throws IOException, ServletException {
140        return nodeMetaClass.loadTearOff(JellyClassTearOff.class).serveIndexJelly(req,rsp,node);
141    }
142
143    /**
144     * Sets the Jelly {@link ExpressionFactory} to be used to parse views.
145     *
146     * <p>
147     * This method should be invoked from your implementation of
148     * {@link ServletContextListener#contextInitialized(ServletContextEvent)}.
149     *
150     * <p>
151     * Once views are parsed, they won't be re-parsed just because you called
152     * this method to override the expression factory.
153     *
154     * <p>
155     * The primary use case of this feature is to customize the behavior
156     * of JEXL evaluation.
157     */
158    public static void setExpressionFactory( ServletContextEvent event, ExpressionFactory factory ) {
159        JellyClassLoaderTearOff.EXPRESSION_FACTORY = factory;
160    }
161
162    /**
163     * This flag will activate the Jelly evaluation trace.
164     * It generates extra comments into HTML, indicating where the fragment was rendered.
165     */
166    public static boolean TRACE = Boolean.getBoolean("stapler.jelly.trace");
167
168    private static final Set<Class<JellyClassTearOff>> TEAROFF_TYPES = Collections.singleton(JellyClassTearOff.class);
169
170    private static final Set<String> EXTENSION = Collections.singleton(".jelly");
171}
