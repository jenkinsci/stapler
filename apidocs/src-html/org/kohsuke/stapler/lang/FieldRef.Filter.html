001package org.kohsuke.stapler.lang;
002
003import org.apache.commons.lang.StringUtils;
004import org.kohsuke.stapler.util.IllegalReflectiveAccessLogHandler;
005
006import java.lang.annotation.Annotation;
007import java.lang.reflect.Field;
008import java.lang.reflect.Modifier;
009import java.util.Arrays;
010import java.util.logging.Logger;
011
012/**
013 * Fields of {@link Klass}.
014 *
015 * @see Klass#getDeclaredFields()
016 * @author Kohsuke Kawaguchi
017 */
018public abstract class FieldRef extends AnnotatedRef {
019
020    private static final Logger LOGGER = Logger.getLogger(FieldRef.class.getName());
021
022    public interface Filter {
023        boolean keep(FieldRef m);
024
025        Filter ALWAYS_OK = new Filter() {
026            @Override
027            public boolean keep(FieldRef m) {
028                return true;
029            }
030        };
031    }
032
033    /**
034     * Name of the method.
035     *
036     * @see Field#getName()
037     */
038    public abstract String getName();
039
040    /**
041     * Whether the field is static.
042     * @return
043     */
044    public abstract boolean isStatic();
045
046    /**
047     * Obtains the value of the field of the instance.
048     */
049    public abstract Object get(Object instance) throws IllegalAccessException;
050
051    /**
052     * Gets a fully qualified name of this field that includes the declaring type.
053     */
054    public abstract String getQualifiedName();
055
056    /**
057     * Gets the signature for this for use in lists
058     *
059     * @see org.kohsuke.stapler.Function#getSignature()
060     *
061     */
062    public abstract String getSignature();
063
064    // TODO Should this be Klass?
065    public abstract Class<?> getReturnType();
066
067    /**
068     * Returns true if this method is a 'public' method that should be used for routing requests.
069     */
070    public boolean isRoutable() {
071        return true;
072    }
073
074    public static FieldRef wrap(final Field f) {
075        return new FieldRef() {
076            @Override
077            public <T extends Annotation> T getAnnotation(Class<T> type) {
078                return f.getAnnotation(type);
079            }
080
081            @Override
082            public String getName() {
083                return f.getName();
084            }
085
086            @Override
087            public Object get(Object instance) throws IllegalAccessException {
088                try {
089                    return f.get(instance);
090                } catch (IllegalAccessException e) {
091                    LOGGER.warning(IllegalReflectiveAccessLogHandler.get(e));
092                    f.setAccessible(true);
093                    return f.get(instance);
094                }
095            }
096
097            @Override
098            public boolean isStatic() {
099                return Modifier.isStatic(f.getModifiers());
100            }
101
102            @Override
103            public Class<?> getReturnType() {
104                return f.getType();
105            }
106
107            @Override
108            public String getSignature() {
109                String prefix = isStatic() ? "staticField" : "field";
110                return StringUtils.join(Arrays.asList(prefix, f.getDeclaringClass().getName(), getName()), ' ');
111            }
112
113            @Override
114            public String getQualifiedName() {
115                return f.getDeclaringClass().getName()+"."+getName();
116            }
117
118            @Override
119            public boolean isRoutable() {
120                return Modifier.isPublic(f.getModifiers());
121            }
122        };
123    }
124}
