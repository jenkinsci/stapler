215    @Override
216    public String[] getParameterValues(String name) {
217        if(!isMultipart()) {
218            return super.getParameterValues(name);
219        }
220        Map<String, String> data = getFormDataFormFields();
221        if (data.isEmpty()) {
222            return super.getParameterValues(name);
223        }
224        
225        String formFieldVal = data.get(name);
226        if (formFieldVal == null) {
227            return super.getParameterValues(name);
228        }
229
230        String[] values = super.getParameterValues(name);
231        if (values == null) {
232            values = new String[0];
233        }
234        String[] extValues = new String[values.length + 1];
235        System.arraycopy(values, 0, extValues, 0, values.length);
236        extValues[extValues.length - 1] = formFieldVal;
237
238        return extValues;
239    }

335    public boolean checkIfModified(long lastModified, StaplerResponse rsp, long expiration) {
336        if(lastModified<=0)
337            return false;
338
339        // send out Last-Modified, or check If-Modified-Since
340        String since = getHeader("If-Modified-Since");
341        SimpleDateFormat format = Stapler.HTTP_DATE_FORMAT.get();
342        if(since!=null) {
343            try {
344                long ims = format.parse(since).getTime();
345                if(lastModified<ims+1000) {
346                    // +1000 because date header is second-precision and Java has milli-second precision
347                    rsp.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
348                    return true;
349                }
350            } catch (NumberFormatException e) {
351                // just ignore and serve the content
352            } catch (ParseException e) {
353                // just ignore and serve the content
354            }
355        }
356        String tm = format.format(new Date(lastModified));
357        rsp.setHeader("Last-Modified", tm);
358        if(expiration==0) {
359            // don't let browsers
360            rsp.setHeader("Expires", tm);
361        } else {
362            // expire in "NOW+expiration" 
363            rsp.setHeader("Expires",format.format(new Date(new Date().getTime()+expiration)));
364        }
365        return false;
366    }

407    public <T>
408    List<T> bindParametersToList(Class<T> type, String prefix) {
409        List<T> r = new ArrayList<T>();
410
411        int len = Integer.MAX_VALUE;
412
413        Enumeration e = getParameterNames();
414        while(e.hasMoreElements()) {
415            String name = (String)e.nextElement();
416            if(name.startsWith(prefix))
417                len = Math.min(len,getParameterValues(name).length);
418        }
419
420        if(len==Integer.MAX_VALUE)
421            return r;   // nothing
422
423        try {
424            new ClassDescriptor(type).loadConstructorParamNames();
425            // use the designated constructor for databinding
426            for( int i=0; i<len; i++ )
427                r.add(bindParameters(type,prefix,i));
428        } catch (NoStaplerConstructorException _) {
429            // no designated data binding constructor. use reflection
430            try {
431                for( int i=0; i<len; i++ ) {
432                    T t = type.newInstance();
433                    r.add(t);
434
435                    e = getParameterNames();
436                    while(e.hasMoreElements()) {
437                        String name = (String)e.nextElement();
438                        if(name.startsWith(prefix))
439                            fill(t, name.substring(prefix.length()), getParameterValues(name)[i] );
440                    }
441                }
442            } catch (InstantiationException x) {
443                throw new InstantiationError(x.getMessage());
444            } catch (IllegalAccessException x) {
445                throw new IllegalAccessError(x.getMessage());
446            }
447        }
448
449        return r;
450    }

456    public <T> T bindParameters(Class<T> type, String prefix, int index) {
457        String[] names = new ClassDescriptor(type).loadConstructorParamNames();
458
459        // the actual arguments to invoke the constructor with.
460        Object[] args = new Object[names.length];
461
462        // constructor
463        Constructor<T> c = findConstructor(type, names.length);
464        Class[] types = c.getParameterTypes();
465
466        // convert parameters
467        for( int i=0; i<names.length; i++ ) {
468            String[] values = getParameterValues(prefix + names[i]);
469            String param;
470            if(values!=null)
471                param = values[index];
472            else
473                param = null;
474
475            Converter converter = Stapler.lookupConverter(types[i]);
476            if (converter==null)
477                throw new IllegalArgumentException("Unable to convert to "+types[i]);
478
479            args[i] = converter.convert(types[i],param);
480        }
481
482        return invokeConstructor(c, args);
483    }

493    public void bindJSON(Object bean, JSONObject src) {
494        try {
495            for( String key : (Set<String>)src.keySet() ) {
496                TypePair type = getPropertyType(bean, key);
497                if(type==null)
498                    continue;
499
500                try {
501                    fill(bean,key, type.convertJSON(src.get(key)));
502                } catch (WrongTypeException e) {
503                    throw new IllegalArgumentException(String.format("Error binding field %s: %s", key, e.getMessage()));
504                }
505            }
506        } catch (IllegalAccessException e) {
507            IllegalAccessError x = new IllegalAccessError(e.getMessage());
508            x.initCause(e);
509            throw x;
510        } catch (InvocationTargetException x) {
511            Throwable e = x.getTargetException();
512            if(e instanceof RuntimeException)
513                throw (RuntimeException)e;
514            if(e instanceof Error)
515                throw (Error)e;
516            throw new RuntimeException(x);
517        }
518    }

520    public <T> List<T> bindJSONToList(Class<T> type, Object src) {
521        ArrayList<T> r = new ArrayList<T>();
522        if (src instanceof JSONObject) {
523            JSONObject j = (JSONObject) src;
524            r.add(bindJSON(type,j));
525        }
526        if (src instanceof JSONArray) {
527            JSONArray a = (JSONArray) src;
528            for (Object o : a) {
529                if (o instanceof JSONObject) {
530                    JSONObject j = (JSONObject) o;
531                    r.add(bindJSON(type,j));
532                }
533            }
534        }
535        return r;
536    }

539    private <T> T invokeConstructor(Constructor<T> c, Object[] args) {
540        try {
541            return c.newInstance(args);
542        } catch (InstantiationException e) {
543            InstantiationError x = new InstantiationError(e.getMessage());
544            x.initCause(e);
545            throw x;
546        } catch (IllegalAccessException e) {
547            IllegalAccessError x = new IllegalAccessError(e.getMessage());
548            x.initCause(e);
549            throw x;
550        } catch (InvocationTargetException e) {
551            Throwable x = e.getTargetException();
552            if(x instanceof Error)
553                throw (Error)x;
554            if(x instanceof RuntimeException)
555                throw (RuntimeException)x;
556            throw new IllegalArgumentException(x);
557        } catch (IllegalArgumentException e) {
558            throw new IllegalArgumentException("Failed to invoke "+c+" with "+ Arrays.asList(args),e);
559        }
560    }

562    private <T> Constructor<T> findConstructor(Class<T> type, int length) {
563        Constructor<?>[] ctrs = type.getConstructors();
564        // one with DataBoundConstructor is the most reliable
565        for (Constructor c : ctrs) {
566            if(c.getAnnotation(DataBoundConstructor.class)!=null) {
567                if(c.getParameterTypes().length!=length)
568                    throw new IllegalArgumentException(c+" has @DataBoundConstructor but it doesn't match with your .stapler file. Try clean rebuild");
569                return c;
570            }
571        }
572        // if not, maybe this was from @stapler-constructor,
573        // so look for the constructor with the expected argument length.
574        // this is not very reliable.
575        for (Constructor c : ctrs) {
576            if(c.getParameterTypes().length==length)
577                return c;
578        }
579        throw new IllegalArgumentException(type+" does not have a constructor with "+length+" arguments");
580    }

582    private static void fill(Object bean, String key, Object value) {
583        StringTokenizer tokens = new StringTokenizer(key);
584        while(tokens.hasMoreTokens()) {
585            String token = tokens.nextToken();
586            boolean last = !tokens.hasMoreTokens();  // is this the last token?
587
588            try {
589                if(last) {
590                    copyProperty(bean,token,value);
591                } else {
592                    bean = BeanUtils.getProperty(bean,token);
593                }
594            } catch (IllegalAccessException x) {
595                throw new IllegalAccessError(x.getMessage());
596            } catch (InvocationTargetException x) {
597                Throwable e = x.getTargetException();
598                if(e instanceof RuntimeException)
599                    throw (RuntimeException)e;
600                if(e instanceof Error)
601                    throw (Error)e;
602                throw new RuntimeException(x);
603            } catch (NoSuchMethodException e) {
604                // ignore if there's no such property
605            }
606        }
607    }

612    private final class TypePair {
613        final Type genericType;
614        /**
615         * Erasure of {@link #genericType}
616         */

619        TypePair(Type genericType, Class type) {
620            this.genericType = genericType;

625            this(f.getGenericType(),f.getType());

628        /**
629         * Converts the given JSON object (either {@link JSONObject}, {@link JSONArray}, or other primitive types
630         * in JSON, to the type represented by the 'this' object.
631         */
632        public Object convertJSON(Object o) {
633            Object r = bindInterceptor.onConvert(genericType, type, o);
634            if (r!= BindInterceptor.DEFAULT)    return r; // taken over by the interceptor
635
636            for (BindInterceptor i : getWebApp().bindInterceptors) {
637                r = i.onConvert(genericType, type, o);
638                if (r!= BindInterceptor.DEFAULT)    return r; // taken over by the interceptor
639            }
640
641            if(o==null || o instanceof JSONNull) {
642                // this method returns null if the type is not primitive, which works.
643                return ReflectionUtils.getVmDefaultValueFor(type);
644            }
645
646            if (type==JSONArray.class) {
647                if (o instanceof JSONArray) return o;
648
649                JSONArray a = new JSONArray();
650                a.add(o);
651                return a;
652            }
653
654            Lister l = Lister.create(type,genericType);
655
656            if (o instanceof JSONObject) {
657                JSONObject j = (JSONObject) o;
658
659                if (j.isNullObject())   // another flavor of null. json-lib sucks.
660                    return ReflectionUtils.getVmDefaultValueFor(type);
661
662                if(l==null) {// single value conversion
663                    try {
664                        Class actualType = type;
665                        String className = null;
666                        if(j.has("stapler-class")) {
667                            // deprecated as of 2.4-jenkins-4 but left here for a while until we are sure nobody uses this
668                            className = j.getString("stapler-class");
669                            LOGGER.log(FINE, "stapler-class is deprecated in favor of $class: {0}", className);
670                        }
671                        if(j.has("$class")) {
672                            className = j.getString("$class");
673                        }
674
675                        if (className != null) {
676                            // sub-type is specified in JSON.
677                            // note that this can come from malicious clients, so we need to make sure we don't have security issues.
678
679                            ClassLoader cl = stapler.getWebApp().getClassLoader();
680                            try {
681                                Class<?> subType = cl.loadClass(className);
682                                if(!actualType.isAssignableFrom(subType))
683                                    throw new IllegalArgumentException("Specified type "+subType+" is not assignable to the expected "+actualType);
684                                actualType = (Class)subType; // I'm being lazy here
685                            } catch (ClassNotFoundException e) {
686                                throw new IllegalArgumentException("Class "+className+" is specified in JSON, but no such class found in "+cl,e);
687                            }
688                        }
689
690                        return instantiate(actualType, j);
691                    } catch (IllegalArgumentException e) {
692                        JSONObject sanitizedJson = getWebApp().getJsonInErrorMessageSanitizer().sanitize(j);
693                        throw new IllegalArgumentException("Failed to instantiate "+type+" from "+sanitizedJson,e);
694                    }
695                } else {// collection conversion
696                    if(j.has("stapler-class-bag")) {
697                        // this object is a hash from class names to their parameters
698                        // build them into a collection via Lister
699
700                        ClassLoader cl = stapler.getWebApp().getClassLoader();
701                        for (Map.Entry<String,Object> e : (Set<Map.Entry<String,Object>>)j.entrySet()) {
702                            Object v = e.getValue();
703
704                            String className = e.getKey().replace('-','.'); // decode JSON-safe class name escaping
705                            try {
706                                Class<?> itemType = cl.loadClass(className);
707                                if (v instanceof JSONObject) {
708                                    l.add(bindJSON(itemType, (JSONObject) v));
709                                }
710                                if (v instanceof JSONArray) {
711                                    for(Object i : bindJSONToList(itemType, (JSONArray) v))
712                                        l.add(i);
713                                }
714                            } catch (ClassNotFoundException e1) {
715                                // ignore unrecognized element
716                            }
717                        }
718                    } else if (Enum.class.isAssignableFrom(l.itemType)) {
719                        // this is a hash of element names as enum constant names
720                        for (Map.Entry<String,Object> e : (Set<Map.Entry<String,Object>>)j.entrySet()) {
721                            Object v = e.getValue();
722                            if (v==null || (v instanceof Boolean && !(Boolean)v))
723                                continue;       // skip if the value is null or false
724
725                            l.add(Enum.valueOf(l.itemType,e.getKey()));
726                        }
727                    } else {
728                        // only one value given to the collection
729                        l.add(new TypePair(l.itemGenericType,l.itemType).convertJSON(j));
730                    }
731                    return l.toCollection();
732                }
733            }
734            if (o instanceof JSONArray) {
735                if (l == null)
736                    throw new WrongTypeException(String.format("Got type array but no lister class found for type %s", type));
737                JSONArray a = (JSONArray) o;
738                TypePair itemType = new TypePair(l.itemGenericType,l.itemType);
739                for (Object item : a)
740                    l.add(itemType.convertJSON(item));
741                return l.toCollection();
742            }
743
744            if(Enum.class.isAssignableFrom(type))
745                return Enum.valueOf(type,o.toString());
746
747            if (l==null) {// single value conversion
748                Converter converter = Stapler.lookupConverter(type);
749                if (converter == null) {
750                    if (type == Object.class) {
751                        return o;
752                    }
753                    throw new IllegalArgumentException("Unable to convert to "+type);
754                }
755
756                return converter.convert(type,o);
757            } else {// single value in a collection
758                Converter converter = Stapler.lookupConverter(l.itemType);
759                if (converter == null) {
760                    if (l.itemType == Object.class) {
761                        l.add(o);
762                    } else {
763                        throw new IllegalArgumentException("Unable to convert to "+l.itemType);
764                    }
765                } else {
766                    l.add(converter.convert(l.itemType, o));
767                }
768                return l.toCollection();
769            }
770        }
771    }

773    /**
774     * Called after the actual type of the binding is figured out.
775     */
776    private Object instantiate(Class actualType, JSONObject j) {
777        Object r = bindInterceptor.instantiate(actualType,j);
778        if (r!=BindInterceptor.DEFAULT) return r;
779        for (BindInterceptor bi : getWebApp().bindInterceptors) {
780            r = bi.instantiate(actualType,j);
781            if (r!=BindInterceptor.DEFAULT) return r;
782        }
783
784        if (actualType==JSONObject.class || actualType==JSON.class) return actualType.cast(j);
785
786        String[] names = new ClassDescriptor(actualType).loadConstructorParamNames();
787
788        // the actual arguments to invoke the constructor with.
789        Object[] args = new Object[names.length];
790
791        // constructor
792        Constructor c = findConstructor(actualType, names.length);
793        Class[] types = c.getParameterTypes();
794        Type[] genTypes = c.getGenericParameterTypes();
795
796        // convert parameters
797        for( int i=0; i<names.length; i++ ) {
798            try {
799                args[i] = bindJSON(genTypes[i],types[i],j.get(names[i]));
800            } catch (IllegalArgumentException e) {
801                throw new IllegalArgumentException("Failed to convert the "+names[i]+" parameter of the constructor "+c,e);
802            }
803        }
804
805        Object o = injectSetters(invokeConstructor(c, args), j, Arrays.asList(names));
806        o = bindResolve(o,j);
807
808        return o;
809    }

822    /**
823     * Performs {@link DataBoundSetter} injections.
824     *
825     * @param exclusions
826     *      Properties that are already injected through the constructor, thus not subject of the setter injection.
827     */
828    private <T> T injectSetters(T r, JSONObject j, Collection<String> exclusions) {
829        // try to assign rest of the properties
830        OUTER:
831        for (String key : (Set<String>)j.keySet()) {
832            if (!exclusions.contains(key)) {
833                try {
834                    // try field injection first
835                    for (Class c=r.getClass(); c!=null; c=c.getSuperclass()) {
836                        try {
837                            Field f = c.getDeclaredField(key);
838                            if (f.getAnnotation(DataBoundSetter.class)!=null) {
839                                try {
840                                    f.set(r, bindJSON(f.getGenericType(), f.getType(), j.get(key)));
841                                } catch(IllegalAccessException e) {
842                                    LOGGER.warning(IllegalReflectiveAccessLogHandler.get(e));
843                                    f.setAccessible(true);
844                                    f.set(r, bindJSON(f.getGenericType(), f.getType(), j.get(key)));
845                                }
846                                continue OUTER;
847                            }
848                        } catch (NoSuchFieldException e) {
849                            // recurse into parents
850                        }
851                    }
852
853                    Method wm = findDataBoundSetter(r.getClass(), key);
854                    if (wm==null)   continue;
855
856                    Class<?>[] pt = wm.getParameterTypes();
857                    assert pt.length==1;
858
859                    // only invoking public methods for security reasons
860                    wm.invoke(r, bindJSON(wm.getGenericParameterTypes()[0], pt[0], j.get(key)));
861                } catch (IllegalAccessException e) {
862                    LOGGER.log(WARNING, "Cannot access property " + key + " of " + r.getClass(), e);
863                } catch (InvocationTargetException e) {
864                    LOGGER.log(WARNING, "Cannot access property " + key + " of " + r.getClass(), e);
865                }
866            }
867        }
868
869        invokePostConstruct(getWebApp().getMetaClass(r).getPostConstructMethods(), r);
870
871        return r;
872    }
873
874    private Method findDataBoundSetter(Class c, String name) {
875        // look for public setter that has the matching name
876        for ( ; c!=null; c=c.getSuperclass()) {
877            for (Method m : c.getDeclaredMethods()) {
878                if (!Modifier.isPublic(m.getModifiers())
879                 || !m.getName().startsWith("set")
880                 || m.getParameterTypes().length!=1
881                 || !m.isAnnotationPresent(DataBoundSetter.class))
882                    continue;
883
884                String propertyName = Introspector.decapitalize(m.getName().substring(3));
885                if (!name.equals(propertyName))
886                    continue;   // not the name we are looking for
887
888                return m;
889            }
890        }
891        return null;
892    }

894    /**
895     * Invoke PostConstruct method from the base class to subtypes.
896     */
897    private void invokePostConstruct(SingleLinkedList<MethodRef> methods, Object r) {
898        if (methods.isEmpty())  return;
899
900        invokePostConstruct(methods.tail,r);
901        try {
902            methods.head.invoke(r);
903        } catch (InvocationTargetException e) {
904            throw new IllegalArgumentException("Unable to post-construct "+r,e);
905        } catch (IllegalAccessException e) {
906            throw (Error)new IllegalAccessError().initCause(e);
907        }
908
909    }

911    /**
912     * Gets the type of the field/property designate by the given name.
913     */
914    private TypePair getPropertyType(Object bean, String name) throws IllegalAccessException, InvocationTargetException {
915        try {
916            PropertyDescriptor propDescriptor = PropertyUtils.getPropertyDescriptor(bean, name);
917            if(propDescriptor!=null) {
918                Method m = propDescriptor.getWriteMethod();
919                if(m!=null)
920                    return new TypePair(m.getGenericParameterTypes()[0], m.getParameterTypes()[0]);
921            }
922        } catch (NoSuchMethodException e) {
923            // no such property
924        }
925
926        // try a field
927        try {
928            return new TypePair(bean.getClass().getField(name));
929        } catch (NoSuchFieldException e) {
930            // no such field
931        }
932
933        return null;
934    }

936    /**
937     * Sets the property/field value of the given name, by performing a value type conversion if necessary.
938     */
939    private static void copyProperty(Object bean, String name, Object value) throws IllegalAccessException, InvocationTargetException {
940        PropertyDescriptor propDescriptor;
941        try {
942            propDescriptor =
943                PropertyUtils.getPropertyDescriptor(bean, name);
944        } catch (NoSuchMethodException e) {
945            propDescriptor = null;
946        }
947        if ((propDescriptor != null) &&
948            (propDescriptor.getWriteMethod() == null)) {
949            propDescriptor = null;
950        }
951        if (propDescriptor != null) {
952            Converter converter = Stapler.lookupConverter(propDescriptor.getPropertyType());
953            if (converter != null)
954                value = converter.convert(propDescriptor.getPropertyType(), value);
955            try {
956                PropertyUtils.setSimpleProperty(bean, name, value);
957            } catch (NoSuchMethodException e) {
958                throw new NoSuchMethodError(e.getMessage());
959            }
960            return;
961        }
962
963        // try a field
964        try {
965            Field field = bean.getClass().getField(name);
966            Converter converter = ConvertUtils.lookup(field.getType());
967            if (converter != null)
968                value = converter.convert(field.getType(), value);
969            field.set(bean,value);
970        } catch (NoSuchFieldException e) {
971            // no such field
972        }
973    }

975    private void parseMultipartFormData() throws ServletException {
976        if(parsedFormData!=null)    return;
977
978        parsedFormData = new HashMap<String,FileItem>();
979        parsedFormDataFormFields = new HashMap<String, String>();
980        ServletFileUpload upload = new ServletFileUpload(new DiskFileItemFactory());
981        try {
982            for( FileItem fi : (List<FileItem>)upload.parseRequest(this) ) {
983                parsedFormData.put(fi.getFieldName(),fi);
984                if (fi.isFormField()) {
985                    parsedFormDataFormFields.put(fi.getFieldName(),fi.getString());
986                }
987            }
988        } catch (FileUploadException e) {
989            throw new ServletException(e);
990        }
991    }

1000            LOGGER.log(SEVERE, "Error parsing multipart/form-data.", e);
1001        }
1002        return parsedFormDataFormFields;

1005    public JSONObject getSubmittedForm() throws ServletException {
1006        final String method = this.getMethod();
1007        if (!ALLOWED_HTTP_VERBS_FOR_FORMS.contains(method)) {
1008            throw HttpResponses.errorWithoutStack(SC_BAD_REQUEST, "Form submission expected but a " + method + " request was sent");
1009        }
1010        if(structuredForm==null) {
1011            String p = null;
1012            boolean isSubmission; // for error diagnosis, if something is submitted, set to true
1013
1014            if(isMultipart()) {
1015                isSubmission=true;
1016                parseMultipartFormData();
1017                FileItem item = parsedFormData.get("json");
1018                if(item!=null) {
1019                    if (item.getContentType() == null && getCharacterEncoding() != null) {
1020                        // JENKINS-11543: If client doesn't set charset per part, use request encoding
1021                        try {
1022                            p = item.getString(getCharacterEncoding());
1023                        } catch (java.io.UnsupportedEncodingException uee) {
1024                            LOGGER.log(WARNING, "Request has unsupported charset, using default for 'json' parameter", uee);
1025                            p = item.getString();
1026                        }
1027                    } else {
1028                        p = item.getString();
1029                    }
1030                }
1031            } else {
1032                p = getParameter("json");
1033                isSubmission = !getParameterMap().isEmpty();
1034            }
1035            
1036            if(p==null || p.length() == 0) {
1037                // no data submitted
1038                try {
1039                    StaplerResponse rsp = Stapler.getCurrentResponse();
1040                    if(isSubmission)
1041                        rsp.sendError(SC_BAD_REQUEST,"This page expects a form submission");
1042                    else
1043                        rsp.sendError(SC_BAD_REQUEST,"Nothing is submitted");
1044                    throw new ServletException("This page expects a form submission but had only " + getParameterMap());
1045                } catch (IOException e) {
1046                    throw new ServletException(e);
1047                }
1048            }
1049            try {
1050                structuredForm = JSONObject.fromObject(p);
1051            } catch (JSONException e) {
1052                throw new ServletException("Failed to parse JSON:" + p, e);
1053            }
1054        }
1055        return structuredForm;
1056    }

1059        String ct = getContentType();
1060        return (ct!=null && ct.startsWith("multipart/"));
1061    }
