001package org.kohsuke.stapler.export;
002
003import static org.hamcrest.MatcherAssert.assertThat;
004import static org.hamcrest.Matchers.contains;
005
006import org.junit.Test;
007
008import java.util.Arrays;
009import java.util.LinkedHashSet;
010import java.util.List;
011import java.util.Set;
012
013
014public class RangeTest {
015    String[] array = new String[]{"a", "b", "c", "d", "e", "f"};
016    List<String> list = Arrays.asList(array);
017    Set<String> set = new LinkedHashSet<String>(list);
018
019    @Test
020    public void normalRange() {
021        Range r = new Range(2,4);
022        assertThat(r.apply(array), contains("c", "d"));
023        assertThat(r.apply(list), contains("c", "d"));
024        assertThat(r.apply(set), contains("c", "d"));
025    }
026
027    @Test
028    public void maxOnlyRange() {
029        Range r = new Range(-1,2);
030        assertThat(r.apply(array), contains("a", "b"));
031        assertThat(r.apply(list), contains("a", "b"));
032        assertThat(r.apply(set), contains("a", "b"));
033    }
034
035    @Test
036    public void minOnlyRange() {
037        Range r = new Range(4,Integer.MAX_VALUE);
038        assertThat(r.apply(array), contains("e", "f"));
039        assertThat(r.apply(list), contains("e", "f"));
040        assertThat(r.apply(set), contains("e", "f"));
041    }
042}
