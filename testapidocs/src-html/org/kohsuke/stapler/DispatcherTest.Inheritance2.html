047    public final VerbMatch verbMatch = new VerbMatch();
048
049    public class VerbMatch {
050        @WebMethod(name="") @GET
051        public HttpResponse doGet() {
052            return HttpResponses.plainText("Got GET");
053        }
054
055        @WebMethod(name="") @POST
056        public HttpResponse doPost() {
057            return HttpResponses.plainText("Got POST");
058        }
059    }
060
061    /**
062     * Tests the dispatching of WebMethod based on verb
063     */
064    public void testVerbMatch() throws Exception {
065        WebClient wc = new WebClient();
066
067        check(wc, HttpMethod.GET);
068        check(wc, HttpMethod.POST);
069        try {
070            check(wc, HttpMethod.DELETE);
071            fail("There's no route for DELETE");
072        } catch (FailingHttpStatusCodeException e) {
073            assertEquals(404, e.getStatusCode());
074        }
075    }
076
077    private void check(WebClient wc, HttpMethod m) throws java.io.IOException {
078        TextPage p = wc.getPage(new WebRequestSettings(new URL(url, "verbMatch/"), m));
079        assertEquals("Got "+m.name()+"\n", p.getContent());
080    }

115    public final InterceptorStage interceptorStage = new InterceptorStage();
116
117    public class InterceptorStage {
118        @POST
119        public HttpResponse doFoo(@JsonBody Point body) {
120            return HttpResponses.plainText(body.x+","+body.y);
121        }
122    }
123
124    public static class Point {
125        public int x,y;
126    }
127
128    /**
129     * POST annotation selection needs to happen before databinding of the parameter happens.
130     */
131    public void testInterceptorStage() throws Exception {
132        WebClient wc = new WebClient();
133        try {
134            wc.getPage(new URL(url, "interceptorStage/foo"));
135            fail("Expected 404");
136        } catch (FailingHttpStatusCodeException e) {
137            assertEquals(404, e.getStatusCode());
138        }
139
140        WebRequestSettings req = new WebRequestSettings(new URL(url, "interceptorStage/foo"), HttpMethod.POST);
141        req.setAdditionalHeader("Content-Type","application/json");
142        req.setRequestBody("{x:3,y:5}");
143        TextPage p = wc.getPage(req);
144        assertEquals("3,5\n",p.getContent());
145
146    }

168    public final Inheritance inheritance = new Inheritance2();
169    public class Inheritance {
170        @WebMethod(name="foo")
171        public HttpResponse doBar(@QueryParameter String q) {
172            return HttpResponses.plainText("base");
173        }
174    }
175
176    public class Inheritance2 extends Inheritance {
177        @Override
178        public HttpResponse doBar(String q) {
179            return HttpResponses.plainText(q);
180        }
181    }
182
183    public void testInheritance() throws Exception {
184        WebClient wc = new WebClient();
185
186        // the request should get to the overriding method and it should still see all the annotations in the base type
187        TextPage p = wc.getPage(new URL(url, "inheritance/foo?q=abc"));
188        assertEquals("abc\n", p.getContent());
189
190        // doBar is a web method for 'foo', so bar endpoint shouldn't respond
191        try {
192            wc.getPage(new URL(url, "inheritance/bar"));
193            fail();
194        } catch (FailingHttpStatusCodeException e) {
195            assertEquals(404,e.getStatusCode());
196        }
197    }

199    public final PutInheritance putInheritance = new PutInheritanceImpl();
200    public abstract class PutInheritance {
201        @WebMethod(name="foo") @PUT
202        public abstract HttpResponse doBar(StaplerRequest req) throws IOException;
203
204        @POST
205        public HttpResponse doAcme(StaplerRequest req) throws IOException {
206            return HttpResponses.plainText("POST: "+IOUtils.toString(req.getInputStream()));
207        }
208    }
209
210    public class PutInheritanceImpl extends PutInheritance{
211        @Override
212        public HttpResponse doBar(StaplerRequest req) throws IOException {
213            return HttpResponses.plainText(IOUtils.toString(req.getInputStream())+" World!");
214        }
215    }
216
217    public void testPutInheritance() throws Exception {
218        WebClient wc = new WebClient();
219
220        // the request should get to the overriding method and it should still see all the annotations in the base type
221        WebRequestSettings wrs = new WebRequestSettings(new URL(url, "putInheritance/foo"), HttpMethod.PUT);
222        wrs.setRequestBody("Hello");
223        TextPage p = wc.getPage(wrs);
224        assertEquals("Hello World!\n", p.getContent());
225
226        // doBar is a web method for 'foo', so bar endpoint shouldn't respond
227        try {
228            wc.getPage(new URL(url, "putInheritance/bar"));
229            fail();
230        } catch (FailingHttpStatusCodeException e) {
231            assertEquals(404,e.getStatusCode());
232        }
233
234        //Invoke Post as well
235        wrs = new WebRequestSettings(new URL(url, "putInheritance/acme"), HttpMethod.POST);
236        wrs.setRequestBody("Hello");
237        p = wc.getPage(wrs);
238        assertEquals("POST: Hello\n", p.getContent());
239    }

246    public final RequirePostOnBase requirePostOnBase = new RequirePostOnBase2();
247    public abstract class RequirePostOnBase {
248        int hit;
249        @RequirePOST
250        public abstract void doSomething();
251    }
252
253    public class RequirePostOnBase2 extends RequirePostOnBase {
254        @Override
255        public void doSomething() {
256            hit++;
257        }
258    }
259
260    public void testRequirePostOnBase() throws Exception {
261        WebClient wc = new WebClient();
262        URL url = new URL(this.url, "requirePostOnBase/something");
263
264        try {
265            wc.getPage(url);
266            fail();
267        } catch (FailingHttpStatusCodeException e) {
268            assertEquals(HttpServletResponse.SC_METHOD_NOT_ALLOWED, e.getStatusCode());
269        }
270
271        // POST should succeed
272        wc.getPage(new WebRequestSettings(url, HttpMethod.POST));
273        assertEquals(1, requirePostOnBase.hit);
274    }
