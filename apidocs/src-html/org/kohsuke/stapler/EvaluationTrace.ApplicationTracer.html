024package org.kohsuke.stapler;
025
026import javax.annotation.Nonnull;
027import java.util.List;
028import java.util.ArrayList;
029import java.io.PrintWriter;
030import java.util.ServiceLoader;
031import java.util.logging.Level;
032import java.util.logging.Logger;
033
034import static org.kohsuke.stapler.Stapler.escape;
035
036/**
037 * Remembers the {@link Stapler#invoke(RequestImpl, ResponseImpl, Object)}
038 * evaluation traces.
039 *
040 * @author Kohsuke Kawaguchi
041 */
042public class EvaluationTrace {
043    private final List<String> traces = new ArrayList<String>();
044
045    private static final Logger LOGGER = Logger.getLogger(EvaluationTrace.class.getName());
046
047    public void trace(StaplerResponse rsp, String msg) {
048        traces.add(msg);
049        // Firefox Live HTTP header plugin cannot nicely render multiple headers
050        // with the same name, so give each one unique name.
051        rsp.addHeader(String.format("Stapler-Trace-%03d",traces.size()),msg.replace("\n","\\n").replace("\r","\\r"));
052    }
053    
054    public void printHtml(PrintWriter w) {
055        for (String trace : traces)
056            w.println(escape(trace));
057    }
058
059    public static EvaluationTrace get(StaplerRequest req) {
060        EvaluationTrace et = (EvaluationTrace) req.getAttribute(KEY);
061        if(et==null)
062            req.setAttribute(KEY,et=new EvaluationTrace());
063        return et;
064    }
065
066    /**
067     * Used for counting trace header.
068     */
069    private static final String KEY = EvaluationTrace.class.getName();
070
071    public abstract static class ApplicationTracer {
072        protected abstract void record(StaplerRequest req, String message);
073
074        public static void trace(StaplerRequest req, String message) {
075            List<ApplicationTracer> tracers = getTracers();
076            for (ApplicationTracer tracer : tracers) {
077                tracer.record(req, message);
078            }
079        }
080
081        private static volatile List<ApplicationTracer> tracers;
082
083        @Nonnull
084        private static List<ApplicationTracer> getTracers() {
085            synchronized (ApplicationTracer.class) {
086                if (tracers == null) {
087                    List<ApplicationTracer> t = new ArrayList<>();
088                    for (ApplicationTracer tracer : ServiceLoader.load(EvaluationTrace.ApplicationTracer.class, Stapler.getCurrent().getWebApp().getClassLoader())) {
089                        try {
090                            t.add(tracer);
091                        } catch (Exception e) {
092                            // robustness
093                            if (LOGGER.isLoggable(Level.FINE)) {
094                                LOGGER.log(Level.FINE, "Exception thrown when notifying tracer", e);
095                            }
096                        }
097                    }
098                    tracers = t;
099                }
100            }
101            return tracers;
102        }
103    }
104}
