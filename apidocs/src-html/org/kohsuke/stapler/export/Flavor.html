024package org.kohsuke.stapler.export;
025
026import org.kohsuke.stapler.StaplerResponse;
027
028import java.io.IOException;
029import java.io.Writer;
030
031/**
032 * Export flavor.
033 *
034 * @author Kohsuke Kawaguchi
035 */
036public enum Flavor {
037    JSON("application/json;charset=UTF-8") {
038        public DataWriter createDataWriter(Object bean, Writer w, ExportConfig config) throws IOException {
039            return new JSONDataWriter(w,config);
040        }
041    },
042    PYTHON("text/x-python;charset=UTF-8") {
043        public DataWriter createDataWriter(Object bean, Writer w, ExportConfig config) throws IOException {
044            return new PythonDataWriter(w,config);
045        }
046    },
047    RUBY("text/x-ruby;charset=UTF-8") {
048        public DataWriter createDataWriter(Object bean, Writer w, ExportConfig config) throws IOException {
049            return new RubyDataWriter(w,config);
050        }
051    },
052    XML("application/xml;charset=UTF-8") {
053        public DataWriter createDataWriter(Object bean, Writer w, ExportConfig config) throws IOException {
054            // TODO: support pretty printing
055            return new XMLDataWriter(bean,w);
056        }
057    };
058
059    /**
060     * Content-type of this flavor, including charset "UTF-8".
061     */
062    public final String contentType;
063
064    Flavor(String contentType) {
065        this.contentType = contentType;
066    }
067
068    public DataWriter createDataWriter(Object bean, StaplerResponse rsp) throws IOException {
069        return createDataWriter(bean,rsp.getWriter());
070    }
071    public DataWriter createDataWriter(Object bean, Writer w) throws IOException {
072        return createDataWriter(bean,w,new ExportConfig());
073    }
074    public abstract DataWriter createDataWriter(Object bean, Writer w, ExportConfig config) throws IOException;
075}
