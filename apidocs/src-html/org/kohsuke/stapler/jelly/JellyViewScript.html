024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.JellyContext;
027import org.apache.commons.jelly.JellyException;
028import org.apache.commons.jelly.JellyTagException;
029import org.apache.commons.jelly.Script;
030import org.apache.commons.jelly.XMLOutput;
031import org.kohsuke.stapler.lang.Klass;
032
033import java.net.URL;
034import java.util.logging.Logger;
035
036/**
037 * Represents a loaded Jelly view script that remembers where it came from.
038 *
039 * @author Kohsuke Kawaguchi
040 */
041public final class JellyViewScript implements Script {
042
043    private static final Logger LOGGER = Logger.getLogger(JellyViewScript.class.getName());
044
045    /**
046     * Which class is this view loaded from?
047     * @deprecated as of 1.177 
048     *      Use {@link #fromKlass}
049     */
050    public final Class from;
051    
052    public final Klass<?> fromKlass;
053    
054    /**
055     * Full URL that points to the source of the script.
056     */
057    public final URL source;
058
059    private Script base;
060
061    /**
062     * @deprecated as of 1.177
063     *      use {@link #JellyViewScript(Klass, URL, Script)}
064     */
065    public JellyViewScript(Class from, URL source, Script base) {
066        this.from = from;
067        this.fromKlass = Klass.java(from);
068        this.source = source;
069        this.base = base;
070    }
071    
072    public JellyViewScript(Klass from, URL source, Script base) {
073        this.from = from.toJavaClass();
074        this.fromKlass = from;
075        this.source = source;
076        this.base = base;
077    }
078
079    public Script compile() throws JellyException {
080        base = base.compile();
081        return this;
082    }
083
084    public void run(JellyContext context, XMLOutput output) throws JellyTagException {
085        Thread t = Thread.currentThread();
086        String n = t.getName();
087        // JellyViewScript.getName() is a bit too verbose for this purpose (we do not really need the package prefix):
088        String url = source.toExternalForm();
089        String c = from.getName();
090        c = c.substring(c.lastIndexOf('.') + 1);
091        String n2 = n + " " + c.replace('$', '/') + "/" + url.substring(url.lastIndexOf('/') + 1);
092        t.setName(n2);
093        LOGGER.fine(n2);
094        try {
095            base.run(context,output);
096        } finally {
097            t.setName(n);
098        }
099    }
100
101    public String getName() {
102        // get to the file name portion
103        String url = source.toExternalForm();
104        url = url.substring(url.lastIndexOf('/') +1);
105        url = url.substring(url.lastIndexOf('\\') +1);
106
107        return from.getName().replace('.','/').replace('$','/')+'/'+url;
108    }
109}
