024package org.kohsuke.stapler.bind;
025
026import org.kohsuke.stapler.HttpResponse;
027import org.kohsuke.stapler.MetaClass;
028import org.kohsuke.stapler.WebApp;
029
030import java.lang.reflect.Method;
031import java.util.Arrays;
032import java.util.Collection;
033import java.util.Collections;
034
035/**
036 * Handles to the object bound via {@link BoundObjectTable}.
037 *
038 * As {@link HttpResponse}, this object generates a redirect to the URL that it points to.
039 *
040 * @author Kohsuke Kawaguchi
041 * @see MetaClass#buildDispatchers
042 */
043public abstract class Bound implements HttpResponse {
044    /**
045     * Explicitly unbind this object. The referenced object
046     * won't be bound to URL anymore.
047     */
048    public abstract void release();
049
050    /**
051     * The URL where the object is bound to. This method
052     * starts with '/' and thus always absolute within the current web server.
053     */
054    public abstract String getURL();
055
056    /**
057     * Gets the bound object.
058     */
059    public abstract Object getTarget();
060
061    /**
062     * Returns a JavaScript expression which evaluates to a JavaScript proxy that
063     * talks back to the bound object that this handle represents.
064     */
065    public final String getProxyScript() {
066        StringBuilder buf = new StringBuilder("makeStaplerProxy('").append(getURL()).append("','").append(
067                WebApp.getCurrent().getCrumbIssuer().issueCrumb()
068        ).append("',[");
069
070        boolean first=true;
071        for (Method m : getTarget().getClass().getMethods()) {
072            Collection<String> names;
073            if (m.getName().startsWith("js")) {
074                names = Collections.singleton(camelize(m.getName().substring(2)));
075            } else {
076                JavaScriptMethod a = m.getAnnotation(JavaScriptMethod.class);
077                if (a!=null) {
078                    names = Arrays.asList(a.name());
079                    if (names.isEmpty())
080                        names = Collections.singleton(m.getName());
081                } else
082                    continue;
083            }
084
085            for (String n : names) {
086                if (first)  first = false;
087                else        buf.append(',');
088                buf.append('\'').append(n).append('\'');
089            }
090        }
091        buf.append("])");
092        
093        return buf.toString();
094    }
095
096    private static String camelize(String name) {
097        return Character.toLowerCase(name.charAt(0))+name.substring(1);
098    }
099}
