015public class DataBindingTest extends TestCase {

017        public int a;
018        String myB;

022        }

024    public void test1() {
025        JSONObject json = new JSONObject();
026        json.put("a",123);
027        json.put("b","string");
028
029        Data data = bind(json, new Data());
030        assertEquals(123,data.a);
031        assertEquals("string",data.myB);
032    }

034    public class DataEnumSet {
035        public EnumSet<Proxy.Type> set;
036    }
037
038    public void testEnumSet() {
039        JSONObject json = new JSONObject();
040        json.put("DIRECT",false);
041        json.put("HTTP",true);
042        json.put("SOCKS",null);
043
044        JSONObject container = new JSONObject();
045        container.put("set",json);
046
047        DataEnumSet data = bind(container,new DataEnumSet());
048        assertTrue(data.set.contains(Proxy.Type.HTTP));
049        assertFalse(data.set.contains(Proxy.Type.DIRECT));
050        assertFalse(data.set.contains(Proxy.Type.SOCKS));
051
052    }

054    public void testFromStaplerMethod() throws Exception {
055        MockRequest mr = new MockRequest();
056        mr.getParameterMap().put("a","123");
057        mr.getParameterMap().put("b","string");
058        RequestImpl req = new RequestImpl(new Stapler(), mr, Collections.<AncestorImpl>emptyList(), null);
059        new Function.InstanceFunction(getClass().getMethod("doFromStaplerMethod",StaplerRequest.class,int.class,Binder.class))
060                .bindAndInvoke(this,req,null);
061    }
062
063    public void doFromStaplerMethod(StaplerRequest req, @QueryParameter int a, Binder b) {
064        assertEquals(123,a);
065        assertSame(req,b.req);
066        assertEquals("string",b.b);
067
068    }
069
070    public static class Binder {
071        StaplerRequest req;
072        String b;
073
074        @CapturedParameterNames({"req","b"})
075        public static Binder fromStapler(StaplerRequest req, @QueryParameter String b) {
076            Binder r = new Binder();
077            r.req = req;
078            r.b = b;
079            return r;
080        }
081    }
082
083    public void testCustomConverter() throws Exception {
084        ReferToObjectWithCustomConverter r = bind("{data:'1,2'}", ReferToObjectWithCustomConverter.class);
085        assertEquals(r.data.x,1);
086        assertEquals(r.data.y,2);
087    }
088
089    public static class ReferToObjectWithCustomConverter {
090        final ObjectWithCustomConverter data;
091
092        @DataBoundConstructor

096    }

098    public void testNullToFalse() throws Exception {
099        TwoBooleans r = bind("{a:false}", TwoBooleans.class);
100        assertFalse(r.a);
101        assertFalse(r.b);
102    }
103
104    public static class TwoBooleans {
105        private boolean a,b;
106

109            this.a = a;

111        }
112    }
113
114    public void testScalarToArray() throws Exception {
115        ScalarToArray r = bind("{a:'x',b:'y',c:5,d:6}", ScalarToArray.class);
116        assertEquals("x",r.a[0]);
117        assertEquals("y",r.b.get(0));
118        assertEquals(5,(int)r.c[0]);
119        assertEquals(6,(int)r.d.get(0));
120    }
121
122    public static class ScalarToArray {
123        private String[] a;
124        private List<String> b;
125        private Integer[] c;
126        private List<Integer> d;
127
128        @DataBoundConstructor
129        public ScalarToArray(String[] a, List<String> b, Integer[] c, List<Integer> d) {
130            this.a = a;
131            this.b = b;
132            this.c = c;
133            this.d = d;
134        }
135    }
136
137    private <T> T bind(String json, Class<T> type) {
138        RequestImpl req = new RequestImpl(new Stapler(), new MockRequest(), Collections.<AncestorImpl>emptyList(), null);
139        return req.bindJSON(type, JSONObject.fromObject(json));
140    }
141
142    private <T> T bind(JSONObject json, T bean) {
143        RequestImpl req = new RequestImpl(new Stapler(), new MockRequest(), Collections.<AncestorImpl>emptyList(), null);
144        req.bindJSON(bean,json);
145        return bean;
146    }
147
148    public static class RawBinding {
149        JSONObject x;
150        JSONArray y;
151        @DataBoundConstructor
152        public RawBinding(JSONObject x, JSONArray y) {
153            this.x = x;
154            this.y = y;
155        }
156    }
157
158    public void testRaw() {
159        RawBinding r = bind("{x:{a:true,b:1},y:[1,2,3]}", RawBinding.class);
160
161        // array coersion on y
162        RawBinding r2 = bind("{x:{a:true,b:1},y:{p:true}}", RawBinding.class);
163        JSONObject o = (JSONObject)r2.y.get(0);
164        assertTrue(o.getBoolean("p"));
165
166        // array coersion on y
167        RawBinding r3 = bind("{x:{a:true,b:1},y:true}", RawBinding.class);
168        assertTrue((Boolean)r3.y.get(0));
169    }
