001package org.kohsuke.stapler.compression;
002
003import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
004import org.kohsuke.stapler.AttributeKey;
005
006import javax.servlet.ServletContext;
007import javax.servlet.ServletException;
008import javax.servlet.http.HttpServletRequest;
009import javax.servlet.http.HttpServletResponse;
010import java.io.IOException;
011import java.io.PrintWriter;
012import java.io.StringWriter;
013import java.text.MessageFormat;
014import java.util.Date;
015
016import static org.kohsuke.stapler.Stapler.escape;
017
018/**
019 * Handles an exception caught by {@link CompressionFilter}.
020 *
021 * See {@link CompressionFilter} javadoc for why this exception needs to be handled
022 * by us and can't just be handled by the servlet container like it does all others.
023 *
024 * @author Kohsuke Kawaguchi
025 */
026public interface UncaughtExceptionHandler {
027    /**
028     * Called to render the exception as an HTTP response.
029     */
030    void reportException(Throwable e, ServletContext context, HttpServletRequest req, HttpServletResponse rsp)
031            throws ServletException, IOException;
032
033
034    UncaughtExceptionHandler DEFAULT = new UncaughtExceptionHandler() {
035        @SuppressFBWarnings(value = "XSS_SERVLET", justification = "Covered by the escape() method.")
036        public void reportException(Throwable e, ServletContext context, HttpServletRequest req, HttpServletResponse rsp) throws ServletException, IOException {
037            StringWriter sw = new StringWriter();
038            PrintWriter pw = new PrintWriter(sw);
039            e.printStackTrace(pw);
040            pw.close();
041
042            rsp.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
043            rsp.setContentType("text/html");
044            PrintWriter w = rsp.getWriter();
045            String message = e.getMessage();
046            w.print(MessageFormat.format("<html><head><title>Error {0}</title></head>\n" +
047                    "<body bgcolor=#fff><h1>Status Code: {0}</h1>Exception: {1}<br>Stacktrace: <pre>{2}</pre><br><hr>\n" +
048                    "<i>Generated by Stapler at {3}</i></body></html>",
049                    HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
050                    message != null ? escape(message) : "?",
051                    escape(sw.toString()),
052                    new Date().toString()
053            ));
054            w.close();
055        }
056    };
057}
