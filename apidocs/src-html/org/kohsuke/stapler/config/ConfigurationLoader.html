001package org.kohsuke.stapler.config;
002
003import com.google.common.base.Function;
004import org.apache.commons.beanutils.ConvertUtils;
005
006import java.beans.Introspector;
007import java.io.File;
008import java.io.FileInputStream;
009import java.io.IOException;
010import java.lang.reflect.InvocationHandler;
011import java.lang.reflect.Method;
012import java.lang.reflect.Proxy;
013import java.util.Comparator;
014import java.util.Map;
015import java.util.Properties;
016import java.util.TreeMap;
017
018import static org.kohsuke.stapler.config.Configuration.UNSPECIFIED;
019
020/**
021 * Provides a type-safe access to the configuration of the application.
022 *
023 * <p>
024 * Often web applications need to load configuration from outside. T
025 *
026 * <p>
027 * Typical usage would be {@code MyConfig config=ConfigurationLoad.from(...).as(MyConfig.class)}
028 * where the <tt>MyConfig</tt> interface defines a bunch of methods named after
029 * the property name:
030 *
031 * <pre>
032 * interface MyConfig {
033 *     File rootDir();
034 *     int retryCount();
035 *     String domainName();
036 *     ...
037 * }
038 * </pre>
039 *
040 * <p>
041 * Method calls translate to respective property lookup. For example, {@code config.rootDir()} would
042 * be equivalent to {@code new File(properties.get("rootDir"))}.
043 *
044 * <p>
045 * The method name can include common prefixes, such as "get", "is", and "has", and those portions
046 * will be excluded from the property name. Thus the {@code rootDir()} could have been named {@code getRootDir()}.
047 *
048 *
049 * @author Kohsuke Kawaguchi
050 */
051public class ConfigurationLoader {
052    private final Function<String,String> source;
053
054    /**
055     * The caller should use one of the fromXyz methods.
056     */
057    private ConfigurationLoader(Function<String,String> source) {
058        this.source = source;
059    }
060
061    private static Properties load(File f) throws IOException {
062        Properties config = new Properties();
063        FileInputStream in = new FileInputStream(f);
064        try {
065            config.load(in);
066            return config;
067        } finally {
068            in.close();
069        }
070    }
071
072    /**
073     * Loads the configuration from the specified property file.
074     */
075    public static ConfigurationLoader from(File configPropertyFile) throws IOException {
076        return from(load(configPropertyFile));
077    }
078
079    /**
080     * Loads the configuration from the specified {@link Properties} object.
081     */
082    public static ConfigurationLoader from(final Properties props) throws IOException {
083        return new ConfigurationLoader(new Function<String, String>() {
084            public String apply(String from) {
085                return props.getProperty(from);
086            }
087        });
088    }
089
090    public static ConfigurationLoader from(final Map<String,String> props) throws IOException {
091        return new ConfigurationLoader(new Function<String, String>() {
092            public String apply(String from) {
093                return props.get(from);
094            }
095        });
096    }
097
098    /**
099     * Creates {@link ConfigurationLoader} that uses all the system properties as the source.
100     */
101    public static ConfigurationLoader fromSystemProperties() throws IOException {
102        return from(System.getProperties());
103    }
104
105    /**
106     * Creates {@link ConfigurationLoader} that uses environment variables as the source.
107     *
108     * Since environment variables are often by convention all caps, while system properties
109     * and other properties tend to be camel cased, this method creates a case-insensitive configuration
110     * (that allows retrievals by both "path" and "PATH" to fill this gap.
111     */
112    public static ConfigurationLoader fromEnvironmentVariables() throws IOException {
113        TreeMap<String, String> m = new TreeMap<String, String>(new Comparator<String>() {
114            public int compare(String o1, String o2) {
115                return o1.compareToIgnoreCase(o2);
116            }
117        });
118        m.putAll(System.getenv());
119        return from(m);
120    }
121
122    /**
123     * Creates a type-safe proxy that reads from the source specified by one of the fromXyz methods.
124     */
125    public <T> T as(Class<T> type) {
126        return type.cast(Proxy.newProxyInstance(type.getClassLoader(), new Class[]{type}, new InvocationHandler() {
127            public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
128                if (method.getDeclaringClass() == Object.class)
129                    return method.invoke(this, args);
130
131                Configuration cn = method.getAnnotation(Configuration.class);
132
133                Class<?> r = method.getReturnType();
134                String key = getKey(method,cn);
135
136                String v = source.apply(key);
137                if (v==null && cn!=null && !cn.defaultValue().equals(UNSPECIFIED))
138                    v = cn.defaultValue();
139
140                if (v==null)    return null;    // TODO: check how the primitive types are handled here
141
142                return ConvertUtils.convert(v,r);
143            }
144
145            private String getKey(Method method, Configuration c) {
146                if (c!=null && !c.name().equals(UNSPECIFIED))
147                    return c.name();        // name override
148
149                String n = method.getName();
150                for (String p : GETTER_PREFIX) {
151                    if (n.startsWith(p))
152                        return Introspector.decapitalize(n.substring(p.length()));
153                }
154                return n;
155            }
156        }));
157    }
158
159    private static final String[] GETTER_PREFIX = {"get","has","is"};
160}
