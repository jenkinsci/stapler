024package org.kohsuke.stapler.jelly;
025
026import org.apache.commons.jelly.XMLOutput;
027import org.dom4j.io.HTMLWriter;
028import org.dom4j.io.OutputFormat;
029import java.io.IOException;
030import java.io.OutputStream;
031import java.io.UnsupportedEncodingException;
032import java.io.Writer;
033
034/**
035 * Wrapper for XMLOutput using HTMLWriter that can turn off its HTML handling
036 * (if the Content-Type gets set to something other than text/html).
037 * 
038 * @author Alan.Harder@Sun.Com
039 */
040public class HTMLWriterOutput extends XMLOutput {
041    private HTMLWriter htmlWriter;
042    private OutputFormat format;
043
044    public static HTMLWriterOutput create(OutputStream out) throws UnsupportedEncodingException {
045        OutputFormat format = createFormat();
046        return new HTMLWriterOutput(new HTMLWriter(out, format), format, false);
047    }
048
049    public static HTMLWriterOutput create(Writer out, boolean escapeText) {
050        OutputFormat format = createFormat();
051        return new HTMLWriterOutput(new HTMLWriter(out, format), format, escapeText);
052    }
053
054    private static OutputFormat createFormat() {
055        OutputFormat format = new OutputFormat();
056        format.setXHTML(true);
057        // Only use short close for tags identified by HTMLWriter:
058        format.setExpandEmptyElements(true);
059        return format;
060    }
061
062    private HTMLWriterOutput(HTMLWriter hw, OutputFormat fmt, boolean escapeText) {
063        super(hw);
064        hw.setEscapeText(escapeText);
065        this.htmlWriter = hw;
066        this.format = fmt;
067    }
068
069    @Override public void close() throws IOException {
070        htmlWriter.close();
071    }
072
073    /**
074     * False to turn off HTML handling and reenable "/>" for any empty XML element.
075     * True to switch back to default mode with HTML handling.
076     */
077    public void useHTML(boolean enabled) {
078        htmlWriter.setEnabled(enabled);
079        format.setExpandEmptyElements(enabled);
080    }
081}
