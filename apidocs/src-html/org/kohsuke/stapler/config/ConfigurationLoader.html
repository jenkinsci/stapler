001package org.kohsuke.stapler.config;
002
003import java.util.function.Function;
004import java.io.InputStream;
005import java.nio.file.Files;
006import java.nio.file.StandardOpenOption;
007
008import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
009import org.apache.commons.beanutils.ConvertUtils;
010
011import java.beans.Introspector;
012import java.io.File;
013import java.io.IOException;
014import java.lang.reflect.InvocationHandler;
015import java.lang.reflect.Method;
016import java.lang.reflect.Proxy;
017import java.util.Comparator;
018import java.util.Map;
019import java.util.Properties;
020import java.util.TreeMap;
021
022import static org.kohsuke.stapler.config.Configuration.UNSPECIFIED;
023
024/**
025 * Provides a type-safe access to the configuration of the application.
026 *
027 * <p>
028 * Often web applications need to load configuration from outside. T
029 *
030 * <p>
031 * Typical usage would be {@code MyConfig config=ConfigurationLoad.from(...).as(MyConfig.class)}
032 * where the <tt>MyConfig</tt> interface defines a bunch of methods named after
033 * the property name:
034 *
035 * <pre>
036 * interface MyConfig {
037 *     File rootDir();
038 *     int retryCount();
039 *     String domainName();
040 *     ...
041 * }
042 * </pre>
043 *
044 * <p>
045 * Method calls translate to respective property lookup. For example, {@code config.rootDir()} would
046 * be equivalent to {@code new File(properties.get("rootDir"))}.
047 *
048 * <p>
049 * The method name can include common prefixes, such as "get", "is", and "has", and those portions
050 * will be excluded from the property name. Thus the {@code rootDir()} could have been named {@code getRootDir()}.
051 *
052 *
053 * @author Kohsuke Kawaguchi
054 */
055public class ConfigurationLoader {
056    private final Function<String,String> source;
057
058    /**
059     * The caller should use one of the fromXyz methods.
060     */
061    private ConfigurationLoader(Function<String,String> source) {
062        this.source = source;
063    }
064
065    private static Properties load(File f) throws IOException {
066        Properties config = new Properties();
067        try (InputStream in = Files.newInputStream(f.toPath(), StandardOpenOption.READ)){
068            config.load(in);
069            return config;
070        }
071    }
072
073    /**
074     * Loads the configuration from the specified property file.
075     */
076    public static ConfigurationLoader from(File configPropertyFile) throws IOException {
077        return from(load(configPropertyFile));
078    }
079
080    /**
081     * Loads the configuration from the specified {@link Properties} object.
082     */
083    public static ConfigurationLoader from(final Properties props) throws IOException {
084        return new ConfigurationLoader(from -> props.getProperty(from));
085    }
086
087    public static ConfigurationLoader from(final Map<String,String> props) throws IOException {
088        return new ConfigurationLoader(new Function<String, String>() {
089            public String apply(String from) {
090                return props.get(from);
091            }
092        });
093    }
094
095    /**
096     * Creates {@link ConfigurationLoader} that uses all the system properties as the source.
097     */
098    public static ConfigurationLoader fromSystemProperties() throws IOException {
099        return from(System.getProperties());
100    }
101
102    /**
103     * Creates {@link ConfigurationLoader} that uses environment variables as the source.
104     *
105     * Since environment variables are often by convention all caps, while system properties
106     * and other properties tend to be camel cased, this method creates a case-insensitive configuration
107     * (that allows retrievals by both "path" and "PATH" to fill this gap.
108     */
109    public static ConfigurationLoader fromEnvironmentVariables() throws IOException {
110        TreeMap<String, String> m = new TreeMap<String, String>(new Comparator<String>() {
111            public int compare(String o1, String o2) {
112                return o1.compareToIgnoreCase(o2);
113            }
114        });
115        m.putAll(System.getenv());
116        return from(m);
117    }
118
119    /**
120     * Creates a type-safe proxy that reads from the source specified by one of the fromXyz methods.
121     */
122    public <T> T as(Class<T> type) {
123        return type.cast(Proxy.newProxyInstance(type.getClassLoader(), new Class[]{type}, new InvocationHandler() {
124            public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
125                if (method.getDeclaringClass() == Object.class)
126                    return method.invoke(this, args);
127
128                Configuration cn = method.getAnnotation(Configuration.class);
129
130                Class<?> r = method.getReturnType();
131                String key = getKey(method,cn);
132
133                String v = source.apply(key);
134                if (v==null && cn!=null && !cn.defaultValue().equals(UNSPECIFIED))
135                    v = cn.defaultValue();
136
137                if (v==null)    return null;    // TODO: check how the primitive types are handled here
138
139                return ConvertUtils.convert(v,r);
140            }
141
142            private String getKey(Method method, Configuration c) {
143                if (c!=null && !c.name().equals(UNSPECIFIED))
144                    return c.name();        // name override
145
146                String n = method.getName();
147                for (String p : GETTER_PREFIX) {
148                    if (n.startsWith(p))
149                        return Introspector.decapitalize(n.substring(p.length()));
150                }
151                return n;
152            }
153        }));
154    }
155
156    private static final String[] GETTER_PREFIX = {"get","has","is"};
157}
