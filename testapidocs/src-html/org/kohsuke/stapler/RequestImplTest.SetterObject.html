059    public static class SetterObject {
060        private List<String> choices;
061
062        @DataBoundConstructor
063        public SetterObject() {
064            choices = new ArrayList<String>();
065        }
066
067        @SuppressWarnings("unchecked")
068        @DataBoundSetter
069        public void setChoices(Object choices) {
070            if (choices instanceof String) {
071                for (String choice : ((String) choices).split("\n")) {
072                    this.choices.add(choice);
073                }
074            } else {
075                this.choices = (List<String>) choices;
076            }
077        }
078
079        public List<String> getChoices() {
080            return choices;
081        }
082
083    }

085    @Issue("JENKINS-61438")
086    @Test
087    public void verify_JSON_bind_work_with_setter_that_accept_object_type() throws Exception {
088        final Stapler stapler = new Stapler();
089        stapler.setWebApp(new WebApp(Mockito.mock(ServletContext.class)));
090        HttpServletRequest request = Mockito.mock(HttpServletRequest.class);
091        RequestImpl req = new RequestImpl(stapler, request, Collections.<AncestorImpl>emptyList(), null);
092
093        JSONObject json = new JSONObject();
094        json.put("$class", SetterObject.class.getName());
095        json.put("choices", "1\n2\n3");
096
097        SetterObject o = req.bindJSON(SetterObject.class, json);
098        Assert.assertEquals(o.getChoices(), Arrays.asList("1", "2", "3"));
099    }

101    @Test
102    public void test_multipart_formdata() throws IOException, ServletException {
103        final Stapler stapler = new Stapler();
104        final byte[] buf = generateMultipartData();
105        final ByteArrayInputStream is = new ByteArrayInputStream(buf);
106        final MockRequest mockRequest = new MockRequest() {
107            @Override
108            public String getContentType() {
109                return "multipart/form-data; boundary=mpboundary";
110            }
111            @Override
112            public String getCharacterEncoding() {
113                return "UTF-8";
114            }
115            @Override
116            public String getHeader(String name) {
117                // FILEUPLOAD-195/FILEUPLOAD-228: ignore FileUploadBase.CONTENT_LENGTH
118                return null;
119            }
120            @Override
121            public int getContentLength() {
122                return buf.length;
123            }
124
125            @Override
126            public Enumeration getParameterNames() {
127                return Collections.enumeration(Arrays.asList("p1"));
128            }
129
130            @Override
131            public ServletInputStream getInputStream() throws IOException {
132                return new ServletInputStream() {
133                    @Override
134                    public int read() throws IOException {
135                        return is.read();
136                    }
137                    @Override
138                    public int read(byte[] b) throws IOException {
139                        return is.read(b);
140                    }
141                    @Override
142                    public int read(byte[] b, int off, int len) throws IOException {
143                        return is.read(b, off, len);
144                    }
145                    @Override
146                    public boolean isFinished() {
147                        return is.available() != 0;
148                    }
149                    @Override
150                    public boolean isReady() {
151                        return true;
152                    }
153                    @Override
154                    public void setReadListener(ReadListener readListener) {
155                        // ignored
156                    }
157                    
158                };
159            }
160        };
161        
162        RequestImpl request = new RequestImpl(stapler, mockRequest, Collections.<AncestorImpl>emptyList(), null);
163
164        // Check that we can get the Form Fields. See https://github.com/stapler/stapler/issues/52
165        Assert.assertEquals("text1_val", request.getParameter("text1"));
166        Assert.assertEquals("text2_val", request.getParameter("text2"));
167        
168        // Check that we can get the file
169        FileItem fileItem = request.getFileItem("pomFile");
170        Assert.assertNotNull(fileItem);
171        
172        // Check getParameterValues
173        Assert.assertEquals("text1_val", request.getParameterValues("text1")[0]);
174        
175        // Check getParameterNames
176        Assert.assertTrue(Collections.list(request.getParameterNames()).contains("p1"));
177        Assert.assertTrue(Collections.list(request.getParameterNames()).contains("text1"));
178        
179        // Check getParameterMap
180        Assert.assertTrue(request.getParameterMap().containsKey("text1"));        
181    }
182
183    private byte[] generateMultipartData() throws IOException {
184        MultipartEntityBuilder reqEntityBuilder = MultipartEntityBuilder.create();
185
186        reqEntityBuilder.setBoundary("mpboundary");
187        reqEntityBuilder.addBinaryBody("pomFile", new File("./pom.xml"), ContentType.TEXT_XML, "pom.xml");
188        reqEntityBuilder.addTextBody("text1", "text1_val");
189        reqEntityBuilder.addTextBody("text2", "text2_val");
190
191        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
192        try {
193            reqEntityBuilder.build().writeTo(outputStream);
194            outputStream.flush();
195            return outputStream.toByteArray();
196        } finally {
197            outputStream.close();
198        }
199    }
